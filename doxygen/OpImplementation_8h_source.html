98  /// Print an optional arrow followed by a type list.
99  void printOptionalArrowTypeList(ArrayRef<Type> types) {
100  if (types.empty())
101  return;
102  auto &os = getStream() << " -> ";
103  bool wrapped = types.size() != 1 || types[0].isa<FunctionType>();
104  if (wrapped)
105  os << '(';
106  interleaveComma(types, *this);
107  if (wrapped)
108  os << ')';
109  }

111  /// Print the complete type of an operation in functional form.
112  void printFunctionalType(Operation *op) {
113  auto &os = getStream();
114  os << "(";
115  interleaveComma(op->getNonSuccessorOperands(), os, [&](Value operand) {
116  if (operand)
117  printType(operand.getType());
118  else
119  os << "<<NULL>";
120  });
121  os << ") -> ";
122  if (op->getNumResults() == 1 &&
123  !op->getResult(0).getType().isa<FunctionType>()) {
124  printType(op->getResult(0).getType());
125  } else {
126  os << '(';
127  interleaveComma(op->getResultTypes(), os);
128  os << ')';
129  }
130  }

203 /// The OpAsmParser has methods for interacting with the asm parser: parsing
204 /// things from it, emitting errors etc. It has an intentionally high-level API
205 /// that is designed to reduce/constrain syntax innovation in individual
206 /// operations.
207 ///
208 /// For example, consider an op like this:
209 ///
210 /// %x = load %p[%1, %2] : memref<...>

212 /// The "%x = load" tokens are already parsed and therefore invisible to the
213 /// custom op parser. This can be supported by calling `parseOperandList` to
