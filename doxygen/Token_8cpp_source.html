67 /// Given a token containing a string literal, return its value, including
68 /// removing the quote characters and unescaping the contents of the string. The
69 /// lexer has already verified that this token is valid.
70 std::string Token::getStringValue() const {
71  assert(getKind() == string ||
72  (getKind() == at_identifier && getSpelling()[1] == '"'));
73  // Start by dropping the quotes.
74  StringRef bytes = getSpelling().drop_front().drop_back();
75  if (getKind() == at_identifier)
76  bytes = bytes.drop_front();
77 
78  std::string result;
79  result.reserve(bytes.size());
80  for (unsigned i = 0, e = bytes.size(); i != e;) {
81  auto c = bytes[i++];
82  if (c != '\\') {
83  result.push_back(c);
84  continue;
85  }
86 
87  assert(i + 1 <= e && "invalid string should be caught by lexer");
88  auto c1 = bytes[i++];
89  switch (c1) {
90  case '"':
91  case '\\':
92  result.push_back(c1);
93  continue;
94  case 'n':
95  result.push_back('\n');
96  continue;
97  case 't':
98  result.push_back('\t');
99  continue;
100  default:
101  break;
102  }
103 
104  assert(i + 1 <= e && "invalid string should be caught by lexer");
105  auto c2 = bytes[i++];
106 
107  assert(llvm::isHexDigit(c1) && llvm::isHexDigit(c2) && "invalid escape");
108  result.push_back((llvm::hexDigitValue(c1) << 4) | llvm::hexDigitValue(c2));
109  }
110 
111  return result;
112 }
