9 // This file implements the verify() methods on the various IR types, performing
10 // (potentially expensive) checks on the holistic structure of the code. This
11 // can be used for detecting bugs in compiler transformations and hand written
12 // .mlir files.
13 //
14 // The checks in this file are only for things that can occur as part of IR
15 // transformations: e.g. violation of dominance information, malformed operation
16 // attributes, etc. MLIR supports transformations moving IR through locally
17 // invalid states (e.g. unlinking an operation from a block before re-inserting
18 // it in a new place), but each transformation must complete with the IR in a
19 // valid form.
20 //
21 // This should not check for things that are always wrong by construction (e.g.
22 // attributes or other immutable structures that are incorrect), because those
23 // are not mutable and can be checked at time of construction.
24 //
25 //===----------------------------------------------------------------------===//

94 /// Verify the given operation.
95 LogicalResult OperationVerifier::verify(Operation &op) {
96  // Verify the operation first.
97  if (failed(verifyOperation(op)))
98  return failure();
99 
100  // Since everything looks structurally ok to this point, we do a dominance
101  // check for any nested regions. We do this as a second pass since malformed
102  // CFG's can cause dominator analysis constructure to crash and we want the
103  // verifier to be resilient to malformed code.
104  DominanceInfo theDomInfo(&op);
105  domInfo = &theDomInfo;
106  for (auto &region : op.getRegions())
107  if (failed(verifyDominance(region)))
108  return failure();
109 
110  domInfo = nullptr;
111  return success();
112 }

114 LogicalResult OperationVerifier::verifyRegion(Region &region) {
115  if (region.empty())
116  return success();
117 
118  // Verify the first block has no predecessors.
119  auto *firstBB = &region.front();
120  if (!firstBB->hasNoPredecessors())
121  return mlir::emitError(region.getLoc(),
122  "entry block of region may not have predecessors");
123 
124  // Verify each of the blocks within the region.
125  for (auto &block : region)
126  if (failed(verifyBlock(block)))
127  return failure();
128  return success();
129 }

131 LogicalResult OperationVerifier::verifyBlock(Block &block) {
132  for (auto arg : block.getArguments())
133  if (arg.getOwner() != &block)
134  return emitError(block, "block argument not owned by block");
135 
136  // Verify that this block has a terminator.
137  if (block.empty())
138  return emitError(block, "block with no terminator");
139 
140  // Verify the non-terminator operations separately so that we can verify
141  // they has no successors.
142  for (auto &op : llvm::make_range(block.begin(), std::prev(block.end()))) {
143  if (op.getNumSuccessors() != 0)
144  return op.emitError(
145  "operation with block successors must terminate its parent block");
146 
147  if (failed(verifyOperation(op)))
148  return failure();
149  }
150 
151  // Verify the terminator.
152  if (failed(verifyOperation(block.back())))
153  return failure();
154  if (block.back().isKnownNonTerminator())
155  return emitError(block, "block with no terminator");
156 
157  // Verify that this block is not branching to a block of a different
158  // region.
159  for (Block *successor : block.getSuccessors())
160  if (successor->getParent() != block.getParent())
161  return block.back().emitOpError(
162  "branching to block of a different region");
163 
164  return success();
165 }

167 LogicalResult OperationVerifier::verifyOperation(Operation &op) {
168  // Check that operands are non-nil and structurally ok.
169  for (auto operand : op.getOperands())
170  if (!operand)
171  return op.emitError("null operand found");
172 
173  /// Verify that all of the attributes are okay.
174  for (auto attr : op.getAttrs()) {
175  if (!identifierRegex.match(attr.first))
176  return op.emitError("invalid attribute name '") << attr.first << "'";
177 
178  // Check for any optional dialect specific attributes.
179  if (!attr.first.strref().contains('.'))
180  continue;
181  if (auto *dialect = getDialectForAttribute(attr))
182  if (failed(dialect->verifyOperationAttribute(&op, attr)))
183  return failure();
184  }
185 
186  // If we can get operation info for this, check the custom hook.
187  auto *opInfo = op.getAbstractOperation();
188  if (opInfo && failed(opInfo->verifyInvariants(&op)))
189  return failure();
190 
191  // Verify that all child regions are ok.
192  for (auto &region : op.getRegions())
193  if (failed(verifyRegion(region)))
194  return failure();
195 
196  // If this is a registered operation, there is nothing left to do.
197  if (opInfo)
198  return success();
199 
200  // Otherwise, verify that the parent dialect allows un-registered operations.
201  auto dialectPrefix = op.getName().getDialect();
202 
203  // Check for an existing answer for the operation dialect.
204  auto it = dialectAllowsUnknownOps.find(dialectPrefix);
205  if (it == dialectAllowsUnknownOps.end()) {
206  // If the operation dialect is registered, query it directly.
207  if (auto *dialect = ctx->getRegisteredDialect(dialectPrefix))
208  it = dialectAllowsUnknownOps
209  .try_emplace(dialectPrefix, dialect->allowsUnknownOperations())
210  .first;
211  // Otherwise, conservatively allow unknown operations.
212  else
213  it = dialectAllowsUnknownOps.try_emplace(dialectPrefix, true).first;
214  }
215 
216  if (!it->second) {
217  return op.emitError("unregistered operation '")
218  << op.getName() << "' found in dialect ('" << dialectPrefix
219  << "') that does not allow unknown operations";
220  }
221 
222  return success();
223 }

225 LogicalResult OperationVerifier::verifyDominance(Region &region) {
226  // Verify the dominance of each of the held operations.
227  for (auto &block : region)
228  for (auto &op : block)
229  if (failed(verifyDominance(op)))
230  return failure();
231  return success();
232 }

234 LogicalResult OperationVerifier::verifyDominance(Operation &op) {
235  // Check that operands properly dominate this use.
236  for (unsigned operandNo = 0, e = op.getNumOperands(); operandNo != e;
237  ++operandNo) {
238  auto operand = op.getOperand(operandNo);
239  if (domInfo->properlyDominates(operand, &op))
240  continue;
241 
242  auto diag = op.emitError("operand #")
243  << operandNo << " does not dominate this use";
244  if (auto *useOp = operand.getDefiningOp())
245  diag.attachNote(useOp->getLoc()) << "operand defined here";
246  return failure();
247  }
248 
249  // Verify the dominance of each of the nested blocks within this operation.
250  for (auto &region : op.getRegions())
251  if (failed(verifyDominance(region)))
252  return failure();
253 
254  return success();
255 }
