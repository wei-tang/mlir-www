38 /// Structure to group information about a passes and pass pipelines (argument
39 /// to invoke via mlir-opt, description, pass pipeline builder).
40 class PassRegistryEntry {
41 public:
42  /// Adds this pass registry entry to the given pass manager. `options` is
43  /// an opaque string that will be parsed by the builder. The success of
44  /// parsing will be returned.
45  LogicalResult addToPipeline(OpPassManager &pm, StringRef options) const {
46  assert(builder &&
47  "cannot call addToPipeline on PassRegistryEntry without builder");

105 /// PassRegistration provides a global initializer that registers a Pass
106 /// allocation routine for a concrete pass instance. The third argument is
107 /// optional and provides a callback to construct a pass that does not have
108 /// a default constructor.
109 ///
110 /// Usage:
111 ///
112 /// /// At namespace scope.
113 /// static PassRegistration<MyPass> reg("my-pass", "My Pass Description.");
114 ///
115 template <typename ConcretePass> struct PassRegistration {
116 
117  PassRegistration(StringRef arg, StringRef description,
118  const PassAllocatorFunction &constructor) {
119  registerPass(arg, description, PassID::getID<ConcretePass>(), constructor);
120  }
121 
122  PassRegistration(StringRef arg, StringRef description)
123  : PassRegistration(arg, description,
124  [] { return std::make_unique<ConcretePass>(); }) {}
125 };

127 /// PassPipelineRegistration provides a global initializer that registers a Pass
128 /// pipeline builder routine.
129 ///
130 /// Usage:
131 ///
132 /// // At namespace scope.
133 /// void pipelineBuilder(OpPassManager &pm) {
134 /// pm.addPass(new MyPass());
135 /// pm.addPass(new MyOtherPass());
136 /// }
137 ///
138 /// static PassPipelineRegistration Unused("unused", "Unused pass",
139 /// pipelineBuilder);
140 template <typename Options = EmptyPipelineOptions>
141 struct PassPipelineRegistration {
142  PassPipelineRegistration(
143  StringRef arg, StringRef description,
144  std::function<void(OpPassManager &, const Options &options)> builder) {
145  registerPassPipeline(arg, description,
146  [builder](OpPassManager &pm, StringRef optionsStr) {
147  Options options;
148  if (failed(options.parseFromString(optionsStr)))
149  return failure();
150  builder(pm, options);
151  return success();
152  });
153  }
154 };
155 
156 /// Convenience specialization of PassPipelineRegistration for EmptyPassOptions
157 /// that does not pass an empty options struct to the pass builder function.
158 template <> struct PassPipelineRegistration<EmptyPipelineOptions> {
159  PassPipelineRegistration(StringRef arg, StringRef description,
160  std::function<void(OpPassManager &)> builder) {
161  registerPassPipeline(arg, description,
162  [builder](OpPassManager &pm, StringRef optionsStr) {
163  if (!optionsStr.empty())
164  return failure();
165  builder(pm);
166  return success();
167  });
168  }
