261 template <typename BarrierOp>
262 static LogicalResult verifyMemorySemantics(BarrierOp op) {
263  // According to the SPIR-V specification:
264  // "Despite being a mask and allowing multiple bits to be combined, it is
265  // invalid for more than one of these four bits to be set: Acquire, Release,
266  // AcquireRelease, or SequentiallyConsistent. Requesting both Acquire and
267  // Release semantics is done by setting the AcquireRelease bit, not by setting
268  // two bits."
269  auto memorySemantics = op.memory_semantics();
270  auto atMostOneInSet = spirv::MemorySemantics::Acquire |
271  spirv::MemorySemantics::Release |
272  spirv::MemorySemantics::AcquireRelease |
273  spirv::MemorySemantics::SequentiallyConsistent;
274 
275  auto bitCount = llvm::countPopulation(
276  static_cast<uint32_t>(memorySemantics & atMostOneInSet));
277  if (bitCount > 1) {
278  return op.emitError("expected at most one of these four memory constraints "
279  "to be set: `Acquire`, `Release`,"
280  "`AcquireRelease` or `SequentiallyConsistent`");
281  }
282  return success();
283 }

302  auto builtInName =
303  convertToSnakeCase(stringifyDecoration(spirv::Decoration::BuiltIn));
304  if (succeeded(parser.parseOptionalKeyword("bind"))) {
305  Attribute set, binding;

311  Type i32Type = parser.getBuilder().getIntegerType(32);

332  return failure();

364 // Extracts an element from the given `composite` by following the given
365 // `indices`. Returns a null Attribute if error happens.
366 static Attribute extractCompositeElement(Attribute composite,
367  ArrayRef<unsigned> indices) {
368  // Check that given composite is a constant.
369  if (!composite)
370  return {};
371  // Return composite itself if we reach the end of the index chain.
372  if (indices.empty())
373  return composite;
374 
375  if (auto vector = composite.dyn_cast<ElementsAttr>()) {
376  assert(indices.size() == 1 && "must have exactly one index for a vector");
377  return vector.getValue({indices[0]});
378  }
379 
380  if (auto array = composite.dyn_cast<ArrayAttr>()) {
381  assert(!indices.empty() && "must have at least one index for an array");
382  return extractCompositeElement(array.getValue()[indices[0]],
383  indices.drop_front());
384  }
385 
386  return {};
387 }

407 /// Walks the given type hierarchy with the given indices, potentially down
408 /// to component granularity, to select an element type. Returns null type and
409 /// emits errors with the given loc on failure.
410 static Type
411 getElementType(Type type, ArrayRef<int32_t> indices,
412  function_ref<InFlightDiagnostic(StringRef)> emitErrorFn) {
413  if (indices.empty()) {
414  emitErrorFn("expected at least one index for spv.CompositeExtract");
415  return nullptr;
416  }
417 
418  for (auto index : indices) {
419  if (auto cType = type.dyn_cast<spirv::CompositeType>()) {
420  if (index < 0 || static_cast<uint64_t>(index) >= cType.getNumElements()) {
421  emitErrorFn("index ") << index << " out of bounds for " << type;
422  return nullptr;
423  }
424  type = cType.getElementType(index);
425  } else {
426  emitErrorFn("cannot extract from non-composite type ")
427  << type << " with index " << index;
428  return nullptr;
429  }
430  }
431  return type;
432 }

696 static Type getElementPtrType(Type type, ValueRange indices, Location baseLoc) {
697  if (indices.empty()) {
698  emitError(baseLoc, "'spv.AccessChain' op expected at least "
699  "one index ");
700  return nullptr;
701  }
702 
703  auto ptrType = type.dyn_cast<spirv::PointerType>();
704  if (!ptrType) {
705  emitError(baseLoc, "'spv.AccessChain' op expected a pointer "
706  "to composite type, but provided ")
707  << type;
708  return nullptr;
709  }
710 
711  auto resultType = ptrType.getPointeeType();
712  auto resultStorageClass = ptrType.getStorageClass();
713  int32_t index = 0;
714 
715  for (auto indexSSA : indices) {
716  auto cType = resultType.dyn_cast<spirv::CompositeType>();
717  if (!cType) {
718  emitError(baseLoc,
719  "'spv.AccessChain' op cannot extract from non-composite type ")
720  << resultType << " with index " << index;
721  return nullptr;
722  }
723  index = 0;
724  if (resultType.isa<spirv::StructType>()) {
725  Operation *op = indexSSA.getDefiningOp();
726  if (!op) {
727  emitError(baseLoc, "'spv.AccessChain' op index must be an "
728  "integer spv.constant to access "
729  "element of spv.struct");
730  return nullptr;
731  }
732 
733  // TODO(denis0x0D): this should be relaxed to allow
734  // integer literals of other bitwidths.
735  if (failed(extractValueFromConstOp(op, index))) {
736  emitError(baseLoc,
737  "'spv.AccessChain' index must be an integer spv.constant to "
738  "access element of spv.struct, but provided ")
739  << op->getName();
740  return nullptr;
741  }
742  if (index < 0 || static_cast<uint64_t>(index) >= cType.getNumElements()) {
743  emitError(baseLoc, "'spv.AccessChain' op index ")
744  << index << " out of bounds for " << resultType;
745  return nullptr;
746  }
747  }
748  resultType = cType.getElementType(index);
749  }

1396 static LogicalResult verify(spirv::ConstantOp constOp) {
1397  auto opType = constOp.getType();
1398  auto value = constOp.value();
1399  auto valueType = value.getType();
1400 
1401  // ODS already generates checks to make sure the result type is valid. We just
1402  // need to additionally check that the value's attribute type is consistent
1403  // with the result type.
1404  switch (value.getKind()) {
1405  case StandardAttributes::Bool:
1406  case StandardAttributes::Integer:
1407  case StandardAttributes::Float: {
1408  if (valueType != opType)
1409  return constOp.emitOpError("result type (")
1410  << opType << ") does not match value type (" << valueType << ")";
1411  return success();
1412  } break;
1413  case StandardAttributes::DenseElements:
1414  case StandardAttributes::SparseElements: {
1415  if (valueType == opType)
1416  break;
1417  auto arrayType = opType.dyn_cast<spirv::ArrayType>();
1418  auto shapedType = valueType.dyn_cast<ShapedType>();
1419  if (!arrayType) {
1420  return constOp.emitOpError(
1421  "must have spv.array result type for array value");
1422  }
1423 
1424  int numElements = arrayType.getNumElements();
1425  auto opElemType = arrayType.getElementType();
1426  while (auto t = opElemType.dyn_cast<spirv::ArrayType>()) {
1427  numElements *= t.getNumElements();
1428  opElemType = t.getElementType();
1429  }
1430  if (!opElemType.isIntOrFloat()) {
1431  return constOp.emitOpError("only support nested array result type");
1432  }
1433 
1434  auto valueElemType = shapedType.getElementType();
1435  if (valueElemType != opElemType) {
1436  return constOp.emitOpError("result element type (")
1437  << opElemType << ") does not match value element type ("
1438  << valueElemType << ")";
1439  }
1440 
1441  if (numElements != shapedType.getNumElements()) {
1442  return constOp.emitOpError("result number of elements (")
1443  << numElements << ") does not match value number of elements ("
1444  << shapedType.getNumElements() << ")";
1445  }
1446  } break;
1447  case StandardAttributes::Array: {
1448  auto arrayType = opType.dyn_cast<spirv::ArrayType>();
1449  if (!arrayType)
1450  return constOp.emitOpError(
1451  "must have spv.array result type for array value");
1452  auto elemType = arrayType.getElementType();
1453  for (auto element : value.cast<ArrayAttr>().getValue()) {
1454  if (element.getType() != elemType)
1455  return constOp.emitOpError("has array element whose type (")
1456  << element.getType()
1457  << ") does not match the result element type (" << elemType
1458  << ')';
1459  }
1460  } break;
1461  default:
1462  return constOp.emitOpError("cannot have value of type ") << valueType;
1463  }
1464 
1465  return success();
1466 }

2192  auto *op = loopOp.getOperation();
2193 
2194  // We need to verify that the blocks follow the following layout:
2195  //
2196  // +-------------+
2197  // | entry block |
2198  // +-------------+
2199  // |
2200  // v
2201  // +-------------+
2202  // | loop header | <-----+
2203  // +-------------+ |
2204  // |
2205  // ... |
2206  // \ | / |

2226  Block &merge = region.back();

2774 // +------------------------+ +------------------------+
2775 //
2776 //

2778 // \ /

2788  PatternMatchResult matchAndRewrite(spirv::SelectionOp selectionOp,
2789  PatternRewriter &rewriter) const override {
2790  auto *op = selectionOp.getOperation();
2791  auto &body = op->getRegion(0);
2792  // Verifier allows an empty region for `spv.selection`.
2793  if (body.empty()) {
2794  return matchFailure();
2795  }
2796 
2797  // Check that region consists of 4 blocks:
2798  // header block, `true` block, `false` block and merge block.
2799  if (std::distance(body.begin(), body.end()) != 4) {
2800  return matchFailure();
2801  }
2802 
2803  auto *headerBlock = selectionOp.getHeaderBlock();
2804  if (!onlyContainsBranchConditionalOp(headerBlock)) {
2805  return matchFailure();
2806  }
2807 
2808  auto brConditionalOp =
2809  cast<spirv::BranchConditionalOp>(headerBlock->front());
2810 
2811  auto *trueBlock = brConditionalOp.getSuccessor(0);
2812  auto *falseBlock = brConditionalOp.getSuccessor(1);
2813  auto *mergeBlock = selectionOp.getMergeBlock();
2814 
2815  if (!canCanonicalizeSelection(trueBlock, falseBlock, mergeBlock)) {
2816  return matchFailure();
2817  }
2818 
2819  auto trueValue = getSrcValue(trueBlock);
2820  auto falseValue = getSrcValue(falseBlock);
2821  auto ptrValue = getDstPtr(trueBlock);
2822  auto storeOpAttributes =
2823  cast<spirv::StoreOp>(trueBlock->front()).getOperation()->getAttrs();

2826  selectionOp.getLoc(), trueValue.getType(), brConditionalOp.condition(),

2828  rewriter.create<spirv::StoreOp>(selectOp.getLoc(), ptrValue,
2829  selectOp.getResult(), storeOpAttributes);

2832  rewriter.eraseOp(op);

2880 PatternMatchResult ConvertSelectionOpToSelect::canCanonicalizeSelection(
2881  Block *trueBlock, Block *falseBlock, Block *mergeBlock) const {
2882  // Each block must consists of 2 operations.
2883  if ((std::distance(trueBlock->begin(), trueBlock->end()) != 2) ||
2884  (std::distance(falseBlock->begin(), falseBlock->end()) != 2)) {
2885  return matchFailure();

2889  auto trueBrBranchOp =
2890  dyn_cast<spirv::BranchOp>(*std::next(trueBlock->begin()));

2892  auto falseBrBranchOp =
2893  dyn_cast<spirv::BranchOp>(*std::next(falseBlock->begin()));

2896  !falseBrBranchOp) {
2897  return matchFailure();
2898  }

2905  return matchFailure();

2908  if ((trueBrBranchOp.getOperation()->getSuccessor(0) != mergeBlock) ||
2909  (falseBrBranchOp.getOperation()->getSuccessor(0) != mergeBlock)) {
2910  return matchFailure();
2911  }

2914 }
