32 /// A multi-dimensional affine map
33 /// Affine map's are immutable like Type's, and they are uniqued.
34 /// Eg: (d0, d1) -> (d0/128, d0 mod 128, d1)
35 /// The names used (d0, d1) don't matter - it's the mathematical function that

118  /// Returns the AffineMap resulting from composing `this` with `map`.
119  /// The resulting AffineMap has as many AffineDimExpr as `map` and as many
120  /// AffineSymbolExpr as the concatenation of `this` and `map` (in which case
121  /// the symbols of `this` map come first).
122  ///
123  /// Prerequisites:
124  /// The maps are composable, i.e. that the number of AffineDimExpr of `this`
125  /// matches the number of results of `map`.
126  ///
127  /// Example:
128  /// map1: `(d0, d1)[s0, s1] -> (d0 + 1 + s1, d1 - 1 - s0)`
129  /// map2: `(d0)[s0] -> (d0 + s0, d0 - s0)`
130  /// map1.compose(map2):
131  /// `(d0)[s0, s1, s2] -> (d0 + s1 + s2 + 1, d0 - s0 - s2 - 1)`
132  AffineMap compose(AffineMap map);

161 /// Returns a map of codomain to domain dimensions such that the first codomain
162 /// dimension for a particular domain dimension is selected.
163 /// Returns an empty map if the input map is empty or if `map` is not invertible
164 /// (i.e. `map` does not contain a subset that is a permutation of full domain
165 /// rank).
166 ///
167 /// Prerequisites:
168 /// 1. `map` has no symbols.
169 ///
170 /// Example 1:
171 ///
172 /// ```mlir
173 /// (d0, d1, d2) -> (d1, d1, d0, d2, d1, d2, d1, d0)
174 /// 0 2 3
175 /// ```
176 ///
177 /// returns:
178 ///
179 /// ```mlir
180 /// (d0, d1, d2, d3, d4, d5, d6, d7) -> (d2, d0, d3)
181 /// ```
182 ///
183 /// Example 2:
184 ///
185 /// ```mlir
186 /// (d0, d1, d2) -> (d1, d0 + d1, d0, d2, d1, d2, d1, d0)
187 /// 0 2 3
188 /// ```
189 ///
190 /// returns:
191 ///
192 /// ```mlir
193 /// (d0, d1, d2, d3, d4, d5, d6, d7) -> (d2, d0, d3)
194 /// ```
195 AffineMap inversePermutation(AffineMap map);

197 /// Concatenates a list of `maps` into a single AffineMap, stepping over
198 /// potentially empty maps. Assumes each of the underlying map has 0 symbols.
199 /// The resulting map has a number of dims equal to the max of `maps`' dims and
200 /// the concatenated results as its results.
201 /// Returns an empty map if all input `maps` are empty.
202 ///
203 /// Example:
204 /// When applied to the following list of 3 affine maps,
205 ///

210 /// (i, j, k) -> (i, j)
211 /// }
212 /// ```
213 ///
214 /// Returns the map:
215 ///
216 /// ```mlir
217 /// (i, j, k) -> (i, k, k, j, i, j)
218 /// ```
219 AffineMap concatAffineMaps(ArrayRef<AffineMap> maps);
