32 // The store to load forwarding relies on three conditions:
33 //
34 // 1) they need to have mathematically equivalent affine access functions
35 // (checked after full composition of load/store operands); this implies that
36 // they access the same single memref element for all iterations of the common
37 // surrounding loop,
38 //
39 // 2) the store op should dominate the load op,
40 //
41 // 3) among all op's that satisfy both (1) and (2), the one that postdominates
42 // all store op's that have a dependence into the load, is provably the last
43 // writer to the particular memref location being loaded at the load op, and its
44 // store value can be forwarded to the load. Note that the only dependences
45 // that are to be considered are those that are satisfied at the block* of the
46 // innermost common surrounding loop of the <store, load> being considered.
47 //
48 // (* A dependence being satisfied at a block: a dependence that is satisfied by
49 // virtue of the destination operation appearing textually / lexically after
50 // the source operation within the body of a 'affine.for' operation; thus, a
51 // dependence is always either satisfied by a loop or by a block).
52 //
53 // The above conditions are simple to check, sufficient, and powerful for most
54 // cases in practice - they are sufficient, but not necessary --- since they
55 // don't reason about loops that are guaranteed to execute at least once or
56 // multiple sources to forward from.
57 //
58 // TODO(mlir-team): more forwarding can be done when support for
59 // loop/conditional live-out SSA values is available.
60 // TODO(mlir-team): do general dead store elimination for memref's. This pass
61 // currently only eliminates the stores only if no other loads/uses (other
62 // than dealloc) remain.
63 //
64 struct MemRefDataFlowOpt : public FunctionPass<MemRefDataFlowOpt> {
65  void runOnFunction() override;

67  void forwardStoreToLoad(AffineLoadOp loadOp);

69  // A list of memref's that are potentially dead / could be eliminated.
70  SmallPtrSet<Value, 4> memrefsToErase;
71  // Load op's whose results were replaced by those forwarded from stores.
72  SmallVector<Operation *, 8> loadOpsToErase;
73 
74  DominanceInfo *domInfo = nullptr;
75  PostDominanceInfo *postDomInfo = nullptr;
76 };

78 } // end anonymous namespace

86 // This is a straightforward implementation not optimized for speed. Optimize
87 // if needed.
88 void MemRefDataFlowOpt::forwardStoreToLoad(AffineLoadOp loadOp) {
89  Operation *loadOpInst = loadOp.getOperation();
90 
91  // First pass over the use list to get minimum number of surrounding
92  // loops common between the load op and the store op, with min taken across
93  // all store ops.
94  SmallVector<Operation *, 8> storeOps;
95  unsigned minSurroundingLoops = getNestingDepth(*loadOpInst);
96  for (auto *user : loadOp.getMemRef().getUsers()) {
97  auto storeOp = dyn_cast<AffineStoreOp>(user);
98  if (!storeOp)
99  continue;
100  auto *storeOpInst = storeOp.getOperation();
101  unsigned nsLoops = getNumCommonSurroundingLoops(*loadOpInst, *storeOpInst);
102  minSurroundingLoops = std::min(nsLoops, minSurroundingLoops);
103  storeOps.push_back(storeOpInst);
104  }
105 
106  // The list of store op candidates for forwarding that satisfy conditions
107  // (1) and (2) above - they will be filtered later when checking (3).
108  SmallVector<Operation *, 8> fwdingCandidates;
109 
110  // Store ops that have a dependence into the load (even if they aren't
111  // forwarding candidates). Each forwarding candidate will be checked for a
112  // post-dominance on these. 'fwdingCandidates' are a subset of depSrcStores.
113  SmallVector<Operation *, 8> depSrcStores;
114 
115  for (auto *storeOpInst : storeOps) {
116  MemRefAccess srcAccess(storeOpInst);
117  MemRefAccess destAccess(loadOpInst);
118  // Find stores that may be reaching the load.
119  FlatAffineConstraints dependenceConstraints;
120  unsigned nsLoops = getNumCommonSurroundingLoops(*loadOpInst, *storeOpInst);
121  unsigned d;
122  // Dependences at loop depth <= minSurroundingLoops do NOT matter.
123  for (d = nsLoops + 1; d > minSurroundingLoops; d--) {
124  DependenceResult result = checkMemrefAccessDependence(
125  srcAccess, destAccess, d, &dependenceConstraints,
126  /*dependenceComponents=*/nullptr);
127  if (hasDependence(result))
128  break;
129  }
130  if (d == minSurroundingLoops)
131  continue;
132 
133  // Stores that *may* be reaching the load.
134  depSrcStores.push_back(storeOpInst);
135 
136  // 1. Check if the store and the load have mathematically equivalent
137  // affine access functions; this implies that they statically refer to the
138  // same single memref element. As an example this filters out cases like:
139  // store %A[%i0 + 1]
140  // load %A[%i0]
141  // store %A[%M]
142  // load %A[%N]
143  // Use the AffineValueMap difference based memref access equality checking.
144  if (srcAccess != destAccess)
145  continue;
146 
147  // 2. The store has to dominate the load op to be candidate.
148  if (!domInfo->dominates(storeOpInst, loadOpInst))
149  continue;
150 
151  // We now have a candidate for forwarding.
152  fwdingCandidates.push_back(storeOpInst);
153  }
154 
155  // 3. Of all the store op's that meet the above criteria, the store that
156  // postdominates all 'depSrcStores' (if one exists) is the unique store
157  // providing the value to the load, i.e., provably the last writer to that
158  // memref loc.
159  // Note: this can be implemented in a cleaner way with postdominator tree
160  // traversals. Consider this for the future if needed.
161  Operation *lastWriteStoreOp = nullptr;
162  for (auto *storeOpInst : fwdingCandidates) {
163  if (llvm::all_of(depSrcStores, [&](Operation *depStore) {
164  return postDomInfo->postDominates(storeOpInst, depStore);
165  })) {
166  lastWriteStoreOp = storeOpInst;
167  break;
168  }
169  }
170  if (!lastWriteStoreOp)
171  return;
172 
173  // Perform the actual store to load forwarding.
174  Value storeVal = cast<AffineStoreOp>(lastWriteStoreOp).getValueToStore();
175  loadOp.replaceAllUsesWith(storeVal);
176  // Record the memref for a later sweep to optimize away.
177  memrefsToErase.insert(loadOp.getMemRef());
178  // Record this to erase later.
179  loadOpsToErase.push_back(loadOpInst);
180 }

182 void MemRefDataFlowOpt::runOnFunction() {
183  // Only supports single block functions at the moment.
184  FuncOp f = getFunction();
185  if (f.getBlocks().size() != 1) {
186  markAllAnalysesPreserved();
187  return;
188  }
189 
190  domInfo = &getAnalysis<DominanceInfo>();
191  postDomInfo = &getAnalysis<PostDominanceInfo>();
192 
193  loadOpsToErase.clear();
194  memrefsToErase.clear();
195 
196  // Walk all load's and perform load/store forwarding.
197  f.walk([&](AffineLoadOp loadOp) { forwardStoreToLoad(loadOp); });
198 
199  // Erase all load op's whose results were replaced with store fwd'ed ones.
200  for (auto *loadOp : loadOpsToErase) {
201  loadOp->erase();
202  }
203 
204  // Check if the store fwd'ed memrefs are now left with only stores and can
205  // thus be completely deleted. Note: the canonicalize pass should be able
206  // to do this as well, but we'll do it here since we collected these anyway.
207  for (auto memref : memrefsToErase) {
208  // If the memref hasn't been alloc'ed in this function, skip.
209  Operation *defInst = memref.getDefiningOp();
210  if (!defInst || !isa<AllocOp>(defInst))
211  // TODO(mlir-team): if the memref was returned by a 'call' operation, we
212  // could still erase it if the call had no side-effects.
213  continue;
214  if (llvm::any_of(memref.getUsers(), [&](Operation *ownerInst) {
215  return (!isa<AffineStoreOp>(ownerInst) && !isa<DeallocOp>(ownerInst));
216  }))
217  continue;
218 
219  // Erase all stores, the dealloc, and the alloc on the memref.
220  for (auto *user : llvm::make_early_inc_range(memref.getUsers()))
221  user->erase();
222  defInst->erase();
223  }
224 }
