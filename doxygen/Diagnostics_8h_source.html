317 /// This class represents a diagnostic that is inflight and set to be reported.
318 /// This allows for last minute modifications of the diagnostic before it is
319 /// emitted by a DiagnosticEngine.
320 class InFlightDiagnostic {
321 public:
322  InFlightDiagnostic() = default;
323  InFlightDiagnostic(InFlightDiagnostic &&rhs)
324  : owner(rhs.owner), impl(std::move(rhs.impl)) {
325  // Reset the rhs diagnostic.
326  rhs.impl.reset();
327  rhs.abandon();
328  }
329  ~InFlightDiagnostic() {
330  if (isInFlight())
331  report();
332  }
333 
334  /// Stream operator for new diagnostic arguments.
335  template <typename Arg> InFlightDiagnostic &operator<<(Arg &&arg) & {
336  return append(std::forward<Arg>(arg));
337  }
338  template <typename Arg> InFlightDiagnostic &&operator<<(Arg &&arg) && {
339  return std::move(append(std::forward<Arg>(arg)));
340  }
341 
342  /// Append arguments to the diagnostic.
343  template <typename... Args> InFlightDiagnostic &append(Args &&... args) & {
344  assert(isActive() && "diagnostic not active");
345  if (isInFlight())
346  impl->append(std::forward<Args>(args)...);
347  return *this;
348  }
349  template <typename... Args> InFlightDiagnostic &&append(Args &&... args) && {
350  return std::move(append(std::forward<Args>(args)...));
351  }
352 
353  /// Attaches a note to this diagnostic.
354  Diagnostic &attachNote(Optional<Location> noteLoc = llvm::None) {
355  assert(isActive() && "diagnostic not active");
356  return impl->attachNote(noteLoc);
357  }
358 
359  /// Reports the diagnostic to the engine.
360  void report();
361 
362  /// Abandons this diagnostic so that it will no longer be reported.
363  void abandon();
364 
365  /// Allow an inflight diagnostic to be converted to 'failure', otherwise
366  /// 'success' if this is an empty diagnostic.
367  operator LogicalResult() const;
368 
369 private:

395 /// This class is the main interface for diagnostics. The DiagnosticEngine
396 /// manages the registration of diagnostic handlers as well as the core API for
397 /// diagnostic emission. This class should not be constructed directly, but
398 /// instead interfaced with via an MLIRContext instance.
399 class DiagnosticEngine {
400 public:
401  ~DiagnosticEngine();
402 
403  // Diagnostic handler registration and use. MLIR supports the ability for the
404  // IR to carry arbitrary metadata about operation location information. If a
405  // problem is detected by the compiler, it can invoke the emitError /
406  // emitWarning / emitRemark method on an Operation and have it get reported
407  // through this interface.
408  //
409  // Tools using MLIR are encouraged to register error handlers and define a
410  // schema for their location information. If they don't, then warnings and
411  // notes will be dropped and errors will be emitted to errs.
412 
413  /// The handler type for MLIR diagnostics. This function takes a diagnostic as
414  /// input, and returns success if the handler has fully processed this
415  /// diagnostic. Returns failure otherwise.
416  using HandlerTy = std::function<LogicalResult(Diagnostic &)>;
417 
418  /// A handle to a specific registered handler object.
419  using HandlerID = uint64_t;
420 
421  /// Register a new handler for diagnostics to the engine. Diagnostics are
422  /// process by handlers in stack-like order, meaning that the last added
423  /// handlers will process diagnostics first. This function returns a unique
424  /// identifier for the registered handler, which can be used to unregister
425  /// this handler at a later time.
426  HandlerID registerHandler(const HandlerTy &handler);
427 
428  /// Set the diagnostic handler with a function that returns void. This is a
429  /// convenient wrapper for handlers that always completely process the given
430  /// diagnostic.
431  template <typename FuncTy, typename RetT = decltype(std::declval<FuncTy>()(
432  std::declval<Diagnostic &>()))>
433  std::enable_if_t<std::is_same<RetT, void>::value, HandlerID>
434  registerHandler(FuncTy &&handler) {
435  return registerHandler([=](Diagnostic &diag) {
436  handler(diag);
437  return success();
438  });
439  }
440 
441  /// Erase the registered diagnostic handler with the given identifier.
442  void eraseHandler(HandlerID id);
443 
444  /// Create a new inflight diagnostic with the given location and severity.
445  InFlightDiagnostic emit(Location loc, DiagnosticSeverity severity) {
446  assert(severity != DiagnosticSeverity::Note &&
447  "notes should not be emitted directly");
448  return InFlightDiagnostic(this, Diagnostic(loc, severity));
449  }
450 
451  /// Emit a diagnostic using the registered issue handler if present, or with
452  /// the default behavior if not.
453  void emit(Diagnostic diag);
454 
455 private:
456  friend class MLIRContextImpl;
457  DiagnosticEngine();
458 
459  /// The internal implementation of the DiagnosticEngine.
460  std::unique_ptr<detail::DiagnosticEngineImpl> impl;
461 };

478 /// the diagnostic arguments directly instead of relying on the returned

492 template <typename... Args>
493 LogicalResult emitOptionalRemark(Optional<Location> loc, Args &&... args) {
494  if (loc)
495  return emitRemark(*loc).append(std::forward<Args>(args)...);
496  return failure();
497 }

503 /// This diagnostic handler is a simple RAII class that registers and erases a
504 /// diagnostic handler on a given context. This class can be either be used
505 /// directly, or in conjunction with a derived diagnostic handler.
506 class ScopedDiagnosticHandler {
507 public:
508  explicit ScopedDiagnosticHandler(MLIRContext *ctx) : handlerID(0), ctx(ctx) {}
509  template <typename FuncTy>
510  ScopedDiagnosticHandler(MLIRContext *ctx, FuncTy &&handler)
511  : handlerID(0), ctx(ctx) {
512  setHandler(std::forward<FuncTy>(handler));
513  }
514  ~ScopedDiagnosticHandler();
515 
516 protected:
517  /// Set the handler to manage via RAII.
518  template <typename FuncTy> void setHandler(FuncTy &&handler) {
519  auto &diagEngine = ctx->getDiagEngine();
520  if (handlerID)
521  diagEngine.eraseHandler(handlerID);
522  handlerID = diagEngine.registerHandler(std::forward<FuncTy>(handler));
523  }
524 
525 private:
526  /// The unique id for the scoped handler.
527  DiagnosticEngine::HandlerID handlerID;
528 
529  /// The context to erase the handler from.
530  MLIRContext *ctx;
531 };

618 /// This class is a utility diagnostic handler for use when multi-threading some
619 /// part of the compiler where diagnostics may be emitted. This handler ensures
620 /// a deterministic ordering to the emitted diagnostics that mirrors that of a
621 /// single-threaded compilation.

624  ParallelDiagnosticHandler(MLIRContext *ctx);

627  /// Set the order id for the current thread. This is required to be set by
628  /// each thread that will be emitting diagnostics to this handler. The orderID
629  /// corresponds to the order in which diagnostics would be emitted when
630  /// executing synchronously. For example, if we were processing a list
631  /// of operations [a, b, c] on a single-thread. Diagnostics emitted while
632  /// processing operation 'a' would be emitted before those for 'b' or 'c'.
633  /// This corresponds 1-1 with the 'orderID'. The thread that is processing 'a'
634  /// should set the orderID to '0'; the thread processing 'b' should set it to
635  /// '1'; and so on and so forth. This provides a way for the handler to
636  /// deterministically order the diagnostics that it receives given the thread
637  /// that it is receiving on.
638  void setOrderIDForThread(size_t orderID);

641  /// diagnostics tracking.
