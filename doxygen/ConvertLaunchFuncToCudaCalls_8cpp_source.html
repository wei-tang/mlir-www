50 /// A pass to convert gpu.launch_func operations into a sequence of CUDA
51 /// runtime calls.
52 ///
53 /// In essence, a gpu.launch_func operations gets compiled into the following
54 /// sequence of runtime calls:
55 ///
56 /// * mcuModuleLoad -- loads the module given the cubin data
57 /// * mcuModuleGetFunction -- gets a handle to the actual kernel function
58 /// * mcuGetStreamHelper -- initializes a new CUDA stream
59 /// * mcuLaunchKernelName -- launches the kernel on a stream
60 /// * mcuStreamSynchronize -- waits for operations on the stream to finish
61 ///
62 /// Intermediate data structures are allocated on the stack.
63 class GpuLaunchFuncToCudaCallsPass

153 // Adds declarations for the needed helper functions from the CUDA wrapper.
154 // The types in comments give the actual types expected/returned but the API
155 // uses void pointers. This is fine as they have the same linkage in C.
156 void GpuLaunchFuncToCudaCallsPass::declareCudaFunctions(Location loc) {
157  ModuleOp module = getModule();
158  OpBuilder builder(module.getBody()->getTerminator());
159  if (!module.lookupSymbol(cuModuleLoadName)) {
160  builder.create<LLVM::LLVMFuncOp>(
161  loc, cuModuleLoadName,
162  LLVM::LLVMType::getFunctionTy(
163  getCUResultType(),
164  {
165  getPointerPointerType(), /* CUmodule *module */
166  getPointerType() /* void *cubin */
167  },
168  /*isVarArg=*/false));
169  }
170  if (!module.lookupSymbol(cuModuleGetFunctionName)) {
171  // The helper uses void* instead of CUDA's opaque CUmodule and
172  // CUfunction.
173  builder.create<LLVM::LLVMFuncOp>(
174  loc, cuModuleGetFunctionName,
175  LLVM::LLVMType::getFunctionTy(
176  getCUResultType(),
177  {
178  getPointerPointerType(), /* void **function */
179  getPointerType(), /* void *module */
180  getPointerType() /* char *name */
181  },
182  /*isVarArg=*/false));
183  }
184  if (!module.lookupSymbol(cuLaunchKernelName)) {
185  // Other than the CUDA api, the wrappers use uintptr_t to match the
186  // LLVM type if MLIR's index type, which the GPU dialect uses.
187  // Furthermore, they use void* instead of CUDA's opaque CUfunction and
188  // CUstream.
189  builder.create<LLVM::LLVMFuncOp>(
190  loc, cuLaunchKernelName,
191  LLVM::LLVMType::getFunctionTy(
192  getCUResultType(),
193  {
194  getPointerType(), /* void* f */
195  getIntPtrType(), /* intptr_t gridXDim */
196  getIntPtrType(), /* intptr_t gridyDim */
197  getIntPtrType(), /* intptr_t gridZDim */
198  getIntPtrType(), /* intptr_t blockXDim */
199  getIntPtrType(), /* intptr_t blockYDim */
200  getIntPtrType(), /* intptr_t blockZDim */
201  getInt32Type(), /* unsigned int sharedMemBytes */
202  getPointerType(), /* void *hstream */
203  getPointerPointerType(), /* void **kernelParams */
204  getPointerPointerType() /* void **extra */
205  },
206  /*isVarArg=*/false));
207  }
208  if (!module.lookupSymbol(cuGetStreamHelperName)) {
209  // Helper function to get the current CUDA stream. Uses void* instead of
210  // CUDAs opaque CUstream.
211  builder.create<LLVM::LLVMFuncOp>(
212  loc, cuGetStreamHelperName,
213  LLVM::LLVMType::getFunctionTy(getPointerType(), /*isVarArg=*/false));
214  }
215  if (!module.lookupSymbol(cuStreamSynchronizeName)) {
216  builder.create<LLVM::LLVMFuncOp>(
217  loc, cuStreamSynchronizeName,
218  LLVM::LLVMType::getFunctionTy(getCUResultType(),
219  getPointerType() /* CUstream stream */,
220  /*isVarArg=*/false));
221  }
222  if (!module.lookupSymbol(kMcuMemHostRegister)) {
223  builder.create<LLVM::LLVMFuncOp>(
224  loc, kMcuMemHostRegister,
225  LLVM::LLVMType::getFunctionTy(getVoidType(),
226  {
227  getPointerType(), /* void *ptr */
228  getInt64Type() /* int64 sizeBytes*/
229  },
230  /*isVarArg=*/false));
231  }
232 }

234 // Generates a parameters array to be used with a CUDA kernel launch call. The
235 // arguments are extracted from the launchOp.
236 // The generated code is essentially as follows:
237 //
238 // %array = alloca(numparams * sizeof(void *))
239 // for (i : [0, NumKernelOperands))
240 // %array[i] = cast<void*>(KernelOperand[i])
241 // return %array
242 Value GpuLaunchFuncToCudaCallsPass::setupParamsArray(gpu::LaunchFuncOp launchOp,
243  OpBuilder &builder) {
244  auto numKernelOperands = launchOp.getNumKernelOperands();
245  Location loc = launchOp.getLoc();
246  auto one = builder.create<LLVM::ConstantOp>(loc, getInt32Type(),
247  builder.getI32IntegerAttr(1));
248  // Provision twice as much for the `array` to allow up to one level of
249  // indirection for each argument.
250  auto arraySize = builder.create<LLVM::ConstantOp>(
251  loc, getInt32Type(), builder.getI32IntegerAttr(numKernelOperands));
252  auto array = builder.create<LLVM::AllocaOp>(loc, getPointerPointerType(),
253  arraySize, /*alignment=*/0);
254  for (unsigned idx = 0; idx < numKernelOperands; ++idx) {
255  auto operand = launchOp.getKernelOperand(idx);
256  auto llvmType = operand.getType().cast<LLVM::LLVMType>();
257  Value memLocation = builder.create<LLVM::AllocaOp>(
258  loc, llvmType.getPointerTo(), one, /*alignment=*/1);
259  builder.create<LLVM::StoreOp>(loc, operand, memLocation);
260  auto casted =
261  builder.create<LLVM::BitcastOp>(loc, getPointerType(), memLocation);
262 
263  // Assume all struct arguments come from MemRef. If this assumption does not
264  // hold anymore then we `launchOp` to lower from MemRefType and not after
265  // LLVMConversion has taken place and the MemRef information is lost.
266  // Extra level of indirection in the `array`:
267  // the descriptor pointer is registered via @mcuMemHostRegisterPtr
268  if (llvmType.isStructTy()) {
269  auto registerFunc =
270  getModule().lookupSymbol<LLVM::LLVMFuncOp>(kMcuMemHostRegister);
271  auto nullPtr = builder.create<LLVM::NullOp>(loc, llvmType.getPointerTo());
272  auto gep = builder.create<LLVM::GEPOp>(loc, llvmType.getPointerTo(),
273  ArrayRef<Value>{nullPtr, one});
274  auto size = builder.create<LLVM::PtrToIntOp>(loc, getInt64Type(), gep);
275  builder.create<LLVM::CallOp>(loc, ArrayRef<Type>{},
276  builder.getSymbolRefAttr(registerFunc),
277  ArrayRef<Value>{casted, size});
278  Value memLocation = builder.create<LLVM::AllocaOp>(
279  loc, getPointerPointerType(), one, /*alignment=*/1);
280  builder.create<LLVM::StoreOp>(loc, casted, memLocation);
281  casted =
282  builder.create<LLVM::BitcastOp>(loc, getPointerType(), memLocation);
283  }
284 
285  auto index = builder.create<LLVM::ConstantOp>(
286  loc, getInt32Type(), builder.getI32IntegerAttr(idx));
287  auto gep = builder.create<LLVM::GEPOp>(loc, getPointerPointerType(), array,
288  ArrayRef<Value>{index});
289  builder.create<LLVM::StoreOp>(loc, casted, gep);
290  }
291  return array;
292 }

294 // Generates an LLVM IR dialect global that contains the name of the given
295 // kernel function as a C string, and returns a pointer to its beginning.
296 // The code is essentially:
297 //
298 // llvm.global constant @kernel_name("function_name\00")
299 // func(...) {
300 // %0 = llvm.addressof @kernel_name
301 // %1 = llvm.constant (0 : index)
302 // %2 = llvm.getelementptr %0[%1, %1] : !llvm<"i8*">
303 // }
304 Value GpuLaunchFuncToCudaCallsPass::generateKernelNameConstant(
305  StringRef name, Location loc, OpBuilder &builder) {
306  // Make sure the trailing zero is included in the constant.
307  std::vector<char> kernelName(name.begin(), name.end());
308  kernelName.push_back('\0');
309 
310  std::string globalName = std::string(llvm::formatv("{0}_kernel_name", name));
311  return LLVM::createGlobalString(
312  loc, builder, globalName, StringRef(kernelName.data(), kernelName.size()),
313  LLVM::Linkage::Internal, llvmDialect);
314 }

316 // Emits LLVM IR to launch a kernel function. Expects the module that contains
317 // the compiled kernel function as a cubin in the 'nvvm.cubin' attribute of the
318 // kernel function in the IR.
319 // While MLIR has no global constants, also expects a cubin getter function in
320 // an 'nvvm.cubingetter' attribute. Such function is expected to return a
321 // pointer to the cubin blob when invoked.
322 // With these given, the generated code in essence is
323 //
324 // %0 = call %cubingetter
325 // %1 = alloca sizeof(void*)
326 // call %mcuModuleLoad(%2, %1)
327 // %2 = alloca sizeof(void*)
328 // %3 = load %1
329 // %4 = <see generateKernelNameConstant>
330 // call %mcuModuleGetFunction(%2, %3, %4)
331 // %5 = call %mcuGetStreamHelper()
332 // %6 = load %2
333 // %7 = <see setupParamsArray>
334 // call %mcuLaunchKernel(%6, <launchOp operands 0..5>, 0, %5, %7, nullptr)
335 // call %mcuStreamSynchronize(%5)
336 void GpuLaunchFuncToCudaCallsPass::translateGpuLaunchCalls(
337  mlir::gpu::LaunchFuncOp launchOp) {
338  OpBuilder builder(launchOp);
339  Location loc = launchOp.getLoc();
340  declareCudaFunctions(loc);
341 
342  auto zero = builder.create<LLVM::ConstantOp>(loc, getInt32Type(),
343  builder.getI32IntegerAttr(0));
344  // Create an LLVM global with CUBIN extracted from the kernel annotation and
345  // obtain a pointer to the first byte in it.
346  auto kernelModule = getModule().lookupSymbol<gpu::GPUModuleOp>(
347  launchOp.getKernelModuleName());
348  assert(kernelModule && "expected a kernel module");
349 
350  auto cubinAttr = kernelModule.getAttrOfType<StringAttr>(kCubinAnnotation);
351  if (!cubinAttr) {
352  kernelModule.emitOpError()
353  << "missing " << kCubinAnnotation << " attribute";
354  return signalPassFailure();
355  }
356 
357  SmallString<128> nameBuffer(kernelModule.getName());
358  nameBuffer.append(kCubinStorageSuffix);
359  Value data = LLVM::createGlobalString(
360  loc, builder, nameBuffer.str(), cubinAttr.getValue(),
361  LLVM::Linkage::Internal, getLLVMDialect());
362 
363  // Emit the load module call to load the module data. Error checking is done
364  // in the called helper function.
365  auto cuModule = allocatePointer(builder, loc);
366  auto cuModuleLoad =
367  getModule().lookupSymbol<LLVM::LLVMFuncOp>(cuModuleLoadName);
368  builder.create<LLVM::CallOp>(loc, ArrayRef<Type>{getCUResultType()},
369  builder.getSymbolRefAttr(cuModuleLoad),
370  ArrayRef<Value>{cuModule, data});
371  // Get the function from the module. The name corresponds to the name of
372  // the kernel function.
373  auto cuOwningModuleRef =
374  builder.create<LLVM::LoadOp>(loc, getPointerType(), cuModule);
375  auto kernelName = generateKernelNameConstant(launchOp.kernel(), loc, builder);
376  auto cuFunction = allocatePointer(builder, loc);
377  auto cuModuleGetFunction =
378  getModule().lookupSymbol<LLVM::LLVMFuncOp>(cuModuleGetFunctionName);
379  builder.create<LLVM::CallOp>(
380  loc, ArrayRef<Type>{getCUResultType()},
381  builder.getSymbolRefAttr(cuModuleGetFunction),
382  ArrayRef<Value>{cuFunction, cuOwningModuleRef, kernelName});
383  // Grab the global stream needed for execution.
384  auto cuGetStreamHelper =
385  getModule().lookupSymbol<LLVM::LLVMFuncOp>(cuGetStreamHelperName);
386  auto cuStream = builder.create<LLVM::CallOp>(
387  loc, ArrayRef<Type>{getPointerType()},
388  builder.getSymbolRefAttr(cuGetStreamHelper), ArrayRef<Value>{});
389  // Invoke the function with required arguments.
390  auto cuLaunchKernel =
391  getModule().lookupSymbol<LLVM::LLVMFuncOp>(cuLaunchKernelName);
392  auto cuFunctionRef =
393  builder.create<LLVM::LoadOp>(loc, getPointerType(), cuFunction);
394  auto paramsArray = setupParamsArray(launchOp, builder);
395  auto nullpointer =
396  builder.create<LLVM::IntToPtrOp>(loc, getPointerPointerType(), zero);
397  builder.create<LLVM::CallOp>(
398  loc, ArrayRef<Type>{getCUResultType()},
399  builder.getSymbolRefAttr(cuLaunchKernel),
400  ArrayRef<Value>{cuFunctionRef, launchOp.getOperand(0),
401  launchOp.getOperand(1), launchOp.getOperand(2),
402  launchOp.getOperand(3), launchOp.getOperand(4),
403  launchOp.getOperand(5), zero, /* sharedMemBytes */
404  cuStream.getResult(0), /* stream */
405  paramsArray, /* kernel params */
406  nullpointer /* extra */});
407  // Sync on the stream to make it synchronous.
408  auto cuStreamSync =
409  getModule().lookupSymbol<LLVM::LLVMFuncOp>(cuStreamSynchronizeName);
410  builder.create<LLVM::CallOp>(loc, ArrayRef<Type>{getCUResultType()},
411  builder.getSymbolRefAttr(cuStreamSync),
412  ArrayRef<Value>(cuStream.getResult(0)));
413  launchOp.erase();
414 }
