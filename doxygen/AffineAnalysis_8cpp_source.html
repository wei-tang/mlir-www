36 /// Returns the sequence of AffineApplyOp Operations operation in
37 /// 'affineApplyOps', which are reachable via a search starting from 'operands',
38 /// and ending at operands which are not defined by AffineApplyOps.
39 // TODO(andydavis) Add a method to AffineApplyOp which forward substitutes
40 // the AffineApplyOp into any user AffineApplyOps.
41 void mlir::getReachableAffineApplyOps(
42  ArrayRef<Value> operands, SmallVectorImpl<Operation *> &affineApplyOps) {
43  struct State {
44  // The ssa value for this node in the DFS traversal.
45  Value value;
46  // The operand index of 'value' to explore next during DFS traversal.
47  unsigned operandIndex;
48  };
49  SmallVector<State, 4> worklist;
50  for (auto operand : operands) {
51  worklist.push_back({operand, 0});
52  }
53 
54  while (!worklist.empty()) {
55  State &state = worklist.back();
56  auto *opInst = state.value.getDefiningOp();
57  // Note: getDefiningOp will return nullptr if the operand is not an
58  // Operation (i.e. block argument), which is a terminator for the search.
59  if (!isa_and_nonnull<AffineApplyOp>(opInst)) {
60  worklist.pop_back();
61  continue;
62  }
63 
64  if (state.operandIndex == 0) {
65  // Pre-Visit: Add 'opInst' to reachable sequence.
66  affineApplyOps.push_back(opInst);
67  }
68  if (state.operandIndex < opInst->getNumOperands()) {
69  // Visit: Add next 'affineApplyOp' operand to worklist.
70  // Get next operand to visit at 'operandIndex'.
71  auto nextOperand = opInst->getOperand(state.operandIndex);
72  // Increment 'operandIndex' in 'state'.
73  ++state.operandIndex;
74  // Add 'nextOperand' to worklist.
75  worklist.push_back({nextOperand, 0});
76  } else {
77  // Post-visit: done visiting operands AffineApplyOp, pop off stack.
78  worklist.pop_back();
79  }
80  }
81 }

83 // Builds a system of constraints with dimensional identifiers corresponding to
84 // the loop IVs of the forOps appearing in that order. Any symbols founds in
85 // the bound operands are added as symbols in the system. Returns failure for
86 // the yet unimplemented cases.
87 // TODO(andydavis,bondhugula) Handle non-unit steps through local variables or
88 // stride information in FlatAffineConstraints. (For eg., by using iv - lb %
89 // step = 0 and/or by introducing a method in FlatAffineConstraints
90 // setExprStride(ArrayRef<int64_t> expr, int64_t stride)
91 LogicalResult mlir::getIndexSet(MutableArrayRef<AffineForOp> forOps,
92  FlatAffineConstraints *domain) {
93  SmallVector<Value, 4> indices;
94  extractForInductionVars(forOps, &indices);
95  // Reset while associated Values in 'indices' to the domain.
96  domain->reset(forOps.size(), /*numSymbols=*/0, /*numLocals=*/0, indices);
97  for (auto forOp : forOps) {
98  // Add constraints from forOp's bounds.
99  if (failed(domain->addAffineForOpDomain(forOp)))
100  return failure();
101  }
102  return success();
103 }

121 // ValuePositionMap manages the mapping from Values which represent dimension
122 // and symbol identifiers from 'src' and 'dst' access functions to positions
123 // in new space where some Values are kept separate (using addSrc/DstValue)
124 // and some Values are merged (addSymbolValue).
125 // Position lookups return the absolute position in the new space which
126 // has the following format:
127 //
128 // [src-dim-identifiers] [dst-dim-identifiers] [symbol-identifiers]
129 //
130 // Note: access function non-IV dimension identifiers (that have 'dimension'
131 // positions in the access function position space) are assigned as symbols
132 // in the output position space. Convenience access functions which lookup
133 // an Value in multiple maps are provided (i.e. getSrcDimOrSymPos) to handle
134 // the common case of resolving positions for all access function operands.
135 //
136 // TODO(andydavis) Generalize this: could take a template parameter for
137 // the number of maps (3 in the current case), and lookups could take indices
138 // of maps to check. So getSrcDimOrSymPos would be "getPos(value, {0, 2})".
139 class ValuePositionMap {
140 public:
141  void addSrcValue(Value value) {
142  if (addValueAt(value, &srcDimPosMap, numSrcDims))
143  ++numSrcDims;
144  }
145  void addDstValue(Value value) {
146  if (addValueAt(value, &dstDimPosMap, numDstDims))
147  ++numDstDims;
148  }
149  void addSymbolValue(Value value) {
150  if (addValueAt(value, &symbolPosMap, numSymbols))
151  ++numSymbols;
152  }
153  unsigned getSrcDimOrSymPos(Value value) const {
154  return getDimOrSymPos(value, srcDimPosMap, 0);
155  }
156  unsigned getDstDimOrSymPos(Value value) const {
157  return getDimOrSymPos(value, dstDimPosMap, numSrcDims);
158  }
159  unsigned getSymPos(Value value) const {
160  auto it = symbolPosMap.find(value);
161  assert(it != symbolPosMap.end());
162  return numSrcDims + numDstDims + it->second;
163  }
164 
165  unsigned getNumSrcDims() const { return numSrcDims; }
166  unsigned getNumDstDims() const { return numDstDims; }
167  unsigned getNumDims() const { return numSrcDims + numDstDims; }
168  unsigned getNumSymbols() const { return numSymbols; }
169 
170 private:
171  bool addValueAt(Value value, DenseMap<Value, unsigned> *posMap,
172  unsigned position) {
173  auto it = posMap->find(value);
174  if (it == posMap->end()) {
175  (*posMap)[value] = position;
176  return true;
177  }
178  return false;
179  }
180  unsigned getDimOrSymPos(Value value,
181  const DenseMap<Value, unsigned> &dimPosMap,
182  unsigned dimPosOffset) const {
183  auto it = dimPosMap.find(value);
184  if (it != dimPosMap.end()) {
185  return dimPosOffset + it->second;
186  }
187  it = symbolPosMap.find(value);
188  assert(it != symbolPosMap.end());
189  return numSrcDims + numDstDims + it->second;
190  }
191 
192  unsigned numSrcDims = 0;
193  unsigned numDstDims = 0;
194  unsigned numSymbols = 0;
195  DenseMap<Value, unsigned> srcDimPosMap;
196  DenseMap<Value, unsigned> dstDimPosMap;
197  DenseMap<Value, unsigned> symbolPosMap;
198 };
199 } // namespace

201 // Builds a map from Value to identifier position in a new merged identifier
202 // list, which is the result of merging dim/symbol lists from src/dst
203 // iteration domains, the format of which is as follows:
204 //
205 // [src-dim-identifiers, dst-dim-identifiers, symbol-identifiers, const_term]
206 //
207 // This method populates 'valuePosMap' with mappings from operand Values in
208 // 'srcAccessMap'/'dstAccessMap' (as well as those in 'srcDomain'/'dstDomain')
209 // to the position of these values in the merged list.
210 static void buildDimAndSymbolPositionMaps(
211  const FlatAffineConstraints &srcDomain,
212  const FlatAffineConstraints &dstDomain, const AffineValueMap &srcAccessMap,
213  const AffineValueMap &dstAccessMap, ValuePositionMap *valuePosMap,
214  FlatAffineConstraints *dependenceConstraints) {
215  auto updateValuePosMap = [&](ArrayRef<Value> values, bool isSrc) {
216  for (unsigned i = 0, e = values.size(); i < e; ++i) {
217  auto value = values[i];
218  if (!isForInductionVar(values[i])) {
219  assert(isValidSymbol(values[i]) &&
220  "access operand has to be either a loop IV or a symbol");
221  valuePosMap->addSymbolValue(value);
222  } else if (isSrc) {
223  valuePosMap->addSrcValue(value);
224  } else {
225  valuePosMap->addDstValue(value);
226  }
227  }
228  };
229 
230  SmallVector<Value, 4> srcValues, destValues;
231  srcDomain.getIdValues(0, srcDomain.getNumDimAndSymbolIds(), &srcValues);
232  dstDomain.getIdValues(0, dstDomain.getNumDimAndSymbolIds(), &destValues);
233  // Update value position map with identifiers from src iteration domain.
234  updateValuePosMap(srcValues, /*isSrc=*/true);
235  // Update value position map with identifiers from dst iteration domain.
236  updateValuePosMap(destValues, /*isSrc=*/false);
237  // Update value position map with identifiers from src access function.
238  updateValuePosMap(srcAccessMap.getOperands(), /*isSrc=*/true);
239  // Update value position map with identifiers from dst access function.
240  updateValuePosMap(dstAccessMap.getOperands(), /*isSrc=*/false);
241 }

243 // Sets up dependence constraints columns appropriately, in the format:
244 // [src-dim-ids, dst-dim-ids, symbol-ids, local-ids, const_term]
245 static void initDependenceConstraints(
246  const FlatAffineConstraints &srcDomain,
247  const FlatAffineConstraints &dstDomain, const AffineValueMap &srcAccessMap,
248  const AffineValueMap &dstAccessMap, const ValuePositionMap &valuePosMap,
249  FlatAffineConstraints *dependenceConstraints) {
250  // Calculate number of equalities/inequalities and columns required to
251  // initialize FlatAffineConstraints for 'dependenceDomain'.
252  unsigned numIneq =
253  srcDomain.getNumInequalities() + dstDomain.getNumInequalities();
254  AffineMap srcMap = srcAccessMap.getAffineMap();
255  assert(srcMap.getNumResults() == dstAccessMap.getAffineMap().getNumResults());
256  unsigned numEq = srcMap.getNumResults();
257  unsigned numDims = srcDomain.getNumDimIds() + dstDomain.getNumDimIds();
258  unsigned numSymbols = valuePosMap.getNumSymbols();
259  unsigned numLocals = srcDomain.getNumLocalIds() + dstDomain.getNumLocalIds();
260  unsigned numIds = numDims + numSymbols + numLocals;
261  unsigned numCols = numIds + 1;
262 
263  // Set flat affine constraints sizes and reserving space for constraints.
264  dependenceConstraints->reset(numIneq, numEq, numCols, numDims, numSymbols,
265  numLocals);
266 
267  // Set values corresponding to dependence constraint identifiers.
268  SmallVector<Value, 4> srcLoopIVs, dstLoopIVs;
269  srcDomain.getIdValues(0, srcDomain.getNumDimIds(), &srcLoopIVs);
270  dstDomain.getIdValues(0, dstDomain.getNumDimIds(), &dstLoopIVs);
271 
272  dependenceConstraints->setIdValues(0, srcLoopIVs.size(), srcLoopIVs);
273  dependenceConstraints->setIdValues(
274  srcLoopIVs.size(), srcLoopIVs.size() + dstLoopIVs.size(), dstLoopIVs);
275 
276  // Set values for the symbolic identifier dimensions.
277  auto setSymbolIds = [&](ArrayRef<Value> values) {
278  for (auto value : values) {
279  if (!isForInductionVar(value)) {
280  assert(isValidSymbol(value) && "expected symbol");
281  dependenceConstraints->setIdValue(valuePosMap.getSymPos(value), value);
282  }
283  }
284  };
285 
286  setSymbolIds(srcAccessMap.getOperands());
287  setSymbolIds(dstAccessMap.getOperands());
288 
289  SmallVector<Value, 8> srcSymbolValues, dstSymbolValues;
290  srcDomain.getIdValues(srcDomain.getNumDimIds(),
291  srcDomain.getNumDimAndSymbolIds(), &srcSymbolValues);
292  dstDomain.getIdValues(dstDomain.getNumDimIds(),
293  dstDomain.getNumDimAndSymbolIds(), &dstSymbolValues);
294  setSymbolIds(srcSymbolValues);
295  setSymbolIds(dstSymbolValues);
296 
297  for (unsigned i = 0, e = dependenceConstraints->getNumDimAndSymbolIds();
298  i < e; i++)
299  assert(dependenceConstraints->getIds()[i].hasValue());
300 }

302 // Adds iteration domain constraints from 'srcDomain' and 'dstDomain' into
303 // 'dependenceDomain'.
304 // Uses 'valuePosMap' to determine the position in 'dependenceDomain' to which a
305 // srcDomain/dstDomain Value maps.
306 static void addDomainConstraints(const FlatAffineConstraints &srcDomain,
307  const FlatAffineConstraints &dstDomain,
308  const ValuePositionMap &valuePosMap,
309  FlatAffineConstraints *dependenceDomain) {
310  unsigned depNumDimsAndSymbolIds = dependenceDomain->getNumDimAndSymbolIds();
311 
312  SmallVector<int64_t, 4> cst(dependenceDomain->getNumCols());
313 
314  auto addDomain = [&](bool isSrc, bool isEq, unsigned localOffset) {
315  const FlatAffineConstraints &domain = isSrc ? srcDomain : dstDomain;
316  unsigned numCsts =
317  isEq ? domain.getNumEqualities() : domain.getNumInequalities();
318  unsigned numDimAndSymbolIds = domain.getNumDimAndSymbolIds();
319  auto at = [&](unsigned i, unsigned j) -> int64_t {
320  return isEq ? domain.atEq(i, j) : domain.atIneq(i, j);
321  };
322  auto map = [&](unsigned i) -> int64_t {
323  return isSrc ? valuePosMap.getSrcDimOrSymPos(domain.getIdValue(i))
324  : valuePosMap.getDstDimOrSymPos(domain.getIdValue(i));
325  };
326 
327  for (unsigned i = 0; i < numCsts; ++i) {
328  // Zero fill.
329  std::fill(cst.begin(), cst.end(), 0);
330  // Set coefficients for identifiers corresponding to domain.
331  for (unsigned j = 0; j < numDimAndSymbolIds; ++j)
332  cst[map(j)] = at(i, j);
333  // Local terms.
334  for (unsigned j = 0, e = domain.getNumLocalIds(); j < e; j++)
335  cst[depNumDimsAndSymbolIds + localOffset + j] =
336  at(i, numDimAndSymbolIds + j);
337  // Set constant term.
338  cst[cst.size() - 1] = at(i, domain.getNumCols() - 1);
339  // Add constraint.
340  if (isEq)
341  dependenceDomain->addEquality(cst);
342  else
343  dependenceDomain->addInequality(cst);
344  }
345  };
346 
347  // Add equalities from src domain.
348  addDomain(/*isSrc=*/true, /*isEq=*/true, /*localOffset=*/0);
349  // Add inequalities from src domain.
350  addDomain(/*isSrc=*/true, /*isEq=*/false, /*localOffset=*/0);
351  // Add equalities from dst domain.
352  addDomain(/*isSrc=*/false, /*isEq=*/true,
353  /*localOffset=*/srcDomain.getNumLocalIds());
354  // Add inequalities from dst domain.
355  addDomain(/*isSrc=*/false, /*isEq=*/false,
356  /*localOffset=*/srcDomain.getNumLocalIds());
357 }

359 // Adds equality constraints that equate src and dst access functions
360 // represented by 'srcAccessMap' and 'dstAccessMap' for each result.
361 // Requires that 'srcAccessMap' and 'dstAccessMap' have the same results count.
362 // For example, given the following two accesses functions to a 2D memref:
363 //
364 // Source access function:
365 // (a0 * d0 + a1 * s0 + a2, b0 * d0 + b1 * s0 + b2)
366 //
367 // Destination access function:
368 // (c0 * d0 + c1 * s0 + c2, f0 * d0 + f1 * s0 + f2)
369 //
370 // This method constructs the following equality constraints in
371 // 'dependenceDomain', by equating the access functions for each result
372 // (i.e. each memref dim). Notice that 'd0' for the destination access function
373 // is mapped into 'd0' in the equality constraint:
374 //
375 // d0 d1 s0 c
376 // -- -- -- --
377 // a0 -c0 (a1 - c1) (a1 - c2) = 0
378 // b0 -f0 (b1 - f1) (b1 - f2) = 0
379 //
380 // Returns failure if any AffineExpr cannot be flattened (due to it being
381 // semi-affine). Returns success otherwise.
382 static LogicalResult
383 addMemRefAccessConstraints(const AffineValueMap &srcAccessMap,
384  const AffineValueMap &dstAccessMap,
385  const ValuePositionMap &valuePosMap,
386  FlatAffineConstraints *dependenceDomain) {
387  AffineMap srcMap = srcAccessMap.getAffineMap();
388  AffineMap dstMap = dstAccessMap.getAffineMap();
389  assert(srcMap.getNumResults() == dstMap.getNumResults());
390  unsigned numResults = srcMap.getNumResults();
391 
392  unsigned srcNumIds = srcMap.getNumDims() + srcMap.getNumSymbols();
393  ArrayRef<Value> srcOperands = srcAccessMap.getOperands();
394 
395  unsigned dstNumIds = dstMap.getNumDims() + dstMap.getNumSymbols();
396  ArrayRef<Value> dstOperands = dstAccessMap.getOperands();
397 
398  std::vector<SmallVector<int64_t, 8>> srcFlatExprs;
399  std::vector<SmallVector<int64_t, 8>> destFlatExprs;
400  FlatAffineConstraints srcLocalVarCst, destLocalVarCst;
401  // Get flattened expressions for the source destination maps.
402  if (failed(getFlattenedAffineExprs(srcMap, &srcFlatExprs, &srcLocalVarCst)) ||
403  failed(getFlattenedAffineExprs(dstMap, &destFlatExprs, &destLocalVarCst)))
404  return failure();
405 
406  unsigned domNumLocalIds = dependenceDomain->getNumLocalIds();
407  unsigned srcNumLocalIds = srcLocalVarCst.getNumLocalIds();
408  unsigned dstNumLocalIds = destLocalVarCst.getNumLocalIds();
409  unsigned numLocalIdsToAdd = srcNumLocalIds + dstNumLocalIds;
410  for (unsigned i = 0; i < numLocalIdsToAdd; i++) {
411  dependenceDomain->addLocalId(dependenceDomain->getNumLocalIds());
412  }
413 
414  unsigned numDims = dependenceDomain->getNumDimIds();
415  unsigned numSymbols = dependenceDomain->getNumSymbolIds();
416  unsigned numSrcLocalIds = srcLocalVarCst.getNumLocalIds();
417  unsigned newLocalIdOffset = numDims + numSymbols + domNumLocalIds;
418 
419  // Equality to add.
420  SmallVector<int64_t, 8> eq(dependenceDomain->getNumCols());
421  for (unsigned i = 0; i < numResults; ++i) {
422  // Zero fill.
423  std::fill(eq.begin(), eq.end(), 0);
424 
425  // Flattened AffineExpr for src result 'i'.
426  const auto &srcFlatExpr = srcFlatExprs[i];
427  // Set identifier coefficients from src access function.
428  for (unsigned j = 0, e = srcOperands.size(); j < e; ++j)
429  eq[valuePosMap.getSrcDimOrSymPos(srcOperands[j])] = srcFlatExpr[j];
430  // Local terms.
431  for (unsigned j = 0, e = srcNumLocalIds; j < e; j++)
432  eq[newLocalIdOffset + j] = srcFlatExpr[srcNumIds + j];
433  // Set constant term.
434  eq[eq.size() - 1] = srcFlatExpr[srcFlatExpr.size() - 1];
435 
436  // Flattened AffineExpr for dest result 'i'.
437  const auto &destFlatExpr = destFlatExprs[i];
438  // Set identifier coefficients from dst access function.
439  for (unsigned j = 0, e = dstOperands.size(); j < e; ++j)
440  eq[valuePosMap.getDstDimOrSymPos(dstOperands[j])] -= destFlatExpr[j];
441  // Local terms.
442  for (unsigned j = 0, e = dstNumLocalIds; j < e; j++)
443  eq[newLocalIdOffset + numSrcLocalIds + j] = -destFlatExpr[dstNumIds + j];
444  // Set constant term.
445  eq[eq.size() - 1] -= destFlatExpr[destFlatExpr.size() - 1];
446 
447  // Add equality constraint.
448  dependenceDomain->addEquality(eq);
449  }
450 
451  // Add equality constraints for any operands that are defined by constant ops.
452  auto addEqForConstOperands = [&](ArrayRef<Value> operands) {
453  for (unsigned i = 0, e = operands.size(); i < e; ++i) {
454  if (isForInductionVar(operands[i]))
455  continue;
456  auto symbol = operands[i];
457  assert(isValidSymbol(symbol));
458  // Check if the symbol is a constant.
459  if (auto cOp = dyn_cast_or_null<ConstantIndexOp>(symbol.getDefiningOp()))
460  dependenceDomain->setIdToConstant(valuePosMap.getSymPos(symbol),
461  cOp.getValue());
462  }
463  };
464 
465  // Add equality constraints for any src symbols defined by constant ops.
466  addEqForConstOperands(srcOperands);
467  // Add equality constraints for any dst symbols defined by constant ops.
468  addEqForConstOperands(dstOperands);
469 
470  // By construction (see flattener), local var constraints will not have any
471  // equalities.
472  assert(srcLocalVarCst.getNumEqualities() == 0 &&
473  destLocalVarCst.getNumEqualities() == 0);
474  // Add inequalities from srcLocalVarCst and destLocalVarCst into the
475  // dependence domain.
476  SmallVector<int64_t, 8> ineq(dependenceDomain->getNumCols());
477  for (unsigned r = 0, e = srcLocalVarCst.getNumInequalities(); r < e; r++) {
478  std::fill(ineq.begin(), ineq.end(), 0);
479 
480  // Set identifier coefficients from src local var constraints.
481  for (unsigned j = 0, e = srcOperands.size(); j < e; ++j)
482  ineq[valuePosMap.getSrcDimOrSymPos(srcOperands[j])] =
483  srcLocalVarCst.atIneq(r, j);
484  // Local terms.
485  for (unsigned j = 0, e = srcNumLocalIds; j < e; j++)
486  ineq[newLocalIdOffset + j] = srcLocalVarCst.atIneq(r, srcNumIds + j);
487  // Set constant term.
488  ineq[ineq.size() - 1] =
489  srcLocalVarCst.atIneq(r, srcLocalVarCst.getNumCols() - 1);
490  dependenceDomain->addInequality(ineq);
491  }
492 
493  for (unsigned r = 0, e = destLocalVarCst.getNumInequalities(); r < e; r++) {
494  std::fill(ineq.begin(), ineq.end(), 0);
495  // Set identifier coefficients from dest local var constraints.
496  for (unsigned j = 0, e = dstOperands.size(); j < e; ++j)
497  ineq[valuePosMap.getDstDimOrSymPos(dstOperands[j])] =
498  destLocalVarCst.atIneq(r, j);
499  // Local terms.
500  for (unsigned j = 0, e = dstNumLocalIds; j < e; j++)
501  ineq[newLocalIdOffset + numSrcLocalIds + j] =
502  destLocalVarCst.atIneq(r, dstNumIds + j);
503  // Set constant term.
504  ineq[ineq.size() - 1] =
505  destLocalVarCst.atIneq(r, destLocalVarCst.getNumCols() - 1);
506 
507  dependenceDomain->addInequality(ineq);
508  }
509  return success();
510 }

512 // Returns the number of outer loop common to 'src/dstDomain'.
513 // Loops common to 'src/dst' domains are added to 'commonLoops' if non-null.
514 static unsigned
515 getNumCommonLoops(const FlatAffineConstraints &srcDomain,
516  const FlatAffineConstraints &dstDomain,
517  SmallVectorImpl<AffineForOp> *commonLoops = nullptr) {
518  // Find the number of common loops shared by src and dst accesses.
519  unsigned minNumLoops =
520  std::min(srcDomain.getNumDimIds(), dstDomain.getNumDimIds());
521  unsigned numCommonLoops = 0;
522  for (unsigned i = 0; i < minNumLoops; ++i) {
523  if (!isForInductionVar(srcDomain.getIdValue(i)) ||
524  !isForInductionVar(dstDomain.getIdValue(i)) ||
525  srcDomain.getIdValue(i) != dstDomain.getIdValue(i))
526  break;
527  if (commonLoops != nullptr)
528  commonLoops->push_back(getForInductionVarOwner(srcDomain.getIdValue(i)));
529  ++numCommonLoops;
530  }
531  if (commonLoops != nullptr)
532  assert(commonLoops->size() == numCommonLoops);
533  return numCommonLoops;

554 // Returns true if the ancestor operation of 'srcAccess' appears before the
555 // ancestor operation of 'dstAccess' in the common ancestral block. Returns
556 // false otherwise.
557 // Note that because 'srcAccess' or 'dstAccess' may be nested in conditionals,
558 // the function is named 'srcAppearsBeforeDstInCommonBlock'. Note that
559 // 'numCommonLoops' is the number of contiguous surrounding outer loops.
560 static bool srcAppearsBeforeDstInAncestralBlock(
561  const MemRefAccess &srcAccess, const MemRefAccess &dstAccess,
562  const FlatAffineConstraints &srcDomain, unsigned numCommonLoops) {

564  auto *commonBlock =

569  assert(srcInst != nullptr);
570  auto *dstInst = commonBlock->findAncestorOpInBlock(*dstAccess.opInst);
571  assert(dstInst != nullptr);

574  return srcInst->isBeforeInBlock(dstInst);

577 // Adds ordering constraints to 'dependenceDomain' based on number of loops
578 // common to 'src/dstDomain' and requested 'loopDepth'.
579 // Note that 'loopDepth' cannot exceed the number of common loops plus one.
580 // EX: Given a loop nest of depth 2 with IVs 'i' and 'j':
581 // *) If 'loopDepth == 1' then one constraint is added: i' >= i + 1
582 // *) If 'loopDepth == 2' then two constraints are added: i == i' and j' > j + 1
583 // *) If 'loopDepth == 3' then two constraints are added: i == i' and j == j'
584 static void addOrderingConstraints(const FlatAffineConstraints &srcDomain,
585  const FlatAffineConstraints &dstDomain,
586  unsigned loopDepth,
587  FlatAffineConstraints *dependenceDomain) {
588  unsigned numCols = dependenceDomain->getNumCols();
589  SmallVector<int64_t, 4> eq(numCols);
590  unsigned numSrcDims = srcDomain.getNumDimIds();
591  unsigned numCommonLoops = getNumCommonLoops(srcDomain, dstDomain);
592  unsigned numCommonLoopConstraints = std::min(numCommonLoops, loopDepth);
593  for (unsigned i = 0; i < numCommonLoopConstraints; ++i) {
594  std::fill(eq.begin(), eq.end(), 0);
595  eq[i] = -1;
596  eq[i + numSrcDims] = 1;
597  if (i == loopDepth - 1) {
598  eq[numCols - 1] = -1;
599  dependenceDomain->addInequality(eq);
600  } else {
601  dependenceDomain->addEquality(eq);
602  }
603  }
604 }

606 // Computes distance and direction vectors in 'dependences', by adding
607 // variables to 'dependenceDomain' which represent the difference of the IVs,
608 // eliminating all other variables, and reading off distance vectors from
609 // equality constraints (if possible), and direction vectors from inequalities.
610 static void computeDirectionVector(
611  const FlatAffineConstraints &srcDomain,
612  const FlatAffineConstraints &dstDomain, unsigned loopDepth,
613  FlatAffineConstraints *dependenceDomain,
614  SmallVector<DependenceComponent, 2> *dependenceComponents) {
615  // Find the number of common loops shared by src and dst accesses.
616  SmallVector<AffineForOp, 4> commonLoops;
617  unsigned numCommonLoops =
618  getNumCommonLoops(srcDomain, dstDomain, &commonLoops);
619  if (numCommonLoops == 0)
620  return;
621  // Compute direction vectors for requested loop depth.
622  unsigned numIdsToEliminate = dependenceDomain->getNumIds();
623  // Add new variables to 'dependenceDomain' to represent the direction
624  // constraints for each shared loop.
625  for (unsigned j = 0; j < numCommonLoops; ++j) {
626  dependenceDomain->addDimId(j);
627  }
628 
629  // Add equality constraints for each common loop, setting newly introduced
630  // variable at column 'j' to the 'dst' IV minus the 'src IV.
631  SmallVector<int64_t, 4> eq;
632  eq.resize(dependenceDomain->getNumCols());
633  unsigned numSrcDims = srcDomain.getNumDimIds();
634  // Constraint variables format:
635  // [num-common-loops][num-src-dim-ids][num-dst-dim-ids][num-symbols][constant]
636  for (unsigned j = 0; j < numCommonLoops; ++j) {
637  std::fill(eq.begin(), eq.end(), 0);
638  eq[j] = 1;
639  eq[j + numCommonLoops] = 1;
640  eq[j + numCommonLoops + numSrcDims] = -1;
641  dependenceDomain->addEquality(eq);
642  }
643 
644  // Eliminate all variables other than the direction variables just added.
645  dependenceDomain->projectOut(numCommonLoops, numIdsToEliminate);
646 
647  // Scan each common loop variable column and set direction vectors based
648  // on eliminated constraint system.
649  dependenceComponents->resize(numCommonLoops);
650  for (unsigned j = 0; j < numCommonLoops; ++j) {
651  (*dependenceComponents)[j].op = commonLoops[j].getOperation();
652  auto lbConst = dependenceDomain->getConstantLowerBound(j);
653  (*dependenceComponents)[j].lb =
654  lbConst.getValueOr(std::numeric_limits<int64_t>::min());
655  auto ubConst = dependenceDomain->getConstantUpperBound(j);
656  (*dependenceComponents)[j].ub =
657  ubConst.getValueOr(std::numeric_limits<int64_t>::max());
658  }
659 }

677 // Builds a flat affine constraint system to check if there exists a dependence
678 // between memref accesses 'srcAccess' and 'dstAccess'.
679 // Returns 'NoDependence' if the accesses can be definitively shown not to
680 // access the same element.
681 // Returns 'HasDependence' if the accesses do access the same element.
682 // Returns 'Failure' if an error or unsupported case was encountered.
683 // If a dependence exists, returns in 'dependenceComponents' a direction
684 // vector for the dependence, with a component for each loop IV in loops
685 // common to both accesses (see Dependence in AffineAnalysis.h for details).
686 //
687 // The memref access dependence check is comprised of the following steps:
688 // *) Compute access functions for each access. Access functions are computed
689 // using AffineValueMaps initialized with the indices from an access, then
690 // composed with AffineApplyOps reachable from operands of that access,
691 // until operands of the AffineValueMap are loop IVs or symbols.
692 // *) Build iteration domain constraints for each access. Iteration domain
693 // constraints are pairs of inequality constraints representing the
694 // upper/lower loop bounds for each AffineForOp in the loop nest associated
695 // with each access.
696 // *) Build dimension and symbol position maps for each access, which map
697 // Values from access functions and iteration domains to their position
698 // in the merged constraint system built by this method.
699 //
700 // This method builds a constraint system with the following column format:
701 //
702 // [src-dim-identifiers, dst-dim-identifiers, symbols, constant]
703 //
704 // For example, given the following MLIR code with "source" and "destination"
705 // accesses to the same memref label, and symbols %M, %N, %K:
706 //
707 // affine.for %i0 = 0 to 100 {
708 // affine.for %i1 = 0 to 50 {
709 // %a0 = affine.apply
710 // (d0, d1) -> (d0 * 2 - d1 * 4 + s1, d1 * 3 - s0) (%i0, %i1)[%M, %N]
711 // // Source memref access.
712 // store %v0, %m[%a0#0, %a0#1] : memref<4x4xf32>
713 // }
714 // }
715 //
716 // affine.for %i2 = 0 to 100 {
717 // affine.for %i3 = 0 to 50 {
718 // %a1 = affine.apply
719 // (d0, d1) -> (d0 * 7 + d1 * 9 - s1, d1 * 11 + s0) (%i2, %i3)[%K, %M]
720 // // Destination memref access.
721 // %v1 = load %m[%a1#0, %a1#1] : memref<4x4xf32>
722 // }
723 // }
724 //
725 // The access functions would be the following:
726 //
727 // src: (%i0 * 2 - %i1 * 4 + %N, %i1 * 3 - %M)
728 // dst: (%i2 * 7 + %i3 * 9 - %M, %i3 * 11 - %K)
729 //
730 // The iteration domains for the src/dst accesses would be the following:
731 //
732 // src: 0 <= %i0 <= 100, 0 <= %i1 <= 50
733 // dst: 0 <= %i2 <= 100, 0 <= %i3 <= 50
734 //
735 // The symbols by both accesses would be assigned to a canonical position order
736 // which will be used in the dependence constraint system:
737 //
738 // symbol name: %M %N %K
739 // symbol pos: 0 1 2
740 //
741 // Equality constraints are built by equating each result of src/destination
742 // access functions. For this example, the following two equality constraints
743 // will be added to the dependence constraint system:
744 //
745 // [src_dim0, src_dim1, dst_dim0, dst_dim1, sym0, sym1, sym2, const]
746 // 2 -4 -7 -9 1 1 0 0 = 0
747 // 0 3 0 -11 -1 0 1 0 = 0
748 //
749 // Inequality constraints from the iteration domain will be meged into
750 // the dependence constraint system
751 //
752 // [src_dim0, src_dim1, dst_dim0, dst_dim1, sym0, sym1, sym2, const]
753 // 1 0 0 0 0 0 0 0 >= 0
754 // -1 0 0 0 0 0 0 100 >= 0
755 // 0 1 0 0 0 0 0 0 >= 0
756 // 0 -1 0 0 0 0 0 50 >= 0
757 // 0 0 1 0 0 0 0 0 >= 0
758 // 0 0 -1 0 0 0 0 100 >= 0
759 // 0 0 0 1 0 0 0 0 >= 0
760 // 0 0 0 -1 0 0 0 50 >= 0
761 //
762 //
763 // TODO(andydavis) Support AffineExprs mod/floordiv/ceildiv.
764 DependenceResult mlir::checkMemrefAccessDependence(
765  const MemRefAccess &srcAccess, const MemRefAccess &dstAccess,
766  unsigned loopDepth, FlatAffineConstraints *dependenceConstraints,
767  SmallVector<DependenceComponent, 2> *dependenceComponents, bool allowRAR) {
768  LLVM_DEBUG(llvm::dbgs() << "Checking for dependence at depth: "
769  << Twine(loopDepth) << " between:\n";);
770  LLVM_DEBUG(srcAccess.opInst->dump(););
771  LLVM_DEBUG(dstAccess.opInst->dump(););
772 
773  // Return 'NoDependence' if these accesses do not access the same memref.
774  if (srcAccess.memref != dstAccess.memref)
775  return DependenceResult::NoDependence;
776 
777  // Return 'NoDependence' if one of these accesses is not an AffineStoreOp.
778  if (!allowRAR && !isa<AffineStoreOp>(srcAccess.opInst) &&
779  !isa<AffineStoreOp>(dstAccess.opInst))
780  return DependenceResult::NoDependence;
781 
782  // Get composed access function for 'srcAccess'.
783  AffineValueMap srcAccessMap;
784  srcAccess.getAccessMap(&srcAccessMap);
785 
786  // Get composed access function for 'dstAccess'.
787  AffineValueMap dstAccessMap;
788  dstAccess.getAccessMap(&dstAccessMap);
789 
790  // Get iteration domain for the 'srcAccess' operation.
791  FlatAffineConstraints srcDomain;
792  if (failed(getInstIndexSet(srcAccess.opInst, &srcDomain)))
793  return DependenceResult::Failure;
794 
795  // Get iteration domain for 'dstAccess' operation.
796  FlatAffineConstraints dstDomain;
797  if (failed(getInstIndexSet(dstAccess.opInst, &dstDomain)))
798  return DependenceResult::Failure;
799 
800  // Return 'NoDependence' if loopDepth > numCommonLoops and if the ancestor
801  // operation of 'srcAccess' does not properly dominate the ancestor
802  // operation of 'dstAccess' in the same common operation block.
803  // Note: this check is skipped if 'allowRAR' is true, because because RAR
804  // deps can exist irrespective of lexicographic ordering b/w src and dst.
805  unsigned numCommonLoops = getNumCommonLoops(srcDomain, dstDomain);
806  assert(loopDepth <= numCommonLoops + 1);
807  if (!allowRAR && loopDepth > numCommonLoops &&
808  !srcAppearsBeforeDstInAncestralBlock(srcAccess, dstAccess, srcDomain,
809  numCommonLoops)) {
810  return DependenceResult::NoDependence;
811  }
812  // Build dim and symbol position maps for each access from access operand
813  // Value to position in merged constraint system.
814  ValuePositionMap valuePosMap;
815  buildDimAndSymbolPositionMaps(srcDomain, dstDomain, srcAccessMap,
816  dstAccessMap, &valuePosMap,
817  dependenceConstraints);
818 
819  initDependenceConstraints(srcDomain, dstDomain, srcAccessMap, dstAccessMap,
820  valuePosMap, dependenceConstraints);
821 
822  assert(valuePosMap.getNumDims() ==
823  srcDomain.getNumDimIds() + dstDomain.getNumDimIds());
824 
825  // Create memref access constraint by equating src/dst access functions.
826  // Note that this check is conservative, and will fail in the future when
827  // local variables for mod/div exprs are supported.
828  if (failed(addMemRefAccessConstraints(srcAccessMap, dstAccessMap, valuePosMap,
829  dependenceConstraints)))
830  return DependenceResult::Failure;
831 
832  // Add 'src' happens before 'dst' ordering constraints.
833  addOrderingConstraints(srcDomain, dstDomain, loopDepth,
834  dependenceConstraints);
835  // Add src and dst domain constraints.
836  addDomainConstraints(srcDomain, dstDomain, valuePosMap,
837  dependenceConstraints);
838 
839  // Return 'NoDependence' if the solution space is empty: no dependence.
840  if (dependenceConstraints->isEmpty()) {
841  return DependenceResult::NoDependence;
842  }
843 
844  // Compute dependence direction vector and return true.
845  if (dependenceComponents != nullptr) {
846  computeDirectionVector(srcDomain, dstDomain, loopDepth,
847  dependenceConstraints, dependenceComponents);
848  }
849 
850  LLVM_DEBUG(llvm::dbgs() << "Dependence polyhedron:\n");
851  LLVM_DEBUG(dependenceConstraints->dump());
852  return DependenceResult::HasDependence;
853 }
