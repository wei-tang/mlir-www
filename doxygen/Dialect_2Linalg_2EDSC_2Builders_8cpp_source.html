25 mlir::edsc::LoopRangeBuilder::LoopRangeBuilder(ValueHandle *iv,
26  ValueHandle range) {
27  assert(range.getType() && "expected !linalg.range type");
28  assert(range.getValue().getDefiningOp() &&
29  "need operations to extract range parts");

32  auto ub = rangeOp.max();

34  auto forOp = OperationHandle::createOp<ForOp>(lb, ub, step);
35  *iv = ValueHandle(forOp.getInductionVar());

37  enter(body, /*prev=*/1);

117 static void getMaxDimIndex(ArrayRef<StructuredIndexed> structuredIndices,
118  unsigned &pos) {
119  for (auto sidx : structuredIndices) {
120  for (auto expr : sidx.getExprs()) {
121  expr.walk([&pos](AffineExpr e) {
122  if (auto d = e.dyn_cast<AffineDimExpr>())
123  pos = std::max(pos, d.getPosition());
124  });
125  }
126  }

129 Operation *mlir::edsc::makeGenericLinalgOp(
130  ArrayRef<IterType> iteratorTypes, ArrayRef<StructuredIndexed> inputs,
131  ArrayRef<StructuredIndexed> outputs,
132  function_ref<void(ArrayRef<BlockArgument>)> regionBuilder,
133  ArrayRef<Value> otherValues, ArrayRef<Attribute> otherAttributes) {
134  for (unsigned i = 0, e = outputs.size(); i + 1 < e; ++i)
135  assert(!(outputs[i].getType().isa<RankedTensorType>() &&
136  outputs[i + 1].getType().isa<MemRefType>()) &&
137  "output tensors must be passed after output buffers");
138  auto &builder = edsc::ScopedContext::getBuilder();
139  auto *ctx = builder.getContext();
140  unsigned nInputs = inputs.size();
141  unsigned nOutputs = outputs.size();
142  unsigned maxPos = 0;
143  getMaxDimIndex(inputs, maxPos);
144  getMaxDimIndex(outputs, maxPos);
145  // maxPos is 0 indexed, need to turn this into a count (i.e. +1)
146  unsigned nDims = maxPos + 1;
147 
148  SmallVector<AffineMap, 4> maps;
149  maps.reserve(nInputs + nOutputs);
150  for (auto in : inputs)
151  maps.push_back(
152  AffineMap::get(/*dimCount=*/nDims, /*symbolCount=*/0, in.getExprs()));
153  for (auto out : outputs)
154  maps.push_back(
155  AffineMap::get(/*dimCount=*/nDims, /*symbolCount=*/0, out.getExprs()));
156 
157  unsigned nViews = nInputs + nOutputs;
158  SmallVector<Value, 4> values;
159  values.reserve(nViews);
160  values.append(inputs.begin(), inputs.end());
161  std::copy_if(outputs.begin(), outputs.end(), std::back_inserter(values),
162  [](StructuredIndexed s) { return s.hasValue(); });
163  SmallVector<Type, 4> types;
164  std::copy_if(outputs.begin(), outputs.end(), std::back_inserter(types),
165  [](StructuredIndexed s) { return !s.hasValue(); });
166 
167  auto iteratorStrTypes = functional::map(toString, iteratorTypes);
168  // clang-format off
169  auto *op =
170  edsc::ScopedContext::getBuilder()
171  .create<linalg::GenericOp>(
172  edsc::ScopedContext::getLocation(),
173  types,
174  values,
175  IntegerAttr::get(IntegerType::get(64, ctx), nInputs),
176  IntegerAttr::get(IntegerType::get(64, ctx), nOutputs),
177  builder.getAffineMapArrayAttr(maps),
178  builder.getStrArrayAttr(iteratorStrTypes),
179  StringAttr() /*doc*/,
180  FlatSymbolRefAttr() /*fun*/,
181  StringAttr() /*library_call*/
182  /* TODO: other attributes in op */
183  )
184  .getOperation();
185  // clang-format on
186 
187  using namespace edsc;
188  SmallVector<Type, 4> blockTypes;
189  blockTypes.reserve(values.size());
190  for (auto it : llvm::enumerate(values))
191  blockTypes.push_back((it.index() < nViews)
192  ? getElementTypeOrSelf(it.value())
193  : it.value().getType());
194 
195  assert(op->getNumRegions() == 1);
196  assert(op->getRegion(0).empty());
197  OpBuilder opBuilder(op);
198  ScopedContext scope(opBuilder, op->getLoc());
199  BlockHandle b;
200  auto handles = makeValueHandles(blockTypes);
201  BlockBuilder(&b, op->getRegion(0),
202  makeHandlePointers(MutableArrayRef<ValueHandle>(handles)))(
203  [&] { regionBuilder(b.getBlock()->getArguments()); });
204  assert(op->getRegion(0).getBlocks().size() == 1);
205  return op;
206 }

216 Operation *mlir::edsc::ops::linalg_pointwise(UnaryPointwiseOpBuilder unaryOp,
217  StructuredIndexed I,
218  StructuredIndexed O) {
219  SmallVector<edsc::IterType, 4> iterTypes(O.getExprs().size(),
220  edsc::IterType::Parallel);
221  if (O.getType().isa<RankedTensorType>()) {
222  auto fun = [&unaryOp](ArrayRef<BlockArgument> args) {
223  assert(args.size() == 1 && "expected 1 block arguments");
224  ValueHandle a(args[0]);
225  linalg_yield(unaryOp(a));
226  };
227  return makeGenericLinalgOp(iterTypes, {I}, {O}, fun);
228  }
229  auto fun = [&unaryOp](ArrayRef<BlockArgument> args) {
230  assert(args.size() == 2 && "expected 2 block arguments");

232  linalg_yield(unaryOp(a));
233  };
234  return makeGenericLinalgOp(iterTypes, {I}, {O}, fun);

301 Operation *mlir::edsc::ops::linalg_conv_nhwc(ValueHandle vI, ValueHandle vW,
302  ValueHandle vO,
303  ArrayRef<int> strides,
304  ArrayRef<int> dilations) {
305  MLIRContext *ctx = ScopedContext::getContext();
306  // TODO(ntv) some template magic to make everything rank-polymorphic.
307  assert((dilations.empty() || dilations.size() == 2) && "only 2-D conv atm");
308  assert((strides.empty() || strides.size() == 2) && "only 2-D conv atm");
309 
310  // Some short names.
311  auto par = IterType::Parallel;
312  auto red = IterType::Reduction;
313  auto s = strides;
314  auto d = dilations;
315 
316  AffineExpr b, f, h, w, kh, kw, c;
317  bindDims(ctx, b, f, h, w, kh, kw, c);
318  unsigned numDims = c.cast<AffineDimExpr>().getPosition() + 1;
319  StructuredIndexed I(vI), W(vW), O(vO);
320  // clang-format off
321  return makeGenericLinalgOp(
322  {par, par, par, par, red, red, red}, {
323  I({b,
324  // Roundtrip to flattened form to serve as canonicalization and ensure
325  // consistent ordering of subexpressions.
326  simplifyAffineExpr(s[0] * h + d[0] * kh, numDims, 0),
327  simplifyAffineExpr(s[1] * w + d[1] * kw, numDims, 0),
328  c}),
329  W({kh, kw, c, f})}, {
330  O({b, h, w, f})},
331  macRegionBuilder);
332  // clang-format on
333 }
