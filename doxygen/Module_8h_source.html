86  /// Insert the operation at the given insertion point. Note: The operation is

88  void insert(Operation *insertPt, Operation *op) {
89  insert(Block::iterator(insertPt), op);
90  }
91  void insert(Block::iterator insertPt, Operation *op) {
92  auto *body = getBody();
93  if (insertPt == body->end())
94  insertPt = Block::iterator(body->getTerminator());
95  body->getOperations().insert(insertPt, op);

114 /// This class acts as an owning reference to a module, and will automatically

120  OwningModuleRef(OwningModuleRef &&other) : module(other.release()) {}
121  ~OwningModuleRef() {

123  module.erase();
124  }
125 
126  // Assign from another module reference.

128  if (module)
129  module.erase();

131  return *this;
132  }
