29 /// Populates 'loops' with IVs of the loops surrounding 'op' ordered from
30 /// the outermost 'affine.for' operation to the innermost one.
31 void mlir::getLoopIVs(Operation &op, SmallVectorImpl<AffineForOp> *loops) {
32  auto *currOp = op.getParentOp();
33  AffineForOp currAffineForOp;
34  // Traverse up the hierarchy collecting all 'affine.for' operation while
35  // skipping over 'affine.if' operations.
36  while (currOp && ((currAffineForOp = dyn_cast<AffineForOp>(currOp)) ||
37  isa<AffineIfOp>(currOp))) {
38  if (currAffineForOp)
39  loops->push_back(currAffineForOp);
40  currOp = currOp->getParentOp();
41  }
42  std::reverse(loops->begin(), loops->end());
43 }

46 LogicalResult
47 ComputationSliceState::getAsConstraints(FlatAffineConstraints *cst) {
48  assert(!lbOperands.empty());
49  // Adds src 'ivs' as dimension identifiers in 'cst'.
50  unsigned numDims = ivs.size();
51  // Adds operands (dst ivs and symbols) as symbols in 'cst'.
52  unsigned numSymbols = lbOperands[0].size();
53 
54  SmallVector<Value, 4> values(ivs);
55  // Append 'ivs' then 'operands' to 'values'.
56  values.append(lbOperands[0].begin(), lbOperands[0].end());
57  cst->reset(numDims, numSymbols, 0, values);
58 
59  // Add loop bound constraints for values which are loop IVs and equality
60  // constraints for symbols which are constants.
61  for (const auto &value : values) {
62  assert(cst->containsId(value) && "value expected to be present");
63  if (isValidSymbol(value)) {
64  // Check if the symbol is a constant.
65 
66  if (auto cOp = dyn_cast_or_null<ConstantIndexOp>(value.getDefiningOp()))
67  cst->setIdToConstant(value, cOp.getValue());
68  } else if (auto loop = getForInductionVarOwner(value)) {
69  if (failed(cst->addAffineForOpDomain(loop)))
70  return failure();
71  }
72  }
73 
74  // Add slices bounds on 'ivs' using maps 'lbs'/'ubs' with 'lbOperands[0]'
75  LogicalResult ret = cst->addSliceBounds(ivs, lbs, ubs, lbOperands[0]);
76  assert(succeeded(ret) &&
77  "should not fail as we never have semi-affine slice maps");
78  (void)ret;
79  return success();
80 }

94 Optional<int64_t> MemRefRegion::getConstantBoundingSizeAndShape(
95  SmallVectorImpl<int64_t> *shape, std::vector<SmallVector<int64_t, 4>> *lbs,
96  SmallVectorImpl<int64_t> *lbDivisors) const {
97  auto memRefType = memref.getType().cast<MemRefType>();
98  unsigned rank = memRefType.getRank();
99  if (shape)
100  shape->reserve(rank);
101 
102  assert(rank == cst.getNumDimIds() && "inconsistent memref region");
103 
104  // Find a constant upper bound on the extent of this memref region along each
105  // dimension.
106  int64_t numElements = 1;
107  int64_t diffConstant;
108  int64_t lbDivisor;
109  for (unsigned d = 0; d < rank; d++) {
110  SmallVector<int64_t, 4> lb;
111  Optional<int64_t> diff = cst.getConstantBoundOnDimSize(d, &lb, &lbDivisor);
112  if (diff.hasValue()) {
113  diffConstant = diff.getValue();
114  assert(lbDivisor > 0);
115  } else {
116  // If no constant bound is found, then it can always be bound by the
117  // memref's dim size if the latter has a constant size along this dim.
118  auto dimSize = memRefType.getDimSize(d);
119  if (dimSize == -1)
120  return None;
121  diffConstant = dimSize;
122  // Lower bound becomes 0.
123  lb.resize(cst.getNumSymbolIds() + 1, 0);
124  lbDivisor = 1;
125  }
126  numElements *= diffConstant;
127  if (lbs) {
128  lbs->push_back(lb);
129  assert(lbDivisors && "both lbs and lbDivisor or none");
130  lbDivisors->push_back(lbDivisor);
131  }
132  if (shape) {
133  shape->push_back(diffConstant);
134  }
135  }
136  return numElements;
137 }

144 /// Computes the memory region accessed by this memref with the region
145 /// represented as constraints symbolic/parametric in 'loopDepth' loops
146 /// surrounding opInst and any additional Function symbols.
147 // For example, the memref region for this load operation at loopDepth = 1 will
148 // be as below:
149 //
150 // affine.for %i = 0 to 32 {
151 // affine.for %ii = %i to (d0) -> (d0 + 8) (%i) {
152 // load %A[%ii]
153 // }
154 // }
155 //
156 // region: {memref = %A, write = false, {%i <= m0 <= %i + 7} }
157 // The last field is a 2-d FlatAffineConstraints symbolic in %i.
158 //
159 // TODO(bondhugula): extend this to any other memref dereferencing ops
160 // (dma_start, dma_wait).
161 LogicalResult MemRefRegion::compute(Operation *op, unsigned loopDepth,
162  ComputationSliceState *sliceState,
163  bool addMemRefDimBounds) {
164  assert((isa<AffineLoadOp>(op) || isa<AffineStoreOp>(op)) &&
165  "affine load/store op expected");
166 
167  MemRefAccess access(op);
168  memref = access.memref;
169  write = access.isStore();
170 
171  unsigned rank = access.getRank();
172 
173  LLVM_DEBUG(llvm::dbgs() << "MemRefRegion::compute: " << *op
174  << "depth: " << loopDepth << "\n";);
175 
176  if (rank == 0) {
177  SmallVector<AffineForOp, 4> ivs;
178  getLoopIVs(*op, &ivs);
179  SmallVector<Value, 8> regionSymbols;
180  extractForInductionVars(ivs, &regionSymbols);
181  // A rank 0 memref has a 0-d region.
182  cst.reset(rank, loopDepth, 0, regionSymbols);
183  return success();
184  }
185 
186  // Build the constraints for this region.
187  AffineValueMap accessValueMap;
188  access.getAccessMap(&accessValueMap);
189  AffineMap accessMap = accessValueMap.getAffineMap();
190 
191  unsigned numDims = accessMap.getNumDims();
192  unsigned numSymbols = accessMap.getNumSymbols();
193  unsigned numOperands = accessValueMap.getNumOperands();
194  // Merge operands with slice operands.
195  SmallVector<Value, 4> operands;
196  operands.resize(numOperands);
197  for (unsigned i = 0; i < numOperands; ++i)
198  operands[i] = accessValueMap.getOperand(i);
199 
200  if (sliceState != nullptr) {
201  operands.reserve(operands.size() + sliceState->lbOperands[0].size());
202  // Append slice operands to 'operands' as symbols.
203  for (auto extraOperand : sliceState->lbOperands[0]) {
204  if (!llvm::is_contained(operands, extraOperand)) {
205  operands.push_back(extraOperand);
206  numSymbols++;
207  }
208  }
209  }
210  // We'll first associate the dims and symbols of the access map to the dims
211  // and symbols resp. of cst. This will change below once cst is
212  // fully constructed out.
213  cst.reset(numDims, numSymbols, 0, operands);
214 
215  // Add equality constraints.
216  // Add inequalities for loop lower/upper bounds.
217  for (unsigned i = 0; i < numDims + numSymbols; ++i) {
218  auto operand = operands[i];
219  if (auto loop = getForInductionVarOwner(operand)) {
220  // Note that cst can now have more dimensions than accessMap if the
221  // bounds expressions involve outer loops or other symbols.
222  // TODO(bondhugula): rewrite this to use getInstIndexSet; this way
223  // conditionals will be handled when the latter supports it.
224  if (failed(cst.addAffineForOpDomain(loop)))
225  return failure();
226  } else {
227  // Has to be a valid symbol.
228  auto symbol = operand;
229  assert(isValidSymbol(symbol));
230  // Check if the symbol is a constant.
231  if (auto *op = symbol.getDefiningOp()) {
232  if (auto constOp = dyn_cast<ConstantIndexOp>(op)) {
233  cst.setIdToConstant(symbol, constOp.getValue());
234  }
235  }
236  }
237  }
238 
239  // Add lower/upper bounds on loop IVs using bounds from 'sliceState'.
240  if (sliceState != nullptr) {
241  // Add dim and symbol slice operands.
242  for (auto operand : sliceState->lbOperands[0]) {
243  cst.addInductionVarOrTerminalSymbol(operand);
244  }
245  // Add upper/lower bounds from 'sliceState' to 'cst'.
246  LogicalResult ret =
247  cst.addSliceBounds(sliceState->ivs, sliceState->lbs, sliceState->ubs,
248  sliceState->lbOperands[0]);
249  assert(succeeded(ret) &&
250  "should not fail as we never have semi-affine slice maps");
251  (void)ret;
252  }
253 
254  // Add access function equalities to connect loop IVs to data dimensions.
255  if (failed(cst.composeMap(&accessValueMap))) {
256  op->emitError("getMemRefRegion: compose affine map failed");
257  LLVM_DEBUG(accessValueMap.getAffineMap().dump());
258  return failure();
259  }
260 
261  // Set all identifiers appearing after the first 'rank' identifiers as
262  // symbolic identifiers - so that the ones corresponding to the memref
263  // dimensions are the dimensional identifiers for the memref region.
264  cst.setDimSymbolSeparation(cst.getNumDimAndSymbolIds() - rank);
265 
266  // Eliminate any loop IVs other than the outermost 'loopDepth' IVs, on which
267  // this memref region is symbolic.
268  SmallVector<AffineForOp, 4> enclosingIVs;
269  getLoopIVs(*op, &enclosingIVs);
270  assert(loopDepth <= enclosingIVs.size() && "invalid loop depth");
271  enclosingIVs.resize(loopDepth);
272  SmallVector<Value, 4> ids;
273  cst.getIdValues(cst.getNumDimIds(), cst.getNumDimAndSymbolIds(), &ids);
274  for (auto id : ids) {
275  AffineForOp iv;
276  if ((iv = getForInductionVarOwner(id)) &&
277  llvm::is_contained(enclosingIVs, iv) == false) {
278  cst.projectOut(id);
279  }
280  }
281 
282  // Project out any local variables (these would have been added for any
283  // mod/divs).
284  cst.projectOut(cst.getNumDimAndSymbolIds(), cst.getNumLocalIds());
285 
286  // Constant fold any symbolic identifiers.
287  cst.constantFoldIdRange(/*pos=*/cst.getNumDimIds(),
288  /*num=*/cst.getNumSymbolIds());
289 
290  assert(cst.getNumDimIds() == rank && "unexpected MemRefRegion format");
291 
292  // Add upper/lower bounds for each memref dimension with static size
293  // to guard against potential over-approximation from projection.
294  // TODO(andydavis) Support dynamic memref dimensions.
295  if (addMemRefDimBounds) {
296  auto memRefType = memref.getType().cast<MemRefType>();
297  for (unsigned r = 0; r < rank; r++) {
298  cst.addConstantLowerBound(r, 0);
299  int64_t dimSize = memRefType.getDimSize(r);
300  if (ShapedType::isDynamic(dimSize))
301  continue;
302  cst.addConstantUpperBound(r, dimSize - 1);
303  }
304  }
305 
306  LLVM_DEBUG(llvm::dbgs() << "Memory region:\n");
307  LLVM_DEBUG(cst.dump());
308  return success();
309 }

370 template <typename LoadOrStoreOpPointer>
371 LogicalResult mlir::boundCheckLoadOrStoreOp(LoadOrStoreOpPointer loadOrStoreOp,
372  bool emitError) {
373  static_assert(std::is_same<LoadOrStoreOpPointer, AffineLoadOp>::value ||
374  std::is_same<LoadOrStoreOpPointer, AffineStoreOp>::value,
375  "argument should be either a AffineLoadOp or a AffineStoreOp");
376 
377  Operation *opInst = loadOrStoreOp.getOperation();
378  MemRefRegion region(opInst->getLoc());
379  if (failed(region.compute(opInst, /*loopDepth=*/0, /*sliceState=*/nullptr,
380  /*addMemRefDimBounds=*/false)))
381  return success();
382 
383  LLVM_DEBUG(llvm::dbgs() << "Memory region");
384  LLVM_DEBUG(region.getConstraints()->dump());
385 
386  bool outOfBounds = false;
387  unsigned rank = loadOrStoreOp.getMemRefType().getRank();
388 
389  // For each dimension, check for out of bounds.
390  for (unsigned r = 0; r < rank; r++) {
391  FlatAffineConstraints ucst(*region.getConstraints());
392 
393  // Intersect memory region with constraint capturing out of bounds (both out
394  // of upper and out of lower), and check if the constraint system is
395  // feasible. If it is, there is at least one point out of bounds.
396  SmallVector<int64_t, 4> ineq(rank + 1, 0);
397  int64_t dimSize = loadOrStoreOp.getMemRefType().getDimSize(r);
398  // TODO(bondhugula): handle dynamic dim sizes.
399  if (dimSize == -1)
400  continue;
401 
402  // Check for overflow: d_i >= memref dim size.
403  ucst.addConstantLowerBound(r, dimSize);
404  outOfBounds = !ucst.isEmpty();
405  if (outOfBounds && emitError) {
406  loadOrStoreOp.emitOpError()
407  << "memref out of upper bound access along dimension #" << (r + 1);
408  }
409 
410  // Check for a negative index.
411  FlatAffineConstraints lcst(*region.getConstraints());
412  std::fill(ineq.begin(), ineq.end(), 0);
413  // d_i <= -1;
414  lcst.addConstantUpperBound(r, -1);
415  outOfBounds = !lcst.isEmpty();
416  if (outOfBounds && emitError) {
417  loadOrStoreOp.emitOpError()
418  << "memref out of lower bound access along dimension #" << (r + 1);
419  }
420  }
421  return failure(outOfBounds);
422 }

430 // Returns in 'positions' the Block positions of 'op' in each ancestor
431 // Block from the Block containing operation, stopping at 'limitBlock'.
432 static void findInstPosition(Operation *op, Block *limitBlock,
433  SmallVectorImpl<unsigned> *positions) {
434  Block *block = op->getBlock();
435  while (block != limitBlock) {
436  // FIXME: This algorithm is unnecessarily O(n) and should be improved to not
437  // rely on linear scans.
438  int instPosInBlock = std::distance(block->begin(), op->getIterator());

440  op = block->getParentOp();
441  block = op->getBlock();
442  }
443  std::reverse(positions->begin(), positions->end());
444 }

446 // Returns the Operation in a possibly nested set of Blocks, where the
447 // position of the operation is represented by 'positions', which has a
448 // Block position for each level of nesting.
449 static Operation *getInstAtPosition(ArrayRef<unsigned> positions,
450  unsigned level, Block *block) {
451  unsigned i = 0;
452  for (auto &op : *block) {
453  if (i != positions[level]) {
454  ++i;
455  continue;
456  }
457  if (level == positions.size() - 1)
458  return &op;
459  if (auto childAffineForOp = dyn_cast<AffineForOp>(op))
460  return getInstAtPosition(positions, level + 1,
461  childAffineForOp.getBody());
462 
463  for (auto &region : op.getRegions()) {
464  for (auto &b : region)
465  if (auto *ret = getInstAtPosition(positions, level + 1, &b))
466  return ret;
467  }
468  return nullptr;
469  }
470  return nullptr;
471 }

488 // Returns the innermost common loop depth for the set of operations in 'ops'.
489 // TODO(andydavis) Move this to LoopUtils.
490 static unsigned
491 getInnermostCommonLoopDepth(ArrayRef<Operation *> ops,
492  SmallVectorImpl<AffineForOp> &surroundingLoops) {
493  unsigned numOps = ops.size();
494  assert(numOps > 0);
495 
496  std::vector<SmallVector<AffineForOp, 4>> loops(numOps);
497  unsigned loopDepthLimit = std::numeric_limits<unsigned>::max();
498  for (unsigned i = 0; i < numOps; ++i) {

517 /// Computes in 'sliceUnion' the union of all slice bounds computed at
518 /// 'loopDepth' between all dependent pairs of ops in 'opsA' and 'opsB'.
519 /// Returns 'Success' if union was computed, 'failure' otherwise.
520 LogicalResult mlir::computeSliceUnion(ArrayRef<Operation *> opsA,
521  ArrayRef<Operation *> opsB,
522  unsigned loopDepth,
523  unsigned numCommonLoops,
524  bool isBackwardSlice,
525  ComputationSliceState *sliceUnion) {
526  // Compute the union of slice bounds between all pairs in 'opsA' and
527  // 'opsB' in 'sliceUnionCst'.
528  FlatAffineConstraints sliceUnionCst;
529  assert(sliceUnionCst.getNumDimAndSymbolIds() == 0);
530  std::vector<std::pair<Operation *, Operation *>> dependentOpPairs;
531  for (unsigned i = 0, numOpsA = opsA.size(); i < numOpsA; ++i) {
532  MemRefAccess srcAccess(opsA[i]);
533  for (unsigned j = 0, numOpsB = opsB.size(); j < numOpsB; ++j) {
534  MemRefAccess dstAccess(opsB[j]);
535  if (srcAccess.memref != dstAccess.memref)
536  continue;
537  // Check if 'loopDepth' exceeds nesting depth of src/dst ops.
538  if ((!isBackwardSlice && loopDepth > getNestingDepth(*opsA[i])) ||
539  (isBackwardSlice && loopDepth > getNestingDepth(*opsB[j]))) {
540  LLVM_DEBUG(llvm::dbgs() << "Invalid loop depth\n.");
541  return failure();
542  }
543 
544  bool readReadAccesses = isa<AffineLoadOp>(srcAccess.opInst) &&
545  isa<AffineLoadOp>(dstAccess.opInst);
546  FlatAffineConstraints dependenceConstraints;
547  // Check dependence between 'srcAccess' and 'dstAccess'.
548  DependenceResult result = checkMemrefAccessDependence(
549  srcAccess, dstAccess, /*loopDepth=*/numCommonLoops + 1,
550  &dependenceConstraints, /*dependenceComponents=*/nullptr,
551  /*allowRAR=*/readReadAccesses);
552  if (result.value == DependenceResult::Failure) {
553  LLVM_DEBUG(llvm::dbgs() << "Dependence check failed\n.");
554  return failure();
555  }
556  if (result.value == DependenceResult::NoDependence)
557  continue;
558  dependentOpPairs.push_back({opsA[i], opsB[j]});
559 
560  // Compute slice bounds for 'srcAccess' and 'dstAccess'.
561  ComputationSliceState tmpSliceState;
562  mlir::getComputationSliceState(opsA[i], opsB[j], &dependenceConstraints,
563  loopDepth, isBackwardSlice,
564  &tmpSliceState);
565 
566  if (sliceUnionCst.getNumDimAndSymbolIds() == 0) {
567  // Initialize 'sliceUnionCst' with the bounds computed in previous step.
568  if (failed(tmpSliceState.getAsConstraints(&sliceUnionCst))) {
569  LLVM_DEBUG(llvm::dbgs()
570  << "Unable to compute slice bound constraints\n.");
571  return failure();
572  }
573  assert(sliceUnionCst.getNumDimAndSymbolIds() > 0);
574  continue;
575  }
576 
577  // Compute constraints for 'tmpSliceState' in 'tmpSliceCst'.
578  FlatAffineConstraints tmpSliceCst;
579  if (failed(tmpSliceState.getAsConstraints(&tmpSliceCst))) {
580  LLVM_DEBUG(llvm::dbgs()
581  << "Unable to compute slice bound constraints\n.");
582  return failure();
583  }
584 
585  // Align coordinate spaces of 'sliceUnionCst' and 'tmpSliceCst' if needed.
586  if (!sliceUnionCst.areIdsAlignedWithOther(tmpSliceCst)) {
587 
588  // Pre-constraint id alignment: record loop IVs used in each constraint
589  // system.
590  SmallPtrSet<Value, 8> sliceUnionIVs;
591  for (unsigned k = 0, l = sliceUnionCst.getNumDimIds(); k < l; ++k)
592  sliceUnionIVs.insert(sliceUnionCst.getIdValue(k));

683  bool isBackwardSlice, ComputationSliceState *sliceState) {

698  unsigned pos = isBackwardSlice ? numSrcLoopIVs + loopDepth : loopDepth;

700  isBackwardSlice ? numDstLoopIVs - loopDepth : numSrcLoopIVs - loopDepth;
701  dependenceConstraints->projectOut(pos, num);

710  sliceState->lbs.resize(numSliceLoopIVs, AffineMap());
711  sliceState->ubs.resize(numSliceLoopIVs, AffineMap());

714  dependenceConstraints->getSliceBounds(offset, numSliceLoopIVs,
715  depSourceOp->getContext(),

719  SmallVector<Value, 4> sliceBoundOperands;
720  unsigned numDimsAndSymbols = dependenceConstraints->getNumDimAndSymbolIds();
721  for (unsigned i = 0; i < numDimsAndSymbols; ++i) {
722  if (i < offset || i >= offset + numSliceLoopIVs) {
723  sliceBoundOperands.push_back(dependenceConstraints->getIdValue(i));

728  // canonicalization.
729  sliceState->lbOperands.resize(numSliceLoopIVs, sliceBoundOperands);
730  sliceState->ubOperands.resize(numSliceLoopIVs, sliceBoundOperands);

747  // using 'kSliceFusionBarrierAttrName'.

765 /// updates the slice loop bounds with any non-null bound maps specified in
766 /// 'sliceState', and inserts this slice into the loop nest surrounding
767 /// 'dstOpInst' at loop depth 'dstLoopDepth'.
768 // TODO(andydavis,bondhugula): extend the slicing utility to compute slices that
769 // aren't necessarily a one-to-one relation b/w the source and destination. The
770 // relation between the source and destination could be many-to-many in general.
771 // TODO(andydavis,bondhugula): the slice computation is incorrect in the cases
772 // where the dependence from the source to the destination does not cover the

775 // solution.

778  unsigned dstLoopDepth,
779  ComputationSliceState *sliceState) {

782  getLoopIVs(*srcOpInst, &srcLoopIVs);
783  unsigned numSrcLoopIVs = srcLoopIVs.size();
784 
785  // Get loop nest surrounding dst operation.
786  SmallVector<AffineForOp, 4> dstLoopIVs;
787  getLoopIVs(*dstOpInst, &dstLoopIVs);
788  unsigned dstLoopIVsSize = dstLoopIVs.size();

790  dstOpInst->emitError("invalid destination loop depth");
791  return AffineForOp();
792  }
793 
794  // Find the op block positions of 'srcOpInst' within 'srcLoopIVs'.
795  SmallVector<unsigned, 4> positions;
796  // TODO(andydavis): This code is incorrect since srcLoopIVs can be 0-d.
797  findInstPosition(srcOpInst, srcLoopIVs[0].getOperation()->getBlock(),
798  &positions);
799 
800  // Clone src loop nest and insert it a the beginning of the operation block
801  // of the loop at 'dstLoopDepth' in 'dstLoopIVs'.
802  auto dstAffineForOp = dstLoopIVs[dstLoopDepth - 1];
803  OpBuilder b(dstAffineForOp.getBody(), dstAffineForOp.getBody()->begin());
804  auto sliceLoopNest =
805  cast<AffineForOp>(b.clone(*srcLoopIVs[0].getOperation()));
806

808  getInstAtPosition(positions, /*level=*/0, sliceLoopNest.getBody());
809  // Get loop nest surrounding 'sliceInst'.
810  SmallVector<AffineForOp, 4> sliceSurroundingLoops;
811  getLoopIVs(*sliceInst, &sliceSurroundingLoops);
812 
813  // Sanity check.
814  unsigned sliceSurroundingLoopsSize = sliceSurroundingLoops.size();
815  (void)sliceSurroundingLoopsSize;
816  assert(dstLoopDepth + numSrcLoopIVs >= sliceSurroundingLoopsSize);
817  unsigned sliceLoopLimit = dstLoopDepth + numSrcLoopIVs;
818  (void)sliceLoopLimit;
819  assert(sliceLoopLimit >= sliceSurroundingLoopsSize);

823  auto forOp = sliceSurroundingLoops[dstLoopDepth + i];

825  forOp.setLowerBound(sliceState->lbOperands[i], lbMap);
826  if (AffineMap ubMap = sliceState->ubs[i])
827  forOp.setUpperBound(sliceState->ubOperands[i], ubMap);
828  }
829  return sliceLoopNest;
830 }

910  Block::iterator start,
911  Block::iterator end,
912  int memorySpace) {
913  SmallDenseMap<Value, std::unique_ptr<MemRefRegion>, 4> regions;
914 
915  // Walk this 'affine.for' operation to gather all memory regions.
916  auto result = block.walk(start, end, [&](Operation *opInst) -> WalkResult {
917  if (!isa<AffineLoadOp>(opInst) && !isa<AffineStoreOp>(opInst)) {
918  // Neither load nor a store op.
919  return WalkResult::advance();
920  }
921 
922  // Compute the memref region symbolic in any IVs enclosing this block.
923  auto region = std::make_unique<MemRefRegion>(opInst->getLoc());
924  if (failed(
925  region->compute(opInst,
926  /*loopDepth=*/getNestingDepth(*block.begin())))) {
927  return opInst->emitError("error obtaining memory region\n");
928  }

931  if (it == regions.end()) {
932  regions[region->memref] = std::move(region);
933  } else if (failed(it->second->unionBoundingBox(*region))) {
934  return opInst->emitWarning(
935  "getMemoryFootprintBytes: unable to perform a union on a memory "
936  "region");
937  }
938  return WalkResult::advance();
939  });
940  if (result.wasInterrupted())
941  return None;

944  for (const auto &region : regions) {

946  if (!size.hasValue())
947  return None;
948  totalSizeInBytes += size.getValue();
949  }
