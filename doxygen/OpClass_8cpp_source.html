30 void tblgen::OpMethodSignature::writeDefTo(raw_ostream &os,
31  StringRef namePrefix) const {
32  // We need to remove the default values for parameters in method definition.
33  // TODO(antiagainst): We are using '=' and ',' as delimiters for parameter
34  // initializers. This is incorrect for initializer list with more than one
35  // element. Change to a more robust approach.
36  auto removeParamDefaultValue = [](StringRef params) {
37  std::string result;
38  std::pair<StringRef, StringRef> parts;
39  while (!params.empty()) {
40  parts = params.split("=");
41  result.append(result.empty() ? "" : ", ");
42  result.append(parts.first);
43  params = parts.second.split(",").second;
44  }
45  return result;
46  };
47 
48  os << returnType << (elideSpaceAfterType(returnType) ? "" : " ") << namePrefix
49  << (namePrefix.empty() ? "" : "::") << methodName << "("
50  << removeParamDefaultValue(parameters) << ")";
51 }
