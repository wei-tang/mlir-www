51 /// ComputationSliceState aggregates loop IVs, loop bound AffineMaps and their
52 /// associated operands for a set of loops within a loop nest (typically the
53 /// set of loops surrounding a store operation). Loop bound AffineMaps which
54 /// are non-null represent slices of that loop's iteration space.
55 struct ComputationSliceState {
56  // List of sliced loop IVs (ordered from outermost to innermost).
57  // EX: 'ivs[i]' has lower bound 'lbs[i]' and upper bound 'ubs[i]'.
58  SmallVector<Value, 4> ivs;
59  // List of lower bound AffineMaps.
60  SmallVector<AffineMap, 4> lbs;
61  // List of upper bound AffineMaps.
62  SmallVector<AffineMap, 4> ubs;
63  // List of lower bound operands (lbOperands[i] are used by 'lbs[i]').
64  std::vector<SmallVector<Value, 4>> lbOperands;
65  // List of upper bound operands (ubOperands[i] are used by 'ubs[i]').
66  std::vector<SmallVector<Value, 4>> ubOperands;
67  // Slice loop nest insertion point in target loop nest.
68  Block::iterator insertPoint;
69  // Adds to 'cst' with constraints which represent the slice bounds on 'ivs'
70  // in 'this'. Specifically, the values in 'ivs' are added to 'cst' as dim
71  // identifiers and the values in 'lb/ubOperands' are added as symbols.
72  // Constraints are added for all loop IV bounds (dim or symbol), and
73  // constraints are added for slice bounds in 'lbs'/'ubs'.
74  // Returns failure if we cannot add loop bounds because of unsupported cases.
75  LogicalResult getAsConstraints(FlatAffineConstraints *cst);
76 
77  // Clears all bounds and operands in slice state.
78  void clearBounds();
79 };

81 /// Computes the computation slice loop bounds for one loop nest as affine maps
82 /// of the other loop nest's IVs and symbols, using 'dependenceConstraints'
83 /// computed between 'depSourceAccess' and 'depSinkAccess'.
84 /// If 'isBackwardSlice' is true, a backwards slice is computed in which the
85 /// slice bounds of loop nest surrounding 'depSourceAccess' are computed in
86 /// terms of loop IVs and symbols of the loop nest surrounding 'depSinkAccess'
87 /// at 'loopDepth'.
88 /// If 'isBackwardSlice' is false, a forward slice is computed in which the
89 /// slice bounds of loop nest surrounding 'depSinkAccess' are computed in terms
90 /// of loop IVs and symbols of the loop nest surrounding 'depSourceAccess' at
91 /// 'loopDepth'.
92 /// The slice loop bounds and associated operands are returned in 'sliceState'.
93 //
94 // Backward slice example:
95 //
96 // affine.for %i0 = 0 to 10 {
97 // affine.store %cst, %0[%i0] : memref<100xf32> // 'depSourceAccess'
98 // }
99 // affine.for %i1 = 0 to 10 {
100 // %v = affine.load %0[%i1] : memref<100xf32> // 'depSinkAccess'
101 // }
102 //
103 // // Backward computation slice of loop nest '%i0'.
104 // affine.for %i0 = (d0) -> (d0)(%i1) to (d0) -> (d0 + 1)(%i1) {
105 // affine.store %cst, %0[%i0] : memref<100xf32> // 'depSourceAccess'
106 // }
107 //
108 // Forward slice example:
109 //
110 // affine.for %i0 = 0 to 10 {
111 // affine.store %cst, %0[%i0] : memref<100xf32> // 'depSourceAccess'
112 // }
113 // affine.for %i1 = 0 to 10 {
114 // %v = affine.load %0[%i1] : memref<100xf32> // 'depSinkAccess'
115 // }
116 //
117 // // Forward computation slice of loop nest '%i1'.
118 // affine.for %i1 = (d0) -> (d0)(%i0) to (d0) -> (d0 + 1)(%i0) {
119 // %v = affine.load %0[%i1] : memref<100xf32> // 'depSinkAccess'
120 // }
121 //
122 void getComputationSliceState(Operation *depSourceOp, Operation *depSinkOp,
123  FlatAffineConstraints *dependenceConstraints,
124  unsigned loopDepth, bool isBackwardSlice,
125  ComputationSliceState *sliceState);

128 /// 'loopDepth' between all dependent pairs of ops in 'opsA' and 'opsB'.
129 /// The parameter 'numCommonLoops' is the number of loops common to the
130 /// operations in 'opsA' and 'opsB'.

159 /// A region of a memref's data space; this is typically constructed by
160 /// analyzing load/store op's on this memref and the index space of loops
161 /// surrounding such op's.
162 // For example, the memref region for a load operation at loop depth = 1:
163 //
164 // affine.for %i = 0 to 32 {
165 // affine.for %ii = %i to (d0) -> (d0 + 8) (%i) {
166 // affine.load %A[%ii]
167 // }
168 // }
169 //
170 // Region: {memref = %A, write = false, {%i <= m0 <= %i + 7} }
171 // The last field is a 2-d FlatAffineConstraints symbolic in %i.
172 //
173 struct MemRefRegion {
174  explicit MemRefRegion(Location loc) : loc(loc) {}
175 
176  /// Computes the memory region accessed by this memref with the region
177  /// represented as constraints symbolic/parametric in 'loopDepth' loops
178  /// surrounding opInst. The computed region's 'cst' field has exactly as many
179  /// dimensional identifiers as the rank of the memref, and *potentially*
180  /// additional symbolic identifiers which could include any of the loop IVs
181  /// surrounding opInst up until 'loopDepth' and another additional Function
182  /// symbols involved with the access (for eg., those appear in affine.apply's,
183  /// loop bounds, etc.). If 'sliceState' is non-null, operands from
184  /// 'sliceState' are added as symbols, and the following constraints are added
185  /// to the system:
186  /// *) Inequality constraints which represent loop bounds for 'sliceState'
187  /// operands which are loop IVS (these represent the destination loop IVs
188  /// of the slice, and are added as symbols to MemRefRegion's constraint
189  /// system).
190  /// *) Inequality constraints for the slice bounds in 'sliceState', which
191  /// represent the bounds on the loop IVs in this constraint system w.r.t
192  /// to slice operands (which correspond to symbols).
193  /// If 'addMemRefDimBounds' is true, constant upper/lower bounds
194  /// [0, memref.getDimSize(i)) are added for each MemRef dimension 'i'.
195  ///
196  /// For example, the memref region for this operation at loopDepth = 1 will
197  /// be:
198  ///
199  /// affine.for %i = 0 to 32 {
200  /// affine.for %ii = %i to (d0) -> (d0 + 8) (%i) {
201  /// load %A[%ii]
202  /// }
203  /// }
204  ///
205  /// {memref = %A, write = false, {%i <= m0 <= %i + 7} }
206  /// The last field is a 2-d FlatAffineConstraints symbolic in %i.
207  ///
208  LogicalResult compute(Operation *op, unsigned loopDepth,
209  ComputationSliceState *sliceState = nullptr,
210  bool addMemRefDimBounds = true);
211 
212  FlatAffineConstraints *getConstraints() { return &cst; }
213  const FlatAffineConstraints *getConstraints() const { return &cst; }
214  bool isWrite() const { return write; }
215  void setWrite(bool flag) { write = flag; }

217  /// Returns a constant upper bound on the number of elements in this region if
218  /// bounded by a known constant (always possible for static shapes), None
219  /// otherwise. Note that the symbols of the region are treated specially,
220  /// i.e., the returned bounding constant holds for *any given* value of the
221  /// symbol identifiers. The 'shape' vector is set to the corresponding
222  /// dimension-wise bounds major to minor. We use int64_t instead of uint64_t
223  /// since index types can be at most int64_t.
224  Optional<int64_t> getConstantBoundingSizeAndShape(
225  SmallVectorImpl<int64_t> *shape = nullptr,
226  std::vector<SmallVector<int64_t, 4>> *lbs = nullptr,
227  SmallVectorImpl<int64_t> *lbDivisors = nullptr) const;
228 
229  /// A wrapper around FlatAffineConstraints::getConstantBoundOnDimSize(). 'pos'
230  /// corresponds to the position of the memref shape's dimension (major to
231  /// minor) which matches 1:1 with the dimensional identifier positions in
232  //'cst'.
233  Optional<int64_t>
234  getConstantBoundOnDimSize(unsigned pos,
235  SmallVectorImpl<int64_t> *lb = nullptr,
236  int64_t *lbFloorDivisor = nullptr) const {
237  assert(pos < getRank() && "invalid position");
238  return cst.getConstantBoundOnDimSize(pos, lb);
239  }

248  unsigned getRank() const;

251  Value memref;

253  /// Read or write.
254  bool write;
255 
256  /// If there is more than one load/store op associated with the region, the
257  /// location information would correspond to one of those op's.
258  Location loc;
259 
260  /// Region (data space) of the memref accessed. This set will thus have at
261  /// least as many dimensional identifiers as the shape dimensionality of the
262  /// memref, and these are the leading dimensions of the set appearing in that
263  /// order (major to minor / outermost to innermost). There may be additional

265  /// and thus the region is symbolic in the outer surrounding loops at that

267  // TODO(bondhugula): Replace this to exploit HyperRectangularSet.
268  FlatAffineConstraints cst;
