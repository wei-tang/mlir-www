24  /// A lookup key for derived instances of storage objects.
25  struct LookupKey {
26  /// The known derived kind for the storage.
27  unsigned kind;
28 
29  /// The known hash value of the key.
30  unsigned hashValue;
31 
32  /// An equality function for comparing with an existing storage instance.
33  function_ref<bool(const BaseStorage *)> isEqual;
34  };

36  /// A utility wrapper object representing a hashed storage object. This class
37  /// contains a storage object and an existing computed hash value.
38  struct HashedStorage {
39  unsigned hashValue;
40  BaseStorage *storage;
41  };

43  /// Get or create an instance of a complex derived type.
44  BaseStorage *
45  getOrCreate(unsigned kind, unsigned hashValue,
46  function_ref<bool(const BaseStorage *)> isEqual,
47  function_ref<BaseStorage *(StorageAllocator &)> ctorFn) {
48  LookupKey lookupKey{kind, hashValue, isEqual};
49 
50  // Check for an existing instance in read-only mode.
51  {
52  llvm::sys::SmartScopedReader<true> typeLock(mutex);
53  auto it = storageTypes.find_as(lookupKey);
54  if (it != storageTypes.end())
55  return it->storage;
56  }
57 
58  // Acquire a writer-lock so that we can safely create the new type instance.
59  llvm::sys::SmartScopedWriter<true> typeLock(mutex);
60 
61  // Check for an existing instance again here, because another writer thread
62  // may have already created one.
63  auto existing = storageTypes.insert_as({}, lookupKey);
64  if (!existing.second)
65  return existing.first->storage;
66 
67  // Otherwise, construct and initialize the derived storage for this type
68  // instance.
69  BaseStorage *storage = initializeStorage(kind, ctorFn);
70  *existing.first = HashedStorage{hashValue, storage};
71  return storage;
72  }
73 
74  /// Get or create an instance of a simple derived type.
75  BaseStorage *
76  getOrCreate(unsigned kind,
77  function_ref<BaseStorage *(StorageAllocator &)> ctorFn) {
78  // Check for an existing instance in read-only mode.
79  {
80  llvm::sys::SmartScopedReader<true> typeLock(mutex);
81  auto it = simpleTypes.find(kind);
82  if (it != simpleTypes.end())
83  return it->second;
84  }
85 
86  // Acquire a writer-lock so that we can safely create the new type instance.
87  llvm::sys::SmartScopedWriter<true> typeLock(mutex);
88 
89  // Check for an existing instance again here, because another writer thread
90  // may have already created one.
91  auto &result = simpleTypes[kind];
92  if (result)
93  return result;
94 
95  // Otherwise, create and return a new storage instance.
96  return result = initializeStorage(kind, ctorFn);
97  }
98 
99  /// Erase an instance of a complex derived type.
100  void erase(unsigned kind, unsigned hashValue,
101  function_ref<bool(const BaseStorage *)> isEqual,
102  function_ref<void(BaseStorage *)> cleanupFn) {
103  LookupKey lookupKey{kind, hashValue, isEqual};
104 
105  // Acquire a writer-lock so that we can safely erase the type instance.
106  llvm::sys::SmartScopedWriter<true> typeLock(mutex);
107  auto existing = storageTypes.find_as(lookupKey);
108  if (existing == storageTypes.end())
109  return;
110 
111  // Cleanup the storage and remove it from the map.
112  cleanupFn(existing->storage);
113  storageTypes.erase(existing);
114  }

122  initializeStorage(unsigned kind,

124  BaseStorage *storage = ctorFn(allocator);

129  /// Storage info for derived TypeStorage objects.

131  static HashedStorage getEmptyKey() {

141  static unsigned getHashValue(LookupKey key) { return key.hashValue; }
142 
143  static bool isEqual(const HashedStorage &lhs, const HashedStorage &rhs) {
144  return lhs.storage == rhs.storage;
145  }
146  static bool isEqual(const LookupKey &lhs, const HashedStorage &rhs) {
147  if (isEqual(rhs, getEmptyKey()) || isEqual(rhs, getTombstoneKey()))
148  return false;
149  // If the lookup kind matches the kind of the storage, then invoke the
150  // equality function on the lookup key.
151  return lhs.kind == rhs.storage->getKind() && lhs.isEqual(rhs.storage);
152  }
153  };
