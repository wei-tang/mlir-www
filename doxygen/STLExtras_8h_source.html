28 /// An STL-style algorithm similar to std::for_each that applies a second
29 /// functor between every pair of elements.
30 ///
31 /// This provides the control flow logic to, for example, print a
32 /// comma-separated list:
33 /// \code
34 /// interleave(names.begin(), names.end(),
35 /// [&](StringRef name) { os << name; },
36 /// [&] { os << ", "; });
37 /// \endcode
38 template <typename ForwardIterator, typename UnaryFunctor,
39  typename NullaryFunctor,
40  typename = typename std::enable_if<
41  !std::is_constructible<StringRef, UnaryFunctor>::value &&
42  !std::is_constructible<StringRef, NullaryFunctor>::value>::type>
43 inline void interleave(ForwardIterator begin, ForwardIterator end,
44  UnaryFunctor each_fn, NullaryFunctor between_fn) {
45  if (begin == end)
46  return;
47  each_fn(*begin);
48  ++begin;
49  for (; begin != end; ++begin) {
50  between_fn();
51  each_fn(*begin);
52  }
53 }

57  !std::is_constructible<StringRef, UnaryFunctor>::value &&
58  !std::is_constructible<StringRef, NullaryFunctor>::value>::type>
59 inline void interleave(const Container &c, UnaryFunctor each_fn,
60  NullaryFunctor between_fn) {
61  interleave(c.begin(), c.end(), each_fn, between_fn);

141 /// A utility class used to implement an iterator that contains some base object
142 /// and an index. The iterator moves the index but keeps the base constant.
143 template <typename DerivedT, typename BaseT, typename T,
144  typename PointerT = T *, typename ReferenceT = T &>
145 class indexed_accessor_iterator
146  : public llvm::iterator_facade_base<DerivedT,
147  std::random_access_iterator_tag, T,
148  std::ptrdiff_t, PointerT, ReferenceT> {
149 public:
150  ptrdiff_t operator-(const indexed_accessor_iterator &rhs) const {
151  assert(base == rhs.base && "incompatible iterators");
152  return index - rhs.index;
153  }
154  bool operator==(const indexed_accessor_iterator &rhs) const {
155  return base == rhs.base && index == rhs.index;
156  }
157  bool operator<(const indexed_accessor_iterator &rhs) const {
158  assert(base == rhs.base && "incompatible iterators");
159  return index < rhs.index;
160  }
161 
162  DerivedT &operator+=(ptrdiff_t offset) {
163  this->index += offset;
164  return static_cast<DerivedT &>(*this);
165  }
166  DerivedT &operator-=(ptrdiff_t offset) {
167  this->index -= offset;
168  return static_cast<DerivedT &>(*this);
169  }
170 
171  /// Returns the current index of the iterator.
172  ptrdiff_t getIndex() const { return index; }
173 
174  /// Returns the current base of the iterator.
175  const BaseT &getBase() const { return base; }
176 
177 protected:
178  indexed_accessor_iterator(BaseT base, ptrdiff_t index)
179  : base(base), index(index) {}
180  BaseT base;
181  ptrdiff_t index;
182 };

185 /// The class represents the base of a range of indexed_accessor_iterators. It
186 /// provides support for many different range functionalities, e.g.
187 /// drop_front/slice/etc.. Derived range classes must implement the following
188 /// static methods:
189 /// * ReferenceT dereference_iterator(const BaseT &base, ptrdiff_t index)
190 /// - Dereference an iterator pointing to the base object at the given
191 /// index.
192 /// * BaseT offset_base(const BaseT &base, ptrdiff_t index)
193 /// - Return a new base that is offset from the provide base by 'index'
194 /// elements.
195 template <typename DerivedT, typename BaseT, typename T,
196  typename PointerT = T *, typename ReferenceT = T &>
197 class indexed_accessor_range_base {
198 public:
199  using RangeBaseT =
200  indexed_accessor_range_base<DerivedT, BaseT, T, PointerT, ReferenceT>;
201 
202  /// An iterator element of this range.
203  class iterator : public indexed_accessor_iterator<iterator, BaseT, T,
204  PointerT, ReferenceT> {
205  public:
206  // Index into this iterator, invoking a static method on the derived type.
207  ReferenceT operator*() const {
208  return DerivedT::dereference_iterator(this->getBase(), this->getIndex());
209  }
210 
211  private:
212  iterator(BaseT owner, ptrdiff_t curIndex)
213  : indexed_accessor_iterator<iterator, BaseT, T, PointerT, ReferenceT>(
214  owner, curIndex) {}
215 
216  /// Allow access to the constructor.
217  friend indexed_accessor_range_base<DerivedT, BaseT, T, PointerT,
218  ReferenceT>;
219  };
220 
221  indexed_accessor_range_base(iterator begin, iterator end)
222  : base(DerivedT::offset_base(begin.getBase(), begin.getIndex())),
223  count(end.getIndex() - begin.getIndex()) {}
224  indexed_accessor_range_base(const iterator_range<iterator> &range)
225  : indexed_accessor_range_base(range.begin(), range.end()) {}
226  indexed_accessor_range_base(BaseT base, ptrdiff_t count)
227  : base(base), count(count) {}
228 
229  iterator begin() const { return iterator(base, 0); }
230  iterator end() const { return iterator(base, count); }
231  ReferenceT operator[](unsigned index) const {
232  assert(index < size() && "invalid index for value range");
233  return DerivedT::dereference_iterator(base, index);
234  }
235 
236  /// Return the size of this range.
237  size_t size() const { return count; }
238 
239  /// Return if the range is empty.
240  bool empty() const { return size() == 0; }
241 
242  /// Drop the first N elements, and keep M elements.
243  DerivedT slice(size_t n, size_t m) const {
244  assert(n + m <= size() && "invalid size specifiers");
245  return DerivedT(DerivedT::offset_base(base, n), m);
246  }
247 
248  /// Drop the first n elements.
249  DerivedT drop_front(size_t n = 1) const {
250  assert(size() >= n && "Dropping more elements than exist");
251  return slice(n, size() - n);
252  }
253  /// Drop the last n elements.
254  DerivedT drop_back(size_t n = 1) const {
255  assert(size() >= n && "Dropping more elements than exist");
256  return DerivedT(base, size() - n);
257  }
258 
259  /// Take the first n elements.
260  DerivedT take_front(size_t n = 1) const {
261  return n < size() ? drop_back(size() - n)
262  : static_cast<const DerivedT &>(*this);
263  }
264 
265  /// Allow conversion to SmallVector if necessary.
266  /// TODO(riverriddle) Remove this when SmallVector accepts different range
267  /// types in its constructor.
268  template <typename SVT, unsigned N> operator SmallVector<SVT, N>() const {
269  return {begin(), end()};
270  }
271 
272 protected:
273  indexed_accessor_range_base(const indexed_accessor_range_base &) = default;
274  indexed_accessor_range_base(indexed_accessor_range_base &&) = default;
275  indexed_accessor_range_base &
276  operator=(const indexed_accessor_range_base &) = default;
277 
278  /// The base that owns the provided range of values.
279  BaseT base;
280  /// The size from the owning range.
281  ptrdiff_t count;
282 };
283 } // end namespace detail

285 /// This class provides an implementation of a range of
286 /// indexed_accessor_iterators where the base is not indexable. Ranges with
287 /// bases that are offsetable should derive from indexed_accessor_range_base
288 /// instead. Derived range classes are expected to implement the following
289 /// static method:
290 /// * ReferenceT dereference(const BaseT &base, ptrdiff_t index)
291 /// - Dereference an iterator pointing to a parent base at the given index.
292 template <typename DerivedT, typename BaseT, typename T,
293  typename PointerT = T *, typename ReferenceT = T &>
294 class indexed_accessor_range
295  : public detail::indexed_accessor_range_base<
296  DerivedT, std::pair<BaseT, ptrdiff_t>, T, PointerT, ReferenceT> {
297 public:
298  indexed_accessor_range(BaseT base, ptrdiff_t startIndex, ptrdiff_t count)
299  : detail::indexed_accessor_range_base<
300  DerivedT, std::pair<BaseT, ptrdiff_t>, T, PointerT, ReferenceT>(
301  std::make_pair(base, startIndex), count) {}
302  using detail::indexed_accessor_range_base<
303  DerivedT, std::pair<BaseT, ptrdiff_t>, T, PointerT,
304  ReferenceT>::indexed_accessor_range_base;
305 
306  /// Returns the current base of the range.
307  const BaseT &getBase() const { return this->base.first; }
308 
309  /// Returns the current start index of the range.
310  ptrdiff_t getStartIndex() const { return this->base.second; }
311 
312  /// See `detail::indexed_accessor_range_base` for details.
313  static std::pair<BaseT, ptrdiff_t>
314  offset_base(const std::pair<BaseT, ptrdiff_t> &base, ptrdiff_t index) {
315  // We encode the internal base as a pair of the derived base and a start
316  // index into the derived base.
317  return std::make_pair(base.first, base.second + index);
318  }
319  /// See `detail::indexed_accessor_range_base` for details.
320  static ReferenceT
321  dereference_iterator(const std::pair<BaseT, ptrdiff_t> &base,
322  ptrdiff_t index) {
323  return DerivedT::dereference(base.first, base.second + index);
324  }
325 };

384 /// Simplistic combination of 32-bit hash values into 32-bit hash values.
385 /// This function is taken from llvm/ADT/DenseMapInfo.h.
386 static inline unsigned llvm_combineHashValue(unsigned a, unsigned b) {
387  uint64_t key = (uint64_t)a << 32 | (uint64_t)b;
388  key += ~(key << 32);
389  key ^= (key >> 22);
390  key += ~(key << 13);
391  key ^= (key >> 8);
392  key += (key << 3);
393  key ^= (key >> 15);
394  key += ~(key << 27);
395  key ^= (key >> 31);
396  return (unsigned)key;
397 }
