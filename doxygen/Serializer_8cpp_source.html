47 /// A pre-order depth-first visitor function for processing basic blocks.
48 ///
49 /// Visits the basic blocks starting from the given `headerBlock` in pre-order
50 /// depth-first manner and calls `blockHandler` on each block. Skips handling
51 /// blocks in the `skipBlocks` list. If `skipHeader` is true, `blockHandler`
52 /// will not be invoked in `headerBlock` but still handles all `headerBlock`'s
53 /// successors.
54 ///
55 /// SPIR-V spec "2.16.1. Universal Validation Rules" requires that "the order
56 /// of blocks in a function must satisfy the rule that blocks appear before
57 /// all blocks they dominate." This can be achieved by a pre-order CFG
58 /// traversal algorithm. To make the serialization output more logical and
59 /// readable to human, we perform depth-first CFG traversal and delay the
60 /// serialization of the merge block and the continue block, if exists, until
61 /// after all other blocks have been processed.
62 static LogicalResult visitInPrettyBlockOrder(
63  Block *headerBlock, function_ref<LogicalResult(Block *)> blockHandler,
64  bool skipHeader = false, ArrayRef<Block *> skipBlocks = {}) {
65  llvm::df_iterator_default_set<Block *, 4> doneBlocks;
66  doneBlocks.insert(skipBlocks.begin(), skipBlocks.end());
67 
68  for (Block *block : llvm::depth_first_ext(headerBlock, doneBlocks)) {
69  if (skipHeader && block == headerBlock)
70  continue;
71  if (failed(blockHandler(block)))
72  return failure();
73  }
74  return success();
75 }

93 /// A SPIR-V binary module is a single linear stream of instructions; each
94 /// instruction is composed of 32-bit words with the layout:

413  /// Map from values used in OpPhi instructions to their offset in the
414  /// `functions` section.
415  ///
416  /// When processing a block with arguments, we need to emit OpPhi
417  /// instructions to record the predecessor block <id>s and the values they
418  /// send to the block in question. But it's not guaranteed all values are
419  /// visited and thus assigned result <id>s. So we need this list to capture
420  /// the offsets into `functions` where a value is used so that we can fix it
421  /// up later after processing all the blocks in a function.
422  ///
423  /// More concretely, say if we are visiting the following blocks:
424  ///
425  /// ```mlir
426  /// ^phi(%arg0: i32):
427  /// ...
428  /// ^parent1:
429  /// ...
430  /// spv.Branch ^phi(%val0: i32)
431  /// ^parent2:
432  /// ...
433  /// spv.Branch ^phi(%val1: i32)
434  /// ```
435  ///
436  /// When we are serializing the `^phi` block, we need to emit at the beginning
437  /// of the block OpPhi instructions which has the following parameters:
438  ///
439  /// OpPhi id-for-i32 id-for-%arg0 id-for-%val0 id-for-^parent1
440  /// id-for-%val1 id-for-^parent2
441  ///
442  /// But we don't know the <id> for %val0 and %val1 yet. One way is to visit
443  /// all the blocks twice and use the first visit to assign an <id> to each
444  /// value. But it's paying the overheads just for OpPhi emission. Instead,
445  /// we still visit the blocks once for emission. When we emit the OpPhi
446  /// instructions, we use 0 as a placeholder for the <id>s for %val0 and %val1.
447  /// At the same time, we record their offsets in the emitted binary (which is
448  /// placed inside `functions`) here. And then after emitting all blocks, we
449  /// replace the dummy <id> 0 with the real result <id> by overwriting
450  /// `functions[offset]`.
451  DenseMap<Value, SmallVector<size_t, 1>> deferredPhiValues;

458 LogicalResult Serializer::serialize() {
459  LLVM_DEBUG(llvm::dbgs() << "+++ starting serialization +++\n");
460 
461  if (failed(module.verify()))
462  return failure();
463 
464  // TODO(antiagainst): handle the other sections
465  processCapability();
466  processExtension();
467  processMemoryModel();
468 
469  // Iterate over the module body to serialize it. Assumptions are that there is
470  // only one basic block in the moduleOp
471  for (auto &op : module.getBlock()) {
472  if (failed(processOperation(&op))) {
473  return failure();
474  }
475  }
476 
477  LLVM_DEBUG(llvm::dbgs() << "+++ completed serialization +++\n");
478  return success();
479 }

481 void Serializer::collect(SmallVectorImpl<uint32_t> &binary) {
482  auto moduleSize = spirv::kHeaderWordCount + capabilities.size() +
483  extensions.size() + extendedSets.size() +
484  memoryModel.size() + entryPoints.size() +
485  executionModes.size() + decorations.size() +
486  typesGlobalValues.size() + functions.size();
487 
488  binary.clear();
489  binary.reserve(moduleSize);
490 
491  spirv::appendModuleHeader(binary, nextID);
492  binary.append(capabilities.begin(), capabilities.end());
493  binary.append(extensions.begin(), extensions.end());
494  binary.append(extendedSets.begin(), extendedSets.end());
495  binary.append(memoryModel.begin(), memoryModel.end());
496  binary.append(entryPoints.begin(), entryPoints.end());
497  binary.append(executionModes.begin(), executionModes.end());
498  binary.append(names.begin(), names.end());
499  binary.append(decorations.begin(), decorations.end());
500  binary.append(typesGlobalValues.begin(), typesGlobalValues.end());
501  binary.append(functions.begin(), functions.end());
502 }

610  auto decorationName = mlir::convertToCamelCase(attrName, true);
611  auto decoration = spirv::symbolizeDecoration(decorationName);
612  if (!decoration) {
613  return emitError(
614  loc, "non-argument attributes expected to have snake-case-ified "
615  "decoration name, unhandled attribute with name : ")
616  << attrName;
617  }
618  SmallVector<uint32_t, 1> args;
619  switch (decoration.getValue()) {
620  case spirv::Decoration::DescriptorSet:
621  case spirv::Decoration::Binding:
622  if (auto intAttr = attr.second.dyn_cast<IntegerAttr>()) {
623  args.push_back(intAttr.getValue().getZExtValue());
624  break;
625  }
626  return emitError(loc, "expected integer attribute for ") << attrName;
627  case spirv::Decoration::BuiltIn:
628  if (auto strAttr = attr.second.dyn_cast<StringAttr>()) {
629  auto enumVal = spirv::symbolizeBuiltIn(strAttr.getValue());
630  if (enumVal) {
631  args.push_back(static_cast<uint32_t>(enumVal.getValue()));
632  break;
633  }
634  return emitError(loc, "invalid ")
635  << attrName << " attribute " << strAttr.getValue();
636  }
637  return emitError(loc, "expected string attribute for ") << attrName;
638  default:
639  return emitError(loc, "unhandled decoration ") << decorationName;
640  }
641  return emitDecoration(resultID, decoration.getValue(), args);
642 }

681 LogicalResult Serializer::processFuncOp(FuncOp op) {
682  LLVM_DEBUG(llvm::dbgs() << "-- start function '" << op.getName() << "' --\n");
683  assert(functionHeader.empty() && functionBody.empty());
684 
685  uint32_t fnTypeID = 0;
686  // Generate type of the function.
687  processType(op.getLoc(), op.getType(), fnTypeID);
688 
689  // Add the function definition.
690  SmallVector<uint32_t, 4> operands;
691  uint32_t resTypeID = 0;
692  auto resultTypes = op.getType().getResults();
693  if (resultTypes.size() > 1) {
694  return op.emitError("cannot serialize function with multiple return types");
695  }
696  if (failed(processType(op.getLoc(),
697  (resultTypes.empty() ? getVoidType() : resultTypes[0]),
698  resTypeID))) {
699  return failure();
700  }
701  operands.push_back(resTypeID);
702  auto funcID = getOrCreateFunctionID(op.getName());
703  operands.push_back(funcID);
704  // TODO : Support other function control options.
705  operands.push_back(static_cast<uint32_t>(spirv::FunctionControl::None));
706  operands.push_back(fnTypeID);
707  encodeInstructionInto(functionHeader, spirv::Opcode::OpFunction, operands);
708 
709  // Add function name.
710  if (failed(processName(funcID, op.getName()))) {
711  return failure();
712  }
713 
714  // Declare the parameters.
715  for (auto arg : op.getArguments()) {
716  uint32_t argTypeID = 0;
717  if (failed(processType(op.getLoc(), arg.getType(), argTypeID))) {
718  return failure();
719  }
720  auto argValueID = getNextID();
721  valueIDMap[arg] = argValueID;
722  encodeInstructionInto(functionHeader, spirv::Opcode::OpFunctionParameter,
723  {argTypeID, argValueID});
724  }
725 
726  // Process the body.
727  if (op.isExternal()) {
728  return op.emitError("external function is unhandled");
729  }
730 
731  // Some instructions (e.g., OpVariable) in a function must be in the first
732  // block in the function. These instructions will be put in functionHeader.
733  // Thus, we put the label in functionHeader first, and omit it from the first
734  // block.
735  encodeInstructionInto(functionHeader, spirv::Opcode::OpLabel,
736  {getOrCreateBlockID(&op.front())});
737  processBlock(&op.front(), /*omitLabel=*/true);
738  if (failed(visitInPrettyBlockOrder(
739  &op.front(), [&](Block *block) { return processBlock(block); },
740  /*skipHeader=*/true))) {
741  return failure();
742  }
743 
744  // There might be OpPhi instructions who have value references needing to fix.
745  for (auto deferredValue : deferredPhiValues) {
746  Value value = deferredValue.first;
747  uint32_t id = getValueID(value);
748  LLVM_DEBUG(llvm::dbgs() << "[phi] fix reference of value " << value
749  << " to id = " << id << '\n');
750  assert(id && "OpPhi references undefined value!");
751  for (size_t offset : deferredValue.second)
752  functionBody[offset] = id;
753  }
754  deferredPhiValues.clear();
755 
756  LLVM_DEBUG(llvm::dbgs() << "-- completed function '" << op.getName()
757  << "' --\n");
758  // Insert OpFunctionEnd.
759  if (failed(encodeInstructionInto(functionBody, spirv::Opcode::OpFunctionEnd,
760  {}))) {
761  return failure();
762  }
763 
764  functions.append(functionHeader.begin(), functionHeader.end());
765  functions.append(functionBody.begin(), functionBody.end());
766  functionHeader.clear();
767  functionBody.clear();
768 
769  return success();
770 }

772 LogicalResult Serializer::processVariableOp(spirv::VariableOp op) {
773  SmallVector<uint32_t, 4> operands;
774  SmallVector<StringRef, 2> elidedAttrs;
775  uint32_t resultID = 0;
776  uint32_t resultTypeID = 0;
777  if (failed(processType(op.getLoc(), op.getType(), resultTypeID))) {
778  return failure();
779  }
780  operands.push_back(resultTypeID);
781  resultID = getNextID();
782  valueIDMap[op.getResult()] = resultID;
783  operands.push_back(resultID);
784  auto attr = op.getAttr(spirv::attributeName<spirv::StorageClass>());
785  if (attr) {
786  operands.push_back(static_cast<uint32_t>(

788  }
789  elidedAttrs.push_back(spirv::attributeName<spirv::StorageClass>());

800  if (llvm::any_of(elidedAttrs,
801  [&](StringRef elided) { return attr.first.is(elided); })) {

803  }

812 Serializer::processGlobalVariableOp(spirv::GlobalVariableOp varOp) {
813  // Get TypeID.
814  uint32_t resultTypeID = 0;

816  if (failed(processType(varOp.getLoc(), varOp.type(), resultTypeID))) {
817  return failure();
818  }

830  }

847  operands.push_back(static_cast<uint32_t>(varOp.storageClass()));

857  elidedAttrs.push_back("initializer");
858  }
859

861  operands))) {
862  elidedAttrs.push_back("initializer");
863  return failure();
864  }
865 
866  // Encode decorations.
867  for (auto attr : varOp.getAttrs()) {
868  if (llvm::any_of(elidedAttrs,
869  [&](StringRef elided) { return attr.first.is(elided); })) {
870  continue;
871  }
872  if (failed(processDecoration(varOp.getLoc(), resultID, attr))) {
873  return failure();
874  }
875  }
876  return success();
877 }

914 LogicalResult
915 Serializer::prepareBasicType(Location loc, Type type, uint32_t resultID,
916  spirv::Opcode &typeEnum,
917  SmallVectorImpl<uint32_t> &operands) {
918  if (isVoidType(type)) {
919  typeEnum = spirv::Opcode::OpTypeVoid;
920  return success();
921  }
922 
923  if (auto intType = type.dyn_cast<IntegerType>()) {
924  if (intType.getWidth() == 1) {
925  typeEnum = spirv::Opcode::OpTypeBool;
926  return success();
927  }
928 
929  typeEnum = spirv::Opcode::OpTypeInt;
930  operands.push_back(intType.getWidth());
931  // TODO(antiagainst): support unsigned integers
932  operands.push_back(1);
933  return success();
934  }
935 
936  if (auto floatType = type.dyn_cast<FloatType>()) {
937  typeEnum = spirv::Opcode::OpTypeFloat;
938  operands.push_back(floatType.getWidth());
939  return success();
940  }
941 
942  if (auto vectorType = type.dyn_cast<VectorType>()) {
943  uint32_t elementTypeID = 0;
944  if (failed(processType(loc, vectorType.getElementType(), elementTypeID))) {
945  return failure();
946  }
947  typeEnum = spirv::Opcode::OpTypeVector;
948  operands.push_back(elementTypeID);
949  operands.push_back(vectorType.getNumElements());
950  return success();
951  }
952 
953  if (auto arrayType = type.dyn_cast<spirv::ArrayType>()) {
954  typeEnum = spirv::Opcode::OpTypeArray;
955  uint32_t elementTypeID = 0;
956  if (failed(processType(loc, arrayType.getElementType(), elementTypeID))) {
957  return failure();
958  }
959  operands.push_back(elementTypeID);
960  if (auto elementCountID = prepareConstantInt(
961  loc, mlirBuilder.getI32IntegerAttr(arrayType.getNumElements()))) {
962  operands.push_back(elementCountID);
963  }
964  return processTypeDecoration(loc, arrayType, resultID);
965  }
966 
967  if (auto ptrType = type.dyn_cast<spirv::PointerType>()) {
968  uint32_t pointeeTypeID = 0;
969  if (failed(processType(loc, ptrType.getPointeeType(), pointeeTypeID))) {
970  return failure();
971  }
972  typeEnum = spirv::Opcode::OpTypePointer;
973  operands.push_back(static_cast<uint32_t>(ptrType.getStorageClass()));
974  operands.push_back(pointeeTypeID);
975  return success();
976  }
977 
978  if (auto runtimeArrayType = type.dyn_cast<spirv::RuntimeArrayType>()) {
979  uint32_t elementTypeID = 0;
980  if (failed(processType(loc, runtimeArrayType.getElementType(),
981  elementTypeID))) {
982  return failure();
983  }
984  operands.push_back(elementTypeID);
985  typeEnum = spirv::Opcode::OpTypeRuntimeArray;
986  return success();
987  }
988 
989  if (auto structType = type.dyn_cast<spirv::StructType>()) {
990  bool hasLayout = structType.hasLayout();
991  for (auto elementIndex :
992  llvm::seq<uint32_t>(0, structType.getNumElements())) {
993  uint32_t elementTypeID = 0;
994  if (failed(processType(loc, structType.getElementType(elementIndex),
995  elementTypeID))) {
996  return failure();
997  }
998  operands.push_back(elementTypeID);
999  if (hasLayout) {
1000  // Decorate each struct member with an offset
1001  if (failed(processMemberDecoration(
1002  resultID, elementIndex, spirv::Decoration::Offset,
1003  static_cast<uint32_t>(structType.getOffset(elementIndex))))) {
1004  return emitError(loc, "cannot decorate ")
1005  << elementIndex << "-th member of " << structType
1006  << " with its offset";
1007  }
1008  }
1009  }
1010  SmallVector<spirv::StructType::MemberDecorationInfo, 4> memberDecorations;
1011  structType.getMemberDecorations(memberDecorations);
1012  for (auto &memberDecoration : memberDecorations) {
1013  if (failed(processMemberDecoration(resultID, memberDecoration.first,
1014  memberDecoration.second))) {
1015  return emitError(loc, "cannot decorate ")
1016  << memberDecoration.first << "-th member of " << structType
1017  << " with " << stringifyDecoration(memberDecoration.second);
1018  }
1019  }
1020  typeEnum = spirv::Opcode::OpTypeStruct;
1021  return success();
1022  }
1023 
1024  // TODO(ravishankarm) : Handle other types.
1025  return emitError(loc, "unhandled type in serialization: ") << type;
1026 }

1205 uint32_t Serializer::prepareConstantInt(Location loc, IntegerAttr intAttr,
1206  bool isSpec) {
1207  if (!isSpec) {
1208  // We can de-duplicate normal constants, but not specialization constants.
1209  if (auto id = getConstantID(intAttr)) {
1210  return id;
1211  }
1212  }
1213 
1214  // Process the type for this integer literal
1215  uint32_t typeID = 0;
1216  if (failed(processType(loc, intAttr.getType(), typeID))) {
1217  return 0;
1218  }
1219 
1220  auto resultID = getNextID();
1221  APInt value = intAttr.getValue();
1222  unsigned bitwidth = value.getBitWidth();
1223  bool isSigned = value.isSignedIntN(bitwidth);
1224 
1225  auto opcode =
1226  isSpec ? spirv::Opcode::OpSpecConstant : spirv::Opcode::OpConstant;
1227 
1228  // According to SPIR-V spec, "When the type's bit width is less than 32-bits,
1229  // the literal's value appears in the low-order bits of the word, and the
1230  // high-order bits must be 0 for a floating-point type, or 0 for an integer
1231  // type with Signedness of 0, or sign extended when Signedness is 1."
1232  if (bitwidth == 32 || bitwidth == 16) {
1233  uint32_t word = 0;
1234  if (isSigned) {
1235  word = static_cast<int32_t>(value.getSExtValue());
1236  } else {
1237  word = static_cast<uint32_t>(value.getZExtValue());
1238  }
1239  encodeInstructionInto(typesGlobalValues, opcode, {typeID, resultID, word});
1240  }
1241  // According to SPIR-V spec: "When the type's bit width is larger than one
1242  // word, the literals low-order words appear first."
1243  else if (bitwidth == 64) {
1244  struct DoubleWord {
1245  uint32_t word1;
1246  uint32_t word2;
1247  } words;
1248  if (isSigned) {
1249  words = llvm::bit_cast<DoubleWord>(value.getSExtValue());
1250  } else {
1251  words = llvm::bit_cast<DoubleWord>(value.getZExtValue());
1252  }
1253  encodeInstructionInto(typesGlobalValues, opcode,
1254  {typeID, resultID, words.word1, words.word2});
1255  } else {
1256  std::string valueStr;
1257  llvm::raw_string_ostream rss(valueStr);
1258  value.print(rss, /*isSigned=*/false);
1259 
1260  emitError(loc, "cannot serialize ")
1261  << bitwidth << "-bit integer literal: " << rss.str();
1262  return 0;
1263  }
1264 
1265  if (!isSpec) {
1266  constIDMap[intAttr] = resultID;
1267  }
1268  return resultID;
1269 }

1333 LogicalResult
1334 Serializer::processBlock(Block *block, bool omitLabel,
1335  function_ref<void()> actionBeforeTerminator) {
1336  LLVM_DEBUG(llvm::dbgs() << "processing block " << block << ":\n");
1337  LLVM_DEBUG(block->print(llvm::dbgs()));
1338  LLVM_DEBUG(llvm::dbgs() << '\n');
1339  if (!omitLabel) {
1340  uint32_t blockID = getOrCreateBlockID(block);
1341  LLVM_DEBUG(llvm::dbgs()
1342  << "[block] " << block << " (id = " << blockID << ")\n");
1343 
1344  // Emit OpLabel for this block.
1345  encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {blockID});
1346  }
1347 
1348  // Emit OpPhi instructions for block arguments, if any.
1349  if (failed(emitPhiForBlockArguments(block)))
1350  return failure();
1351 
1352  // Process each op in this block except the terminator.
1353  for (auto &op : llvm::make_range(block->begin(), std::prev(block->end()))) {
1354  if (failed(processOperation(&op)))
1355  return failure();
1356  }
1357 
1358  // Process the terminator.
1359  if (actionBeforeTerminator)
1360  actionBeforeTerminator();
1361  if (failed(processOperation(&block->back())))
1362  return failure();
1363 
1364  return success();
1365 }

1367 LogicalResult Serializer::emitPhiForBlockArguments(Block *block) {
1368  // Nothing to do if this block has no arguments or it's the entry block, which
1369  // always has the same arguments as the function signature.
1370  if (block->args_empty() || block->isEntryBlock())
1371  return success();
1372 
1373  // If the block has arguments, we need to create SPIR-V OpPhi instructions.
1374  // A SPIR-V OpPhi instruction is of the syntax:
1375  // OpPhi | result type | result <id> | (value <id>, parent block <id>) pair
1376  // So we need to collect all predecessor blocks and the arguments they send
1377  // to this block.
1378  SmallVector<std::pair<Block *, Operation::operand_iterator>, 4> predecessors;
1379  for (Block *predecessor : block->getPredecessors()) {
1380  auto *terminator = predecessor->getTerminator();
1381  // Check whether this predecessor block contains a structured control flow
1382  // op. If so, the structured control flow op will be serialized to multiple
1383  // SPIR-V blocks. The branch op jumping to the OpPhi's block then resides in
1384  // the last structured control flow op's merge block.
1385  if (auto *merge = getLastStructuredControlFlowOpMergeBlock(predecessor))
1386  predecessor = merge;
1387  if (auto branchOp = dyn_cast<spirv::BranchOp>(terminator)) {
1388  predecessors.emplace_back(predecessor, branchOp.operand_begin());
1389  } else {
1390  return terminator->emitError("unimplemented terminator for Phi creation");
1391  }
1392  }
1393 
1394  // Then create OpPhi instruction for each of the block argument.
1395  for (auto argIndex : llvm::seq<unsigned>(0, block->getNumArguments())) {
1396  BlockArgument arg = block->getArgument(argIndex);
1397 
1398  // Get the type <id> and result <id> for this OpPhi instruction.
1399  uint32_t phiTypeID = 0;
1400  if (failed(processType(arg.getLoc(), arg.getType(), phiTypeID)))
1401  return failure();
1402  uint32_t phiID = getNextID();
1403 
1404  LLVM_DEBUG(llvm::dbgs() << "[phi] for block argument #" << argIndex << ' '
1405  << arg << " (id = " << phiID << ")\n");
1406 
1407  SmallVector<uint32_t, 8> phiArgs;
1408  phiArgs.push_back(phiTypeID);
1409  phiArgs.push_back(phiID);
1410 
1411  for (auto predIndex : llvm::seq<unsigned>(0, predecessors.size())) {
1412  Value value = *(predecessors[predIndex].second + argIndex);
1413  uint32_t predBlockId = getOrCreateBlockID(predecessors[predIndex].first);
1414  LLVM_DEBUG(llvm::dbgs() << "[phi] use predecessor (id = " << predBlockId
1415  << ") value " << value << ' ');
1416  // Each pair is a value <id> ...
1417  uint32_t valueId = getValueID(value);
1418  if (valueId == 0) {
1419  // The op generating this value hasn't been visited yet so we don't have
1420  // an <id> assigned yet. Record this to fix up later.
1421  LLVM_DEBUG(llvm::dbgs() << "(need to fix)\n");
1422  deferredPhiValues[value].push_back(functionBody.size() + 1 +
1423  phiArgs.size());
1424  } else {
1425  LLVM_DEBUG(llvm::dbgs() << "(id = " << valueId << ")\n");
1426  }
1427  phiArgs.push_back(valueId);
1428  // ... and a parent block <id>.
1429  phiArgs.push_back(predBlockId);
1430  }
1431 
1432  encodeInstructionInto(functionBody, spirv::Opcode::OpPhi, phiArgs);
1433  valueIDMap[arg] = phiID;
1434  }
1435 
1436  return success();
1437 }

1439 LogicalResult Serializer::processSelectionOp(spirv::SelectionOp selectionOp) {
1440  // Assign <id>s to all blocks so that branches inside the SelectionOp can
1441  // resolve properly.
1442  auto &body = selectionOp.body();
1443  for (Block &block : body)
1444  getOrCreateBlockID(&block);
1445 
1446  auto *headerBlock = selectionOp.getHeaderBlock();
1447  auto *mergeBlock = selectionOp.getMergeBlock();
1448  auto mergeID = getBlockID(mergeBlock);
1449 
1450  // Emit the selection header block, which dominates all other blocks, first.
1451  // We need to emit an OpSelectionMerge instruction before the selection header
1452  // block's terminator.
1453  auto emitSelectionMerge = [&]() {
1454  // TODO(antiagainst): properly support selection control here
1455  encodeInstructionInto(
1456  functionBody, spirv::Opcode::OpSelectionMerge,
1457  {mergeID, static_cast<uint32_t>(spirv::SelectionControl::None)});
1458  };
1459  // For structured selection, we cannot have blocks in the selection construct
1460  // branching to the selection header block. Entering the selection (and
1461  // reaching the selection header) must be from the block containing the
1462  // spv.selection op. If there are ops ahead of the spv.selection op in the
1463  // block, we can "merge" them into the selection header. So here we don't need
1464  // to emit a separate block; just continue with the existing block.
1465  if (failed(processBlock(headerBlock, /*omitLabel=*/true, emitSelectionMerge)))
1466  return failure();
1467 
1468  // Process all blocks with a depth-first visitor starting from the header
1469  // block. The selection header block and merge block are skipped by this
1470  // visitor.
1471  if (failed(visitInPrettyBlockOrder(
1472  headerBlock, [&](Block *block) { return processBlock(block); },
1473  /*skipHeader=*/true, /*skipBlocks=*/{mergeBlock})))
1474  return failure();
1475 
1476  // There is nothing to do for the merge block in the selection, which just
1477  // contains a spv._merge op, itself. But we need to have an OpLabel
1478  // instruction to start a new SPIR-V block for ops following this SelectionOp.
1479  // The block should use the <id> for the merge block.
1480  return encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {mergeID});
1481 }

1483 LogicalResult Serializer::processLoopOp(spirv::LoopOp loopOp) {
1484  // Assign <id>s to all blocks so that branches inside the LoopOp can resolve
1485  // properly. We don't need to assign for the entry block, which is just for
1486  // satisfying MLIR region's structural requirement.
1487  auto &body = loopOp.body();
1488  for (Block &block :
1489  llvm::make_range(std::next(body.begin(), 1), body.end())) {
1490  getOrCreateBlockID(&block);
1491  }
1492  auto *headerBlock = loopOp.getHeaderBlock();
1493  auto *continueBlock = loopOp.getContinueBlock();
1494  auto *mergeBlock = loopOp.getMergeBlock();
1495  auto headerID = getBlockID(headerBlock);
1496  auto continueID = getBlockID(continueBlock);
1497  auto mergeID = getBlockID(mergeBlock);
1498 
1499  // This LoopOp is in some MLIR block with preceding and following ops. In the
1500  // binary format, it should reside in separate SPIR-V blocks from its
1501  // preceding and following ops. So we need to emit unconditional branches to
1502  // jump to this LoopOp's SPIR-V blocks and jumping back to the normal flow
1503  // afterwards.
1504  encodeInstructionInto(functionBody, spirv::Opcode::OpBranch, {headerID});
1505 
1506  // We omit the LoopOp's entry block and start serialization from the loop
1507  // header block. The entry block should not contain any additional ops other
1508  // than a single spv.Branch that jumps to the loop header block. However,
1509  // the spv.Branch can contain additional block arguments. Those block
1510  // arguments must come from out of the loop using implicit capture. We will
1511  // need to query the <id> for the value sent and the <id> for the incoming
1512  // parent block. For the latter, we need to make sure this block is
1513  // registered. The value sent should come from the block this loop resides in.
1514  blockIDMap[loopOp.getEntryBlock()] =
1515  getBlockID(loopOp.getOperation()->getBlock());
1516 
1517  // Emit the loop header block, which dominates all other blocks, first. We
1518  // need to emit an OpLoopMerge instruction before the loop header block's
1519  // terminator.
1520  auto emitLoopMerge = [&]() {
1521  // TODO(antiagainst): properly support loop control here
1522  encodeInstructionInto(
1523  functionBody, spirv::Opcode::OpLoopMerge,
1524  {mergeID, continueID, static_cast<uint32_t>(spirv::LoopControl::None)});
1525  };
1526  if (failed(processBlock(headerBlock, /*omitLabel=*/false, emitLoopMerge)))
1527  return failure();
1528 
1529  // Process all blocks with a depth-first visitor starting from the header
1530  // block. The loop header block, loop continue block, and loop merge block are
1531  // skipped by this visitor and handled later in this function.
1532  if (failed(visitInPrettyBlockOrder(
1533  headerBlock, [&](Block *block) { return processBlock(block); },
1534  /*skipHeader=*/true, /*skipBlocks=*/{continueBlock, mergeBlock})))
1535  return failure();
1536 
1537  // We have handled all other blocks. Now get to the loop continue block.
1538  if (failed(processBlock(continueBlock)))
1539  return failure();
1540 
1541  // There is nothing to do for the merge block in the loop, which just contains
1542  // a spv._merge op, itself. But we need to have an OpLabel instruction to
1543  // start a new SPIR-V block for ops following this LoopOp. The block should
1544  // use the <id> for the merge block.
1545  return encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {mergeID});
1546 }

1629 LogicalResult Serializer::processOperation(Operation *opInst) {
1630  LLVM_DEBUG(llvm::dbgs() << "[op] '" << opInst->getName() << "'\n");
1631 
1632  // First dispatch the ops that do not directly mirror an instruction from
1633  // the SPIR-V spec.
1634  return TypeSwitch<Operation *, LogicalResult>(opInst)
1635  .Case([&](spirv::AddressOfOp op) { return processAddressOfOp(op); })
1636  .Case([&](spirv::BranchOp op) { return processBranchOp(op); })
1637  .Case([&](spirv::BranchConditionalOp op) {

1639  })

1641  .Case([&](FuncOp op) { return processFuncOp(op); })
1642  .Case([&](spirv::GlobalVariableOp op) {
1643  return processGlobalVariableOp(op);
1644  })
1645  .Case([&](spirv::LoopOp op) { return processLoopOp(op); })
1646  .Case([&](spirv::ModuleEndOp) { return success(); })
1647  .Case([&](spirv::ReferenceOfOp op) { return processReferenceOfOp(op); })
1648  .Case([&](spirv::SelectionOp op) { return processSelectionOp(op); })
1649  .Case([&](spirv::SpecConstantOp op) { return processSpecConstantOp(op); })
1650  .Case([&](spirv::UndefOp op) { return processUndefOp(op); })
1651  .Case([&](spirv::VariableOp op) { return processVariableOp(op); })
1652

1654  // auto-generated methods.
1655  .Default(
1656  [&](Operation *op) { return dispatchToAutogenSerialization(op); });
