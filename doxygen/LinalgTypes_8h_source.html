36 /// A RangeType represents a minimal range abstraction (min, max, step).
37 /// It is constructed by calling the linalg.range op with three values index of
38 /// index type:
39 ///
40 /// ```mlir
41 /// func @foo(%arg0 : index, %arg1 : index, %arg2 : index) {
42 /// %0 = linalg.range %arg0:%arg1:%arg2 : !linalg.range
43 /// }
44 /// ```
45 class RangeType : public Type::TypeBase<RangeType, Type> {
46 public:
47  // Used for generic hooks in TypeBase.
48  using Base::Base;
49  /// Construction hook.
50  static RangeType get(MLIRContext *context) {
51  /// Custom, uniq'ed construction in the MLIRContext.
52  return Base::get(context, LinalgTypes::Range);
53  }
54  /// Used to implement llvm-style cast.
55  static bool kindof(unsigned kind) { return kind == LinalgTypes::Range; }
56 };
