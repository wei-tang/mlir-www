16 // Values that can be used to signal success/failure. This should be used in
17 // conjunction with the utility functions below.
18 struct LogicalResult {
19  enum ResultEnum { Success, Failure } value;
20  LogicalResult(ResultEnum v) : value(v) {}
21 };

23 /// Utility function to generate a LogicalResult. If isSuccess is true a
24 /// `success` result is generated, otherwise a 'failure' result is generated.
25 inline LogicalResult success(bool isSuccess = true) {
26  return LogicalResult{isSuccess ? LogicalResult::Success
27  : LogicalResult::Failure};
28 }
29 
30 /// Utility function to generate a LogicalResult. If isFailure is true a
31 /// `failure` result is generated, otherwise a 'success' result is generated.
32 inline LogicalResult failure(bool isFailure = true) {
33  return LogicalResult{isFailure ? LogicalResult::Failure
34  : LogicalResult::Success};
35 }

37 /// Utility function that returns true if the provided LogicalResult corresponds

43 /// Utility function that returns true if the provided LogicalResult corresponds
