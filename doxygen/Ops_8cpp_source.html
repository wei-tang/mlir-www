258 static ParseResult parseAllocOp(OpAsmParser &parser, OperationState &result) {
259  MemRefType type;
260 
261  // Parse the dimension operands and optional symbol operands, followed by a
262  // memref type.
263  unsigned numDimOperands;
264  if (parseDimAndSymbolList(parser, result.operands, numDimOperands) ||
265  parser.parseOptionalAttrDict(result.attributes) ||
266  parser.parseColonType(type))
267  return failure();
268 
269  // Check numDynamicDims against number of question marks in memref type.
270  // Note: this check remains here (instead of in verify()), because the
271  // partition between dim operands and symbol operands is lost after parsing.
272  // Verification still checks that the total number of operands matches
273  // the number of symbols in the affine map, plus the number of dynamic
274  // dimensions in the memref.
275  if (numDimOperands != type.getNumDynamicDims())
276  return parser.emitError(parser.getNameLoc())
277  << "dimension operand count does not equal memref dynamic dimension "
278  "count";
279  result.types.push_back(type);
280  return success();
281 }

283 static LogicalResult verify(AllocOp op) {
284  auto memRefType = op.getResult().getType().dyn_cast<MemRefType>();
285  if (!memRefType)
286  return op.emitOpError("result must be a memref");
287 
288  unsigned numSymbols = 0;
289  if (!memRefType.getAffineMaps().empty()) {
290  // Store number of symbols used in affine map (used in subsequent check).
291  AffineMap affineMap = memRefType.getAffineMaps()[0];
292  numSymbols = affineMap.getNumSymbols();
293  }
294 
295  // Check that the total number of operands matches the number of symbols in
296  // the affine map, plus the number of dynamic dimensions specified in the
297  // memref type.
298  unsigned numDynamicDims = memRefType.getNumDynamicDims();
299  if (op.getNumOperands() != numDynamicDims + numSymbols)
300  return op.emitOpError(
301  "operand count does not equal dimension plus symbol operand count");
302 
303  // Verify that all operands are of type Index.
304  for (auto operandType : op.getOperandTypes())
305  if (!operandType.isIndex())
306  return op.emitOpError("requires operands to be of type Index");
307  return success();
308 }

311 /// Fold constant dimensions into an alloc operation.
312 struct SimplifyAllocConst : public OpRewritePattern<AllocOp> {
313  using OpRewritePattern<AllocOp>::OpRewritePattern;
314 
315  PatternMatchResult matchAndRewrite(AllocOp alloc,
316  PatternRewriter &rewriter) const override {
317  // Check to see if any dimensions operands are constants. If so, we can
318  // substitute and drop them.
319  if (llvm::none_of(alloc.getOperands(), [](Value operand) {
320  return matchPattern(operand, m_ConstantIndex());
321  }))
322  return matchFailure();
323 
324  auto memrefType = alloc.getType();
325 
326  // Ok, we have one or more constant operands. Collect the non-constant ones
327  // and keep track of the resultant memref type to build.
328  SmallVector<int64_t, 4> newShapeConstants;
329  newShapeConstants.reserve(memrefType.getRank());
330  SmallVector<Value, 4> newOperands;
331  SmallVector<Value, 4> droppedOperands;
332 
333  unsigned dynamicDimPos = 0;
334  for (unsigned dim = 0, e = memrefType.getRank(); dim < e; ++dim) {
335  int64_t dimSize = memrefType.getDimSize(dim);
336  // If this is already static dimension, keep it.
337  if (dimSize != -1) {
338  newShapeConstants.push_back(dimSize);
339  continue;
340  }
341  auto *defOp = alloc.getOperand(dynamicDimPos).getDefiningOp();
342  if (auto constantIndexOp = dyn_cast_or_null<ConstantIndexOp>(defOp)) {
343  // Dynamic shape dimension will be folded.
344  newShapeConstants.push_back(constantIndexOp.getValue());
345  // Record to check for zero uses later below.
346  droppedOperands.push_back(constantIndexOp);
347  } else {
348  // Dynamic shape dimension not folded; copy operand from old memref.
349  newShapeConstants.push_back(-1);
350  newOperands.push_back(alloc.getOperand(dynamicDimPos));
351  }
352  dynamicDimPos++;
353  }
354 
355  // Create new memref type (which will have fewer dynamic dimensions).
356  auto newMemRefType = MemRefType::get(
357  newShapeConstants, memrefType.getElementType(),
358  memrefType.getAffineMaps(), memrefType.getMemorySpace());
359  assert(static_cast<int64_t>(newOperands.size()) ==
360  newMemRefType.getNumDynamicDims());
361 
362  // Create and insert the alloc op for the new memref.
363  auto newAlloc = rewriter.create<AllocOp>(alloc.getLoc(), newMemRefType,
364  newOperands, IntegerAttr());
365  // Insert a cast so we have the same type as the old alloc.
366  auto resultCast = rewriter.create<MemRefCastOp>(alloc.getLoc(), newAlloc,
367  alloc.getType());
368 
369  rewriter.replaceOp(alloc, {resultCast}, droppedOperands);
370  return matchSuccess();
371  }
372 };

400 /// Simplify a branch to a block that has a single predecessor. This effectively
401 /// merges the two blocks.
402 struct SimplifyBrToBlockWithSinglePred : public OpRewritePattern<BranchOp> {
403  using OpRewritePattern<BranchOp>::OpRewritePattern;

405  PatternMatchResult matchAndRewrite(BranchOp op,
406  PatternRewriter &rewriter) const override {
407  // Check that the successor block has a single predecessor.
408  Block *succ = op.getDest();

414  rewriter.mergeBlocks(succ, opParent, op.getOperands());
415  rewriter.eraseOp(op);
416  return matchSuccess();
417  }

843 // Compute `lhs` `pred` `rhs`, where `pred` is one of the known floating point
844 // comparison predicates.
845 static bool applyCmpPredicate(CmpFPredicate predicate, const APFloat &lhs,
846  const APFloat &rhs) {
847  auto cmpResult = lhs.compare(rhs);
848  switch (predicate) {
849  case CmpFPredicate::AlwaysFalse:
850  return false;
851  case CmpFPredicate::OEQ:
852  return cmpResult == APFloat::cmpEqual;
853  case CmpFPredicate::OGT:
854  return cmpResult == APFloat::cmpGreaterThan;
855  case CmpFPredicate::OGE:
856  return cmpResult == APFloat::cmpGreaterThan ||
857  cmpResult == APFloat::cmpEqual;
858  case CmpFPredicate::OLT:
859  return cmpResult == APFloat::cmpLessThan;
860  case CmpFPredicate::OLE:
861  return cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
862  case CmpFPredicate::ONE:
863  return cmpResult != APFloat::cmpUnordered && cmpResult != APFloat::cmpEqual;
864  case CmpFPredicate::ORD:
865  return cmpResult != APFloat::cmpUnordered;
866  case CmpFPredicate::UEQ:
867  return cmpResult == APFloat::cmpUnordered || cmpResult == APFloat::cmpEqual;
868  case CmpFPredicate::UGT:
869  return cmpResult == APFloat::cmpUnordered ||
870  cmpResult == APFloat::cmpGreaterThan;
871  case CmpFPredicate::UGE:
872  return cmpResult == APFloat::cmpUnordered ||
873  cmpResult == APFloat::cmpGreaterThan ||
874  cmpResult == APFloat::cmpEqual;
875  case CmpFPredicate::ULT:
876  return cmpResult == APFloat::cmpUnordered ||
877  cmpResult == APFloat::cmpLessThan;
878  case CmpFPredicate::ULE:
879  return cmpResult == APFloat::cmpUnordered ||
880  cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
881  case CmpFPredicate::UNE:
882  return cmpResult != APFloat::cmpEqual;
883  case CmpFPredicate::UNO:
884  return cmpResult == APFloat::cmpUnordered;
885  case CmpFPredicate::AlwaysTrue:
886  return true;
887  default:
888  llvm_unreachable("unknown comparison predicate");
889  }
890 }

1013 /// The constant op requires an attribute, and furthermore requires that it
1014 /// matches the return type.
1015 static LogicalResult verify(ConstantOp &op) {
1016  auto value = op.getValue();
1017  if (!value)
1018  return op.emitOpError("requires a 'value' attribute");
1019 
1020  auto type = op.getType();
1021  if (!value.getType().isa<NoneType>() && type != value.getType())
1022  return op.emitOpError() << "requires attribute's type (" << value.getType()
1023  << ") to match op's return type (" << type << ")";

1025  if (type.isa<IndexType>() || value.isa<BoolAttr>())
1026  return success();

1032  auto intVal = intAttr.getValue();

1034  return op.emitOpError("requires 'value' to be an integer within the "
1035  "range of the integer result type");
1036  return success();
1037  }
1038

1040  if (!value.isa<FloatAttr>())
1041  return op.emitOpError("requires 'value' to be a floating point constant");
1042  return success();

1046  if (!value.isa<ElementsAttr>())
1047  return op.emitOpError("requires 'value' to be a shaped constant");

1064  return op.emitOpError("reference to function with mismatched type");

1067  }

1070  return success();

1682  auto ubT = b.dyn_cast<UnrankedMemRefType>();

1685  if (aT.getElementType() != bT.getElementType())
1686  return false;
1687  if (aT.getAffineMaps() != bT.getAffineMaps()) {
1688  int64_t aOffset, bOffset;
1689  SmallVector<int64_t, 4> aStrides, bStrides;
1690  if (failed(getStridesAndOffset(aT, aStrides, aOffset)) ||
1691  failed(getStridesAndOffset(bT, bStrides, bOffset)) ||
1692  aStrides.size() != bStrides.size())
1693  return false;
1694 
1695  // Strides along a dimension/offset are compatible if the value in the
1696  // source memref is static and the value in the target memref is the
1697  // same. They are also compatible if either one is dynamic (see
1698  // description of MemRefCastOp for details).
1699  auto checkCompatible = [](int64_t a, int64_t b) {
1700  return (a == MemRefType::getDynamicStrideOrOffset() ||
1701  b == MemRefType::getDynamicStrideOrOffset() || a == b);
1702  };
1703  if (!checkCompatible(aOffset, bOffset))
1704  return false;
1705  for (auto aStride : enumerate(aStrides))
1706  if (!checkCompatible(aStride.value(), bStrides[aStride.index()]))
1707  return false;
1708  }
1709  if (aT.getMemorySpace() != bT.getMemorySpace())
1710  return false;
1711 
1712  // They must have the same rank, and any specified dimensions must match.
1713  if (aT.getRank() != bT.getRank())
1714  return false;
1715 
1716  for (unsigned i = 0, e = aT.getRank(); i != e; ++i) {
1717  int64_t aDim = aT.getDimSize(i), bDim = bT.getDimSize(i);
1718  if (aDim != -1 && bDim != -1 && aDim != bDim)
1719  return false;
1720  }
1721  return true;
1722  } else {
1723  if (!aT && !uaT)
1724  return false;
1725  if (!bT && !ubT)
1726  return false;
1727  // Unranked to unranked casting is unsupported
1728  if (uaT && ubT)
1729  return false;
1730

1732  auto bEltType = (bT) ? bT.getElementType() : ubT.getElementType();
1733  if (aEltType != bEltType)
1734  return false;

1737  auto bMemSpace = (bT) ? bT.getMemorySpace() : ubT.getMemorySpace();
1738  if (aMemSpace != bMemSpace)
1739  return false;
1740 
1741  return true;
1742  }
1743

2431  }))

2444  int64_t oldOffset;
2445  SmallVector<int64_t, 4> oldStrides;
2446  if (failed(getStridesAndOffset(memrefType, oldStrides, oldOffset)))
2447  return matchFailure();

2450  SmallVector<Value, 4> droppedOperands;
2451

2453  auto dynamicOffset = viewOp.getDynamicOffset();
2454  int64_t newOffset = oldOffset;
2455  unsigned dynamicOffsetOperandCount = 0;

2466  }

2470  SmallVector<int64_t, 4> newShapeConstants;
2471  newShapeConstants.reserve(memrefType.getRank());
2472

2474  unsigned rank = memrefType.getRank();
2475  for (unsigned dim = 0, e = rank; dim < e; ++dim) {
2476  int64_t dimSize = memrefType.getDimSize(dim);
2477  // If this is already static dimension, keep it.
2478  if (!ShapedType::isDynamic(dimSize)) {

2483  if (auto constantIndexOp = dyn_cast_or_null<ConstantIndexOp>(defOp)) {

2485  newShapeConstants.push_back(constantIndexOp.getValue());

2487  droppedOperands.push_back(constantIndexOp);
2488  } else {

2490  newShapeConstants.push_back(dimSize);
2491  newOperands.push_back(viewOp.getOperand(dynamicDimPos));
2492  }

2502  dynamicStrides = true;

2504  newStrides[i] = MemRefType::getDynamicStrideOrOffset();
2505  else
2506  newStrides[i] = newShapeConstants[i + 1] * newStrides[i + 1];
2507  }

2510  map = makeStridedLinearLayoutMap(newStrides, newOffset,
2511  rewriter.getContext());

2514  auto newMemRefType =
2515  MemRefType::get(newShapeConstants, memrefType.getElementType(), {map},
2516  memrefType.getMemorySpace());

2518  assert(static_cast<int64_t>(newOperands.size()) ==
2519  dynamicOffsetOperandCount + newMemRefType.getNumDynamicDims());
2520 
2521  // Create new ViewOp.
2522  auto newViewOp = rewriter.create<ViewOp>(viewOp.getLoc(), newMemRefType,
2523  viewOp.getOperand(0), newOperands);
2524  // Insert a cast so we have the same type as the old memref type.
2525  rewriter.replaceOpWithNewOp<MemRefCastOp>(droppedOperands, viewOp,
2526  newViewOp, viewOp.getType());

2650 static LogicalResult verify(SubViewOp op) {
2651  auto baseType = op.getBaseMemRefType().cast<MemRefType>();
2652  auto subViewType = op.getType();
2653 
2654  // The rank of the base and result subview must match.
2655  if (baseType.getRank() != subViewType.getRank()) {
2656  return op.emitError(
2657  "expected rank of result type to match rank of base type ");
2658  }
2659 
2660  // The base memref and the view memref should be in the same memory space.
2661  if (baseType.getMemorySpace() != subViewType.getMemorySpace())
2662  return op.emitError("different memory spaces specified for base memref "
2663  "type ")
2664  << baseType << " and subview memref type " << subViewType;
2665 
2666  // Verify that the base memref type has a strided layout map.
2667  int64_t baseOffset;
2668  SmallVector<int64_t, 4> baseStrides;
2669  if (failed(getStridesAndOffset(baseType, baseStrides, baseOffset)))
2670  return op.emitError("base type ") << subViewType << " is not strided";
2671 
2672  // Verify that the result memref type has a strided layout map.
2673  int64_t subViewOffset;
2674  SmallVector<int64_t, 4> subViewStrides;
2675  if (failed(getStridesAndOffset(subViewType, subViewStrides, subViewOffset)))
2676  return op.emitError("result type ") << subViewType << " is not strided";
2677 
2678  // Num offsets should either be zero or rank of memref.
2679  if (op.getNumOffsets() != 0 && op.getNumOffsets() != subViewType.getRank()) {
2680  return op.emitError("expected number of dynamic offsets specified to match "
2681  "the rank of the result type ")
2682  << subViewType;
2683  }
2684 
2685  // Num sizes should either be zero or rank of memref.
2686  if (op.getNumSizes() != 0 && op.getNumSizes() != subViewType.getRank()) {
2687  return op.emitError("expected number of dynamic sizes specified to match "
2688  "the rank of the result type ")
2689  << subViewType;
2690  }
2691 
2692  // Num strides should either be zero or rank of memref.
2693  if (op.getNumStrides() != 0 && op.getNumStrides() != subViewType.getRank()) {
2694  return op.emitError("expected number of dynamic strides specified to match "
2695  "the rank of the result type ")
2696  << subViewType;
2697  }
2698 
2699  // Verify that if the shape of the subview type is static, then sizes are not
2700  // dynamic values, and vice versa.
2701  if ((subViewType.hasStaticShape() && op.getNumSizes() != 0) ||
2702  (op.getNumSizes() == 0 && !subViewType.hasStaticShape())) {
2703  return op.emitError("invalid to specify dynamic sizes when subview result "
2704  "type is statically shaped and viceversa");
2705  }
2706 
2707  // Verify that if dynamic sizes are specified, then the result memref type
2708  // have full dynamic dimensions.
2709  if (op.getNumSizes() > 0) {
2710  if (llvm::any_of(subViewType.getShape(), [](int64_t dim) {
2711  return dim != ShapedType::kDynamicSize;
2712  })) {
2713  // TODO: This is based on the assumption that number of size arguments are
2714  // either 0, or the rank of the result type. It is possible to have more
2715  // fine-grained verification where only particular dimensions are
2716  // dynamic. That probably needs further changes to the shape op
2717  // specification.
2718  return op.emitError("expected shape of result type to be fully dynamic "
2719  "when sizes are specified");
2720  }
2721  }
2722 
2723  // Verify that if dynamic offsets are specified or base memref has dynamic
2724  // offset or base memref has dynamic strides, then the subview offset is
2725  // dynamic.
2726  if ((op.getNumOffsets() > 0 ||
2727  baseOffset == MemRefType::getDynamicStrideOrOffset() ||
2728  llvm::is_contained(baseStrides,
2729  MemRefType::getDynamicStrideOrOffset())) &&
2730  subViewOffset != MemRefType::getDynamicStrideOrOffset()) {
2731  return op.emitError(
2732  "expected result memref layout map to have dynamic offset");
2733  }
2734 
2735  // For now, verify that if dynamic strides are specified, then all the result
2736  // memref type have dynamic strides.
2737  if (op.getNumStrides() > 0) {
2738  if (llvm::any_of(subViewStrides, [](int64_t stride) {
2739  return stride != MemRefType::getDynamicStrideOrOffset();
2740  })) {
2741  return op.emitError("expected result type to have dynamic strides");
2742  }
2743  }
2744 
2745  // If any of the base memref has dynamic stride, then the corresponding
2746  // stride of the subview must also have dynamic stride.
2747  assert(baseStrides.size() == subViewStrides.size());
2748  for (auto stride : enumerate(baseStrides)) {
2749  if (stride.value() == MemRefType::getDynamicStrideOrOffset() &&
2750  subViewStrides[stride.index()] !=
2751  MemRefType::getDynamicStrideOrOffset()) {
2752  return op.emitError(
2753  "expected result type to have dynamic stride along a dimension if "
2754  "the base memref type has dynamic stride along that dimension");
2755  }
2756  }
2757  return success();
2758 }

2776 SubViewOp::getStaticStrides(SmallVectorImpl<int64_t> &staticStrides) {
2777  // If the strides are dynamic return failure.
2778  if (getNumStrides())
2779  return failure();
2780 
2781  // When static, the stride operands can be retrieved by taking the strides of
2782  // the result of the subview op, and dividing the strides of the base memref.
2783  int64_t resultOffset, baseOffset;

2785  if (failed(

2788  failed(getStridesAndOffset(getType(), resultStrides, resultOffset)))
2789  return failure();
2790

2792  baseStrides.size() == resultStrides.size() &&
2793  "base and result memrefs must have the same rank");
2794  assert(!llvm::is_contained(resultStrides,
2795  MemRefType::getDynamicStrideOrOffset()) &&
2796  "strides of subview op must be static, when there are no dynamic "
2797  "strides specified");
2798  staticStrides.resize(getType().getRank());
2799  for (auto resultStride : enumerate(resultStrides)) {
2800  auto baseStride = baseStrides[resultStride.index()];
2801  // The result stride is expected to be a multiple of the base stride. Abort
2802  // if that is not the case.
2803  if (resultStride.value() < baseStride ||
2804  resultStride.value() % baseStride != 0)
2805  return failure();
2806  staticStrides[resultStride.index()] = resultStride.value() / baseStride;
2807  }
2808  return success();
