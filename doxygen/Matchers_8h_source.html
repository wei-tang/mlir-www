25 /// The matcher that matches a certain kind of Attribute and binds the value
26 /// inside the Attribute.
27 template <
28  typename AttrClass,
29  // Require AttrClass to be a derived class from Attribute and get its
30  // value type
31  typename ValueType =
32  typename std::enable_if<std::is_base_of<Attribute, AttrClass>::value,
33  AttrClass>::type::ValueType,
34  // Require the ValueType is not void
35  typename = typename std::enable_if<!std::is_void<ValueType>::value>::type>
36 struct attr_value_binder {
37  ValueType *bind_value;
38 
39  /// Creates a matcher instance that binds the value to bv if match succeeds.
40  attr_value_binder(ValueType *bv) : bind_value(bv) {}
41 
42  bool match(const Attribute &attr) {
43  if (auto intAttr = attr.dyn_cast<AttrClass>()) {
44  *bind_value = intAttr.getValue();
45  return true;
46  }
47  return false;
48  }
49 };

51 /// The matcher that matches a constant foldable operation that has no side
52 /// effect, no operands and produces a single result.
53 template <typename AttrT> struct constant_op_binder {
54  AttrT *bind_value;
55 
56  /// Creates a matcher instance that binds the constant attribute value to
57  /// bind_value if match succeeds.
58  constant_op_binder(AttrT *bind_value) : bind_value(bind_value) {}
59  /// Creates a matcher instance that doesn't bind if match succeeds.
60  constant_op_binder() : bind_value(nullptr) {}
61 
62  bool match(Operation *op) {
63  if (op->getNumOperands() > 0 || op->getNumResults() != 1)
64  return false;
65  if (!op->hasNoSideEffect())
66  return false;
67 
68  SmallVector<OpFoldResult, 1> foldedOp;
69  if (succeeded(op->fold(/*operands=*/llvm::None, foldedOp))) {
70  if (auto attr = foldedOp.front().dyn_cast<Attribute>()) {
71  if (auto attrT = attr.dyn_cast<AttrT>()) {
72  if (bind_value)
73  *bind_value = attrT;
74  return true;
75  }
76  }
77  }
78  return false;
79  }
80 };

82 /// The matcher that matches a constant scalar / vector splat / tensor splat
83 /// integer operation and binds the constant integer value.
84 struct constant_int_op_binder {
85  IntegerAttr::ValueType *bind_value;
86 
87  /// Creates a matcher instance that binds the value to bv if match succeeds.
88  constant_int_op_binder(IntegerAttr::ValueType *bv) : bind_value(bv) {}
89 
90  bool match(Operation *op) {
91  Attribute attr;
92  if (!constant_op_binder<Attribute>(&attr).match(op))
93  return false;
94  auto type = op->getResult(0).getType();
95 
96  if (type.isIntOrIndex()) {
97  return attr_value_binder<IntegerAttr>(bind_value).match(attr);
98  }
99  if (type.isa<VectorType>() || type.isa<RankedTensorType>()) {
100  if (auto splatAttr = attr.dyn_cast<SplatElementsAttr>()) {
101  return attr_value_binder<IntegerAttr>(bind_value)
102  .match(splatAttr.getSplatValue());
103  }
104  }
105  return false;
106  }
107 };

132 /// Trait to check whether T provides a 'match' method with type

170 template <typename TupleT, class CallbackT, std::size_t... Is>
171 constexpr void enumerateImpl(TupleT &&tuple, CallbackT &&callback,
172  std::index_sequence<Is...>) {
173  (void)std::initializer_list<int>{
174  0,
175  (callback(std::integral_constant<std::size_t, Is>{}, std::get<Is>(tuple)),
176  0)...};
177 }

179 template <typename... Tys, typename CallbackT>
180 constexpr void enumerate(std::tuple<Tys...> &tuple, CallbackT &&callback) {
181  detail::enumerateImpl(tuple, std::forward<CallbackT>(callback),
182  std::make_index_sequence<sizeof...(Tys)>{});

185 /// RecursivePatternMatcher that composes.
186 template <typename OpType, typename... OperandMatchers>
187 struct RecursivePatternMatcher {
188  RecursivePatternMatcher(OperandMatchers... matchers)
189  : operandMatchers(matchers...) {}
190  bool match(Operation *op) {
191  if (!isa<OpType>(op) || op->getNumOperands() != sizeof...(OperandMatchers))
192  return false;
193  bool res = true;
194  enumerate(operandMatchers, [&](size_t index, auto &matcher) {
195  res &= matchOperandOrValueAtIndex(op, index, matcher);
196  });
197  return res;
198  }
199  std::tuple<OperandMatchers...> operandMatchers;
200 };
