25 class SPIRVGlobalVariableOpLayoutInfoDecoration
26  : public OpRewritePattern<spirv::GlobalVariableOp> {
27 public:
28  using OpRewritePattern<spirv::GlobalVariableOp>::OpRewritePattern;
29 
30  PatternMatchResult matchAndRewrite(spirv::GlobalVariableOp op,
31  PatternRewriter &rewriter) const override {
32  spirv::StructType::LayoutInfo structSize = 0;
33  VulkanLayoutUtils::Size structAlignment = 1;
34  SmallVector<NamedAttribute, 4> globalVarAttrs;
35 
36  auto ptrType = op.type().cast<spirv::PointerType>();
37  auto structType = VulkanLayoutUtils::decorateType(
38  ptrType.getPointeeType().cast<spirv::StructType>(), structSize,
39  structAlignment);
40  auto decoratedType =
41  spirv::PointerType::get(structType, ptrType.getStorageClass());
42 
43  // Save all named attributes except "type" attribute.
44  for (const auto &attr : op.getAttrs()) {
45  if (attr.first == "type") {
46  continue;
47  }
48  globalVarAttrs.push_back(attr);
49  }
50 
51  rewriter.replaceOpWithNewOp<spirv::GlobalVariableOp>(
52  op, TypeAttr::get(decoratedType), globalVarAttrs);
53  return matchSuccess();
54  }
55 };

57 class SPIRVAddressOfOpLayoutInfoDecoration
58  : public OpRewritePattern<spirv::AddressOfOp> {
59 public:
60  using OpRewritePattern<spirv::AddressOfOp>::OpRewritePattern;
61 
62  PatternMatchResult matchAndRewrite(spirv::AddressOfOp op,
63  PatternRewriter &rewriter) const override {
64  auto spirvModule = op.getParentOfType<spirv::ModuleOp>();
65  auto varName = op.variable();
66  auto varOp = spirvModule.lookupSymbol<spirv::GlobalVariableOp>(varName);
67 
68  rewriter.replaceOpWithNewOp<spirv::AddressOfOp>(
69  op, varOp.type(), rewriter.getSymbolRefAttr(varName));
70  return matchSuccess();
71  }

89 void DecorateSPIRVCompositeTypeLayoutPass::runOnModule() {
90  auto module = getModule();
91  OwningRewritePatternList patterns;
92  populateSPIRVLayoutInfoPatterns(patterns, module.getContext());
93  ConversionTarget target(*(module.getContext()));
94  target.addLegalDialect<spirv::SPIRVDialect>();
95  target.addLegalOp<FuncOp>();
96  target.addDynamicallyLegalOp<spirv::GlobalVariableOp>(
97  [](spirv::GlobalVariableOp op) {
98  return VulkanLayoutUtils::isLegalType(op.type());
99  });
100 
101  // Change the type for the direct users.
102  target.addDynamicallyLegalOp<spirv::AddressOfOp>([](spirv::AddressOfOp op) {
103  return VulkanLayoutUtils::isLegalType(op.pointer().getType());
104  });
105 
106  // TODO: Change the type for the indirect users such as spv.Load, spv.Store,
107  // spv.FunctionCall and so on.
108 
109  for (auto spirvModule : module.getOps<spirv::ModuleOp>()) {
110  if (failed(applyFullConversion(spirvModule, target, patterns))) {
111  signalPassFailure();
112  }
113  }
114 }
