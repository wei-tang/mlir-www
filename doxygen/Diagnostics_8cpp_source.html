93 /// Convert a Twine to a StringRef. Memory used for generating the StringRef is
94 /// stored in 'strings'.
95 static StringRef twineToStrRef(const Twine &val,
96  std::vector<std::unique_ptr<char[]>> &strings) {
97  // Allocate memory to hold this string.
98  SmallString<64> data;
99  auto strRef = val.toStringRef(data);
100  strings.push_back(std::unique_ptr<char[]>(new char[strRef.size()]));
101  memcpy(&strings.back()[0], strRef.data(), strRef.size());
102 
103  // Return a reference to the new string.
104  return StringRef(&strings.back()[0], strRef.size());
105 }

199 struct DiagnosticEngineImpl {
200  /// Emit a diagnostic using the registered issue handle if present, or with
201  /// the default behavior if not.
202  void emit(Diagnostic diag);
203 
204  /// A mutex to ensure that diagnostics emission is thread-safe.
205  llvm::sys::SmartMutex<true> mutex;
206 
207  /// These are the handlers used to report diagnostics.
208  llvm::SmallMapVector<DiagnosticEngine::HandlerID, DiagnosticEngine::HandlerTy,
209  2>
210  handlers;
211 
212  /// This is a unique identifier counter for diagnostic handlers in the
213  /// context. This id starts at 1 to allow for 0 to be used as a sentinel.
214  DiagnosticEngine::HandlerID uniqueHandlerId = 1;
215 };
216 } // namespace detail
217 } // namespace mlir

219 /// Emit a diagnostic using the registered issue handle if present, or with
220 /// the default behavior if not.
221 void DiagnosticEngineImpl::emit(Diagnostic diag) {
222  llvm::sys::SmartScopedLock<true> lock(mutex);
223 
224  // Try to process the given diagnostic on one of the registered handlers.
225  // Handlers are walked in reverse order, so that the most recent handler is
226  // processed first.
227  for (auto &handlerIt : llvm::reverse(handlers))
228  if (succeeded(handlerIt.second(diag)))
229  return;
230 
231  // Otherwise, if this is an error we emit it to stderr.
232  if (diag.getSeverity() != DiagnosticSeverity::Error)
233  return;
234 
235  auto &os = llvm::errs();
236  if (!diag.getLocation().isa<UnknownLoc>())
237  os << diag.getLocation() << ": ";
238  os << "error: ";
239 
240  // The default behavior for errors is to emit them to stderr.
241  os << diag << '\n';
242  os.flush();
243 }

252 /// Register a new handler for diagnostics to the engine. This function returns
253 /// a unique identifier for the registered handler, which can be used to
254 /// unregister this handler at a later time.
255 auto DiagnosticEngine::registerHandler(const HandlerTy &handler) -> HandlerID {
256  llvm::sys::SmartScopedLock<true> lock(impl->mutex);
257  auto uniqueID = impl->uniqueHandlerId++;
258  impl->handlers.insert({uniqueID, handler});
259  return uniqueID;
260 }

276 /// Helper function used to emit a diagnostic with an optionally empty twine
277 /// message. If the message is empty, then it is not inserted into the
278 /// diagnostic.
279 static InFlightDiagnostic
280 emitDiag(Location location, DiagnosticSeverity severity, const Twine &message) {
281  auto &diagEngine = location->getContext()->getDiagEngine();
282  auto diag = diagEngine.emit(location, severity);
283  if (!message.isTriviallyEmpty())
284  diag << message;
285 
286  // Add the stack trace as a note if necessary.
287  if (printStackTraceOnDiagnostic) {
288  std::string bt;
289  {
290  llvm::raw_string_ostream stream(bt);
291  llvm::sys::PrintStackTrace(stream);
292  }
293  if (!bt.empty())
294  diag.attachNote() << "diagnostic emitted with trace:\n" << bt;
295  }
296 
297  return diag;
298 }

336 struct SourceMgrDiagnosticHandlerImpl {
337  /// Get a memory buffer for the given file, or nullptr if one is not found.
338  const llvm::MemoryBuffer *getBufferForFile(llvm::SourceMgr &mgr,
339  StringRef filename) {
340  // Check for an existing mapping to the buffer id for this file.
341  auto bufferIt = filenameToBuf.find(filename);
342  if (bufferIt != filenameToBuf.end())
343  return bufferIt->second;
344 
345  // Look for a buffer in the manager that has this filename.
346  for (unsigned i = 1, e = mgr.getNumBuffers() + 1; i != e; ++i) {
347  auto *buf = mgr.getMemoryBuffer(i);
348  if (buf->getBufferIdentifier() == filename)
349  return filenameToBuf[filename] = buf;
350  }
351 
352  // Otherwise, try to load the source file.
353  const llvm::MemoryBuffer *newBuf = nullptr;
354  std::string ignored;
355  if (auto newBufID =
356  mgr.AddIncludeFile(std::string(filename), llvm::SMLoc(), ignored))
357  newBuf = mgr.getMemoryBuffer(newBufID);
358  return filenameToBuf[filename] = newBuf;
359  }
360 
361  /// Mapping between file name and buffer pointer.
362  llvm::StringMap<const llvm::MemoryBuffer *> filenameToBuf;
363 };
364 } // end namespace detail
365 } // end namespace mlir

411 void SourceMgrDiagnosticHandler::emitDiagnostic(Location loc, Twine message,
412  DiagnosticSeverity kind) {
413  // Extract a file location from this loc.
414  auto fileLoc = getFileLineColLoc(loc);
415 
416  // If one doesn't exist, then print the raw message without a source location.
417  if (!fileLoc) {
418  std::string str;
419  llvm::raw_string_ostream strOS(str);
420  if (!loc.isa<UnknownLoc>())
421  strOS << loc << ": ";
422  strOS << message;
423  return mgr.PrintMessage(os, llvm::SMLoc(), getDiagKind(kind), strOS.str());
424  }
425 
426  // Otherwise, try to convert the file location to an SMLoc.
427  auto smloc = convertLocToSMLoc(*fileLoc);
428  if (smloc.isValid())
429  return mgr.PrintMessage(os, smloc, getDiagKind(kind), message);
430 
431  // If the conversion was unsuccessful, create a diagnostic with the file
432  // information.
433  llvm::SMDiagnostic diag(fileLoc->getFilename(), getDiagKind(kind),
434  message.str());
435  diag.print(nullptr, os);
436 }

438 /// Emit the given diagnostic with the held source manager.
439 void SourceMgrDiagnosticHandler::emitDiagnostic(Diagnostic &diag) {
440  // Emit the diagnostic.
441  auto loc = diag.getLocation();
442  emitDiagnostic(loc, diag.str(), diag.getSeverity());
443 
444  // If the diagnostic location was a call site location, then print the call
445  // stack as well.
446  if (auto callLoc = loc.dyn_cast<CallSiteLoc>()) {
447  // Print the call stack while valid, or until the limit is reached.
448  Location callerLoc = callLoc.getCaller();
449  for (unsigned curDepth = 0; curDepth < callStackLimit; ++curDepth) {
450  emitDiagnostic(callerLoc, "called from", DiagnosticSeverity::Note);
451  if ((callLoc = callerLoc.dyn_cast<CallSiteLoc>()))
452  callerLoc = callLoc.getCaller();
453  else
454  break;
455  }
456  }
457 
458  // Emit each of the notes.
459  for (auto &note : diag.getNotes())
460  emitDiagnostic(note.getLocation(), note.str(), note.getSeverity());
461 }

469 /// Get a memory buffer for the given file, or the main file of the source
470 /// manager if one doesn't exist. This always returns non-null.
471 llvm::SMLoc SourceMgrDiagnosticHandler::convertLocToSMLoc(FileLineColLoc loc) {
472  // Get the buffer for this filename.
473  auto *membuf = getBufferForFile(loc.getFilename());
474  if (!membuf)
475  return llvm::SMLoc();
476 
477  // TODO: This should really be upstreamed to be a method on llvm::SourceMgr.
478  // Doing so would allow it to use the offset cache that is already maintained
479  // by SrcBuffer, making this more efficient.
480  unsigned lineNo = loc.getLine();
481  unsigned columnNo = loc.getColumn();
482 
483  // Scan for the correct line number.
484  const char *position = membuf->getBufferStart();
485  const char *end = membuf->getBufferEnd();
486 
487  // We start counting line and column numbers from 1.
488  if (lineNo != 0)
489  --lineNo;
490  if (columnNo != 0)
491  --columnNo;
492 
493  while (position < end && lineNo) {
494  auto curChar = *position++;
495 
496  // Scan for newlines. If this isn't one, ignore it.
497  if (curChar != '\r' && curChar != '\n')
498  continue;
499 
500  // We saw a line break, decrement our counter.
501  --lineNo;
502 
503  // Check for \r\n and \n\r and treat it as a single escape. We know that
504  // looking past one character is safe because MemoryBuffer's are always nul
505  // terminated.
506  if (*position != curChar && (*position == '\r' || *position == '\n'))
507  ++position;
508  }
509 
510  // If the line/column counter was invalid, return a pointer to the start of
511  // the buffer.
512  if (lineNo || position + columnNo > end)
513  return llvm::SMLoc::getFromPointer(membuf->getBufferStart());
514 
515  // If the column is zero, try to skip to the first non-whitespace character.
516  if (columnNo == 0) {
517  auto isNewline = [](char c) { return c == '\n' || c == '\r'; };
518  auto isWhitespace = [](char c) { return c == ' ' || c == '\t'; };
519 
520  // Look for a valid non-whitespace character before the next line.
521  for (auto *newPos = position; newPos < end && !isNewline(*newPos); ++newPos)
522  if (!isWhitespace(*newPos))
523  return llvm::SMLoc::getFromPointer(newPos);
524  }
525 
526  // Otherwise return the right pointer.
527  return llvm::SMLoc::getFromPointer(position + columnNo);
528 }

593 /// Computes the expected diagnostics for the given source buffer.
594 MutableArrayRef<ExpectedDiag>
595 SourceMgrDiagnosticVerifierHandlerImpl::computeExpectedDiags(
596  const llvm::MemoryBuffer *buf) {
597  // If the buffer is invalid, return an empty list.
598  if (!buf)
599  return llvm::None;
600  auto &expectedDiags = expectedDiagsPerFile[buf->getBufferIdentifier()];
601 
602  // The number of the last line that did not correlate to a designator.
603  unsigned lastNonDesignatorLine = 0;
604 
605  // The indices of designators that apply to the next non designator line.
606  SmallVector<unsigned, 1> designatorsForNextLine;
607 
608  // Scan the file for expected-* designators.
609  SmallVector<StringRef, 100> lines;
610  buf->getBuffer().split(lines, '\n');
611  for (unsigned lineNo = 0, e = lines.size(); lineNo < e; ++lineNo) {
612  SmallVector<StringRef, 4> matches;
613  if (!expected.match(lines[lineNo], &matches)) {
614  // Check for designators that apply to this line.
615  if (!designatorsForNextLine.empty()) {
616  for (unsigned diagIndex : designatorsForNextLine)
617  expectedDiags[diagIndex].lineNo = lineNo + 1;
618  designatorsForNextLine.clear();
619  }
620  lastNonDesignatorLine = lineNo;
621  continue;
622  }
623 
624  // Point to the start of expected-*.
625  auto expectedStart = llvm::SMLoc::getFromPointer(matches[0].data());
626 
627  DiagnosticSeverity kind;
628  if (matches[1] == "error")
629  kind = DiagnosticSeverity::Error;
630  else if (matches[1] == "warning")
631  kind = DiagnosticSeverity::Warning;
632  else if (matches[1] == "remark")
633  kind = DiagnosticSeverity::Remark;
634  else {
635  assert(matches[1] == "note");
636  kind = DiagnosticSeverity::Note;
637  }
638 
639  ExpectedDiag record{kind, lineNo + 1, matches[4], expectedStart, false};
640  auto offsetMatch = matches[2];
641  if (!offsetMatch.empty()) {
642  offsetMatch = offsetMatch.drop_front(1);
643 
644  // Get the integer value without the @ and +/- prefix.
645  if (offsetMatch[0] == '+' || offsetMatch[0] == '-') {
646  int offset;
647  offsetMatch.drop_front().getAsInteger(0, offset);
648 
649  if (offsetMatch.front() == '+')
650  record.lineNo += offset;
651  else
652  record.lineNo -= offset;
653  } else if (offsetMatch.consume_front("above")) {
654  // If the designator applies 'above' we add it to the last non
655  // designator line.
656  record.lineNo = lastNonDesignatorLine + 1;
657  } else {
658  // Otherwise, this is a 'below' designator and applies to the next
659  // non-designator line.

702 /// Returns the status of the verifier and verifies that all expected
703 /// diagnostics were emitted. This return success if all diagnostics were
704 /// verified correctly, failure otherwise.
705 LogicalResult SourceMgrDiagnosticVerifierHandler::verify() {
706  // Verify that all expected errors were seen.
707  for (auto &expectedDiagsPair : impl->expectedDiagsPerFile) {
708  for (auto &err : expectedDiagsPair.second) {
709  if (err.matched)
710  continue;
711  llvm::SMRange range(err.fileLoc,
712  llvm::SMLoc::getFromPointer(err.fileLoc.getPointer() +
713  err.substring.size()));
714  mgr.PrintMessage(os, err.fileLoc, llvm::SourceMgr::DK_Error,
715  "expected " + getDiagKindStr(err.kind) + " \"" +
716  err.substring + "\" was not produced",
717  range);
718  impl->status = failure();
719  }
720  }
721  impl->expectedDiagsPerFile.clear();
722  return impl->status;
723 }

725 /// Process a single diagnostic.
726 void SourceMgrDiagnosticVerifierHandler::process(Diagnostic &diag) {
727  auto kind = diag.getSeverity();
728 
729  // Process a FileLineColLoc.
730  if (auto fileLoc = getFileLineColLoc(diag.getLocation()))
731  return process(*fileLoc, diag.str(), kind);
732 
733  emitDiagnostic(diag.getLocation(),
734  "unexpected " + getDiagKindStr(kind) + ": " + diag.str(),
735  DiagnosticSeverity::Error);
736  impl->status = failure();
737 }

739 /// Process a FileLineColLoc diagnostic.
740 void SourceMgrDiagnosticVerifierHandler::process(FileLineColLoc loc,
741  StringRef msg,
742  DiagnosticSeverity kind) {
743  // Get the expected diagnostics for this file.
744  auto diags = impl->getExpectedDiags(loc.getFilename());
745  if (!diags)
746  diags = impl->computeExpectedDiags(getBufferForFile(loc.getFilename()));
747 
748  // Search for a matching expected diagnostic.
749  // If we find something that is close then emit a more specific error.
750  ExpectedDiag *nearMiss = nullptr;
751 
752  // If this was an expected error, remember that we saw it and return.
753  unsigned line = loc.getLine();
754  for (auto &e : *diags) {
755  if (line == e.lineNo && msg.contains(e.substring)) {
756  if (e.kind == kind) {
757  e.matched = true;
758  return;
759  }
760 
761  // If this only differs based on the diagnostic kind, then consider it
762  // to be a near miss.
763  nearMiss = &e;
764  }
765  }
766 
767  // Otherwise, emit an error for the near miss.
768  if (nearMiss)
769  mgr.PrintMessage(os, nearMiss->fileLoc, llvm::SourceMgr::DK_Error,
770  "'" + getDiagKindStr(kind) +
771  "' diagnostic emitted when expecting a '" +
772  getDiagKindStr(nearMiss->kind) + "'");
773  else
774  emitDiagnostic(loc, "unexpected " + getDiagKindStr(kind) + ": " + msg,
775  DiagnosticSeverity::Error);
776  impl->status = failure();
777 }

795

800  ParallelDiagnosticHandlerImpl(MLIRContext *ctx) : handlerID(0), context(ctx) {
801  handlerID = ctx->getDiagEngine().registerHandler([this](Diagnostic &diag) {
802  uint64_t tid = llvm::get_threadid();
803  llvm::sys::SmartScopedLock<true> lock(mutex);
804 
805  // If this thread is not tracked, then return failure to let another
806  // handler process this diagnostic.
807  if (!threadToOrderID.count(tid))
808  return failure();
809 
810  // Append a new diagnostic.
811  diagnostics.emplace_back(threadToOrderID[tid], std::move(diag));
812  return success();
813  });
814  }
815 
816  ~ParallelDiagnosticHandlerImpl() override {
817  // Erase this handler from the context.
818  context->getDiagEngine().eraseHandler(handlerID);
819 
820  // Early exit if there are no diagnostics, this is the common case.
821  if (diagnostics.empty())
822  return;
823 
824  // Emit the diagnostics back to the context.
825  emitDiagnostics([&](Diagnostic diag) {
826  return context->getDiagEngine().emit(std::move(diag));
827  });
828  }

830  /// Utility method to emit any held diagnostics.
831  void emitDiagnostics(std::function<void(Diagnostic)> emitFn) const {
832  // Stable sort all of the diagnostics that were emitted. This creates a
833  // deterministic ordering for the diagnostics based upon which order id they
834  // were emitted for.
835  std::stable_sort(diagnostics.begin(), diagnostics.end());
836 
837  // Emit each diagnostic to the context again.
838  for (ThreadDiagnostic &diag : diagnostics)
839  emitFn(std::move(diag.diag));
840  }

885  });
886  }
