99  /// This class represents a single pass statistic. This statistic functions

187 /// The opaque CRTP model of a pass. This class provides utilities for derived
188 /// pass execution and handles all of the necessary polymorphic API.
189 template <typename PassT, typename BasePassT>
190 class PassModel : public BasePassT {
191 public:
192  /// Support isa/dyn_cast functionality for the derived pass class.
193  static bool classof(const Pass *pass) {
194  return pass->getPassID() == PassID::getID<PassT>();
195  }
196 
197 protected:
198  explicit PassModel(Optional<StringRef> opName = llvm::None)
199  : BasePassT(PassID::getID<PassT>(), opName) {}
200 
201  /// Signal that some invariant was broken when running. The IR is allowed to
202  /// be in an invalid state.
203  void signalPassFailure() {
204  this->getPassState().irAndPassFailed.setInt(true);
205  }
206 
207  /// Query an analysis for the current ir unit.
208  template <typename AnalysisT> AnalysisT &getAnalysis() {
209  return this->getAnalysisManager().template getAnalysis<AnalysisT>();
210  }
211 
212  /// Query a cached instance of an analysis for the current ir unit if one
213  /// exists.
214  template <typename AnalysisT>
215  Optional<std::reference_wrapper<AnalysisT>> getCachedAnalysis() {
216  return this->getAnalysisManager().template getCachedAnalysis<AnalysisT>();
217  }
218 
219  /// Mark all analyses as preserved.
220  void markAllAnalysesPreserved() {
221  this->getPassState().preservedAnalyses.preserveAll();
222  }
223 
224  /// Mark the provided analyses as preserved.
225  template <typename... AnalysesT> void markAnalysesPreserved() {
226  this->getPassState().preservedAnalyses.template preserve<AnalysesT...>();
227  }
228  void markAnalysesPreserved(const AnalysisID *id) {
229  this->getPassState().preservedAnalyses.preserve(id);
230  }
231 
232  /// Returns the derived pass name.
233  StringRef getName() override {
234  StringRef name = llvm::getTypeName<PassT>();
235  if (!name.consume_front("mlir::"))
236  name.consume_front("(anonymous namespace)::");
237  return name;
238  }
239 
240  /// A clone method to create a copy of this pass.
241  std::unique_ptr<Pass> clone() const override {
242  auto newInst = std::make_unique<PassT>(*static_cast<const PassT *>(this));
243  newInst->copyOptionValuesFrom(this);
244  return newInst;
245  }
246 
247  /// Returns the analysis for the parent operation if it exists.
248  template <typename AnalysisT>
249  Optional<std::reference_wrapper<AnalysisT>>
250  getCachedParentAnalysis(Operation *parent) {
251  return this->getAnalysisManager()
252  .template getCachedParentAnalysis<AnalysisT>(parent);
253  }
254  template <typename AnalysisT>
255  Optional<std::reference_wrapper<AnalysisT>> getCachedParentAnalysis() {
256  return this->getAnalysisManager()
257  .template getCachedParentAnalysis<AnalysisT>(
258  this->getOperation()->getParentOp());
259  }
260 
261  /// Returns the analysis for the given child operation if it exists.
262  template <typename AnalysisT>
263  Optional<std::reference_wrapper<AnalysisT>>
264  getCachedChildAnalysis(Operation *child) {
265  return this->getAnalysisManager()
266  .template getCachedChildAnalysis<AnalysisT>(child);
267  }
268 
269  /// Returns the analysis for the given child operation, or creates it if it
270  /// doesn't exist.
271  template <typename AnalysisT> AnalysisT &getChildAnalysis(Operation *child) {
272  return this->getAnalysisManager().template getChildAnalysis<AnalysisT>(
273  child);
274  }
275 };

290 /// Pass to transform an operation of a specific type.
291 ///
292 /// Operation passes must not:
293 /// - modify any other operations within the parent region, as other threads
294 /// may be manipulating them concurrently.
295 /// - modify any state within the parent operation, this includes adding
296 /// additional operations.
297 ///
298 /// Derived function passes are expected to provide the following:
299 /// - A 'void runOnOperation()' method.
300 template <typename PassT, typename OpT = void>
301 class OperationPass : public detail::PassModel<PassT, OpPassBase<OpT>> {
302 protected:
303  OperationPass()
304  : detail::PassModel<PassT, OpPassBase<OpT>>(OpT::getOperationName()) {}
305 
306  /// Return the current operation being transformed.
307  OpT getOperation() { return cast<OpT>(Pass::getOperation()); }
308 };

310 /// Pass to transform an operation.
311 ///
312 /// Operation passes must not:
313 /// - modify any other operations within the parent region, as other threads
314 /// may be manipulating them concurrently.
315 /// - modify any state within the parent operation, this includes adding
316 /// additional operations.
317 ///
318 /// Derived function passes are expected to provide the following:
319 /// - A 'void runOnOperation()' method.
320 template <typename PassT>
321 struct OperationPass<PassT, void> : public detail::PassModel<PassT, Pass> {};
322 
323 /// A model for providing function pass specific utilities.
324 ///
325 /// Derived function passes are expected to provide the following:
326 /// - A 'void runOnFunction()' method.
327 template <typename T> struct FunctionPass : public OperationPass<T, FuncOp> {
328  /// The polymorphic API that runs the pass over the currently held function.
329  virtual void runOnFunction() = 0;
330 
331  /// The polymorphic API that runs the pass over the currently held operation.
332  void runOnOperation() final {
333  if (!getFunction().isExternal())
334  runOnFunction();
335  }
336 
337  /// Return the current module being transformed.
338  FuncOp getFunction() { return this->getOperation(); }
339 };
340 
341 /// A model for providing module pass specific utilities.
342 ///
343 /// Derived module passes are expected to provide the following:
344 /// - A 'void runOnModule()' method.
345 template <typename T> struct ModulePass : public OperationPass<T, ModuleOp> {
346  /// The polymorphic API that runs the pass over the currently held module.
347  virtual void runOnModule() = 0;
348 
349  /// The polymorphic API that runs the pass over the currently held operation.
350  void runOnOperation() final { runOnModule(); }
351 
352  /// Return the current module being transformed.
353  ModuleOp getModule() { return this->getOperation(); }
354 };
