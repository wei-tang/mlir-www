26 /// According to the Vulkan spec "14.5.4. Offset and Stride Assignment":
27 /// "There are different alignment requirements depending on the specific
28 /// resources and on the features enabled on the device."
29 ///
30 /// There are 3 types of alignment: scalar, base, extended.
31 /// See the spec for details.
32 ///
33 /// Note: Even if scalar alignment is supported, it is generally more
34 /// performant to use the base alignment. So here the calculation is based on
35 /// base alignment.
36 ///
37 /// The memory layout must obey the following rules:
38 /// 1. The Offset decoration of any member must be a multiple of its alignment.
39 /// 2. Any ArrayStride or MatrixStride decoration must be a multiple of the
40 /// alignment of the array or matrix as defined above.
41 ///
42 /// According to the SPIR-V spec:
43 /// "The ArrayStride, MatrixStride, and Offset decorations must be large
44 /// enough to hold the size of the objects they affect (that is, specifying
45 /// overlap is invalid)."
46 class VulkanLayoutUtils {
47 public:
48  using Size = uint64_t;
49 
50  /// Returns a new StructType with layout info. Assigns the type size in bytes
51  /// to the `size`. Assigns the type alignment in bytes to the `alignment`.
52  static spirv::StructType decorateType(spirv::StructType structType,
53  Size &size, Size &alignment);
54  /// Checks whether a type is legal in terms of Vulkan layout info
55  /// decoration. A type is dynamically illegal if it's a composite type in the
56  /// StorageBuffer, PhysicalStorageBuffer, Uniform, and PushConstant Storage
57  /// Classes without layout information.
58  static bool isLegalType(Type type);
59 
60 private:
61  static Type decorateType(Type type, Size &size, Size &alignment);
62  static Type decorateType(VectorType vectorType, Size &size, Size &alignment);
63  static Type decorateType(spirv::ArrayType arrayType, Size &size,
64  Size &alignment);
65  /// Calculates the alignment for the given scalar type.
66  static Size getScalarTypeAlignment(Type scalarType);
67 };
