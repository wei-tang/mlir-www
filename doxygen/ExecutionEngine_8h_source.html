50 /// JIT-backed execution engine for MLIR modules. Assumes the module can be
51 /// converted to LLVM IR. For each function, creates a wrapper function with
52 /// the fixed interface
53 ///
54 /// void _mlir_funcName(void **)
55 ///
56 /// where the only argument is interpreted as a list of pointers to the actual
57 /// arguments of the function, followed by a pointer to the result. This allows
58 /// the engine to provide the caller with a generic function pointer that can
59 /// be used to invoke the JIT-compiled function.
60 class ExecutionEngine {
61 public:
62  ExecutionEngine(bool enableObjectCache);
63 
64  /// Creates an execution engine for the given module. If `transformer` is
65  /// provided, it will be called on the LLVM module during JIT-compilation and
66  /// can be used, e.g., for reporting or optimization. `jitCodeGenOptLevel`,
67  /// when provided, is used as the optimization level for target code
68  /// generation. If `sharedLibPaths` are provided, the underlying
69  /// JIT-compilation will open and link the shared libraries for symbol
70  /// resolution. If `objectCache` is provided, JIT compiler will use it to
71  /// store the object generated for the given module.
72  static llvm::Expected<std::unique_ptr<ExecutionEngine>> create(
73  ModuleOp m, std::function<llvm::Error(llvm::Module *)> transformer = {},
74  Optional<llvm::CodeGenOpt::Level> jitCodeGenOptLevel = llvm::None,
75  ArrayRef<StringRef> sharedLibPaths = {}, bool enableObjectCache = false);
76 
77  /// Looks up a packed-argument function with the given name and returns a
78  /// pointer to it. Propagates errors in case of failure.
79  llvm::Expected<void (*)(void **)> lookup(StringRef name) const;
80 
81  /// Invokes the function with the given name passing it the list of arguments.
82  /// The arguments are accepted by lvalue-reference since the packed function
83  /// interface expects a list of non-null pointers.

111 template <typename... Args>
112 llvm::Error ExecutionEngine::invoke(StringRef name, Args &... args) {
113  auto expectedFPtr = lookup(name);
114  if (!expectedFPtr)
115  return expectedFPtr.takeError();
116  auto fptr = *expectedFPtr;
117 
118  SmallVector<void *, 8> packedArgs{static_cast<void *>(&args)...};
119  (*fptr)(packedArgs.data());
120 
121  return llvm::Error::success();
122 }
