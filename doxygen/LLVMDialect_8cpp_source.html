50 // <operation> ::= `llvm.icmp` string-literal ssa-use `,` ssa-use
51 // attribute-dict? `:` type
52 // <operation> ::= `llvm.fcmp` string-literal ssa-use `,` ssa-use
53 // attribute-dict? `:` type
54 template <typename CmpPredicateType>
55 static ParseResult parseCmpOp(OpAsmParser &parser, OperationState &result) {
56  Builder &builder = parser.getBuilder();
57 
58  StringAttr predicateAttr;
59  OpAsmParser::OperandType lhs, rhs;
60  Type type;
61  llvm::SMLoc predicateLoc, trailingTypeLoc;
62  if (parser.getCurrentLocation(&predicateLoc) ||
63  parser.parseAttribute(predicateAttr, "predicate", result.attributes) ||
64  parser.parseOperand(lhs) || parser.parseComma() ||
65  parser.parseOperand(rhs) ||
66  parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
67  parser.getCurrentLocation(&trailingTypeLoc) || parser.parseType(type) ||
68  parser.resolveOperand(lhs, type, result.operands) ||
69  parser.resolveOperand(rhs, type, result.operands))
70  return failure();
71 
72  // Replace the string attribute `predicate` with an integer attribute.
73  int64_t predicateValue = 0;
74  if (std::is_same<CmpPredicateType, ICmpPredicate>()) {
75  Optional<ICmpPredicate> predicate =
76  symbolizeICmpPredicate(predicateAttr.getValue());
77  if (!predicate)
78  return parser.emitError(predicateLoc)
79  << "'" << predicateAttr.getValue()
80  << "' is an incorrect value of the 'predicate' attribute";
81  predicateValue = static_cast<int64_t>(predicate.getValue());
82  } else {
83  Optional<FCmpPredicate> predicate =
84  symbolizeFCmpPredicate(predicateAttr.getValue());
85  if (!predicate)
86  return parser.emitError(predicateLoc)
87  << "'" << predicateAttr.getValue()
88  << "' is an incorrect value of the 'predicate' attribute";
89  predicateValue = static_cast<int64_t>(predicate.getValue());
90  }
91 
92  result.attributes[0].second =
93  parser.getBuilder().getI64IntegerAttr(predicateValue);
94 
95  // The result type is either i1 or a vector type <? x i1> if the inputs are
96  // vectors.
97  auto *dialect = builder.getContext()->getRegisteredDialect<LLVMDialect>();
98  auto resultType = LLVMType::getInt1Ty(dialect);
99  auto argType = type.dyn_cast<LLVM::LLVMType>();
100  if (!argType)
101  return parser.emitError(trailingTypeLoc, "expected LLVM IR dialect type");
102  if (argType.getUnderlyingType()->isVectorTy())
103  resultType = LLVMType::getVectorTy(
104  resultType, argType.getUnderlyingType()->getVectorNumElements());
105 
106  result.addTypes({resultType});
107  return success();
108 }

279 /// <operation> ::= `llvm.invoke` (function-id | ssa-use) `(` ssa-use-list `)`
280 /// `to` bb-id (`[` ssa-use-and-type-list `]`)?
281 /// `unwind` bb-id (`[` ssa-use-and-type-list `]`)?
282 /// attribute-dict? `:` function-type
283 static ParseResult parseInvokeOp(OpAsmParser &parser, OperationState &result) {
284  SmallVector<OpAsmParser::OperandType, 8> operands;
285  FunctionType funcType;
286  SymbolRefAttr funcAttr;
287  llvm::SMLoc trailingTypeLoc;
288  Block *normalDest, *unwindDest;
289  SmallVector<Value, 4> normalOperands, unwindOperands;
290 
291  // Parse an operand list that will, in practice, contain 0 or 1 operand. In
292  // case of an indirect call, there will be 1 operand before `(`. In case of a
293  // direct call, there will be no operands and the parser will stop at the
294  // function identifier without complaining.
295  if (parser.parseOperandList(operands))
296  return failure();
297  bool isDirect = operands.empty();
298 
299  // Optionally parse a function identifier.
300  if (isDirect && parser.parseAttribute(funcAttr, "callee", result.attributes))
301  return failure();
302 
303  if (parser.parseOperandList(operands, OpAsmParser::Delimiter::Paren) ||
304  parser.parseKeyword("to") ||
305  parser.parseSuccessorAndUseList(normalDest, normalOperands) ||
306  parser.parseKeyword("unwind") ||
307  parser.parseSuccessorAndUseList(unwindDest, unwindOperands) ||
308  parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
309  parser.getCurrentLocation(&trailingTypeLoc) || parser.parseType(funcType))
310  return failure();
311 
312  if (isDirect) {
313  // Make sure types match.
314  if (parser.resolveOperands(operands, funcType.getInputs(),
315  parser.getNameLoc(), result.operands))
316  return failure();
317  result.addTypes(funcType.getResults());
318  } else {
319  // Construct the LLVM IR Dialect function type that the first operand
320  // should match.
321  if (funcType.getNumResults() > 1)
322  return parser.emitError(trailingTypeLoc,
323  "expected function with 0 or 1 result");
324 
325  Builder &builder = parser.getBuilder();
326  auto *llvmDialect =
327  builder.getContext()->getRegisteredDialect<LLVM::LLVMDialect>();
328  LLVM::LLVMType llvmResultType;
329  if (funcType.getNumResults() == 0) {
330  llvmResultType = LLVM::LLVMType::getVoidTy(llvmDialect);
331  } else {
332  llvmResultType = funcType.getResult(0).dyn_cast<LLVM::LLVMType>();
333  if (!llvmResultType)
334  return parser.emitError(trailingTypeLoc,
335  "expected result to have LLVM type");
336  }
337 
338  SmallVector<LLVM::LLVMType, 8> argTypes;
339  argTypes.reserve(funcType.getNumInputs());
340  for (Type ty : funcType.getInputs()) {
341  if (auto argType = ty.dyn_cast<LLVM::LLVMType>())
342  argTypes.push_back(argType);
343  else
344  return parser.emitError(trailingTypeLoc,
345  "expected LLVM types as inputs");
346  }
347 
348  auto llvmFuncType = LLVM::LLVMType::getFunctionTy(llvmResultType, argTypes,
349  /*isVarArg=*/false);
350  auto wrappedFuncType = llvmFuncType.getPointerTo();
351 
352  auto funcArguments = llvm::makeArrayRef(operands).drop_front();
353 
354  // Make sure that the first operand (indirect callee) matches the wrapped
355  // LLVM IR function type, and that the types of the other call operands
356  // match the types of the function arguments.
357  if (parser.resolveOperand(operands[0], wrappedFuncType, result.operands) ||
358  parser.resolveOperands(funcArguments, funcType.getInputs(),
359  parser.getNameLoc(), result.operands))
360  return failure();
361 
362  result.addTypes(llvmResultType);
363  }
364  result.addSuccessor(normalDest, normalOperands);
365  result.addSuccessor(unwindDest, unwindOperands);
366  return success();
367 }

373 static LogicalResult verify(LandingpadOp op) {
374  Value value;
375 
376  if (!op.cleanup() && op.getOperands().empty())
377  return op.emitError("landingpad instruction expects at least one clause or "
378  "cleanup attribute");
379 
380  for (unsigned idx = 0, ie = op.getNumOperands(); idx < ie; idx++) {
381  value = op.getOperand(idx);
382  bool isFilter = value.getType().cast<LLVMType>().isArrayTy();
383  if (isFilter) {
384  // FIXME: Verify filter clauses when arrays are appropriately handled
385  } else {
386  // catch - global addresses only.
387  // Bitcast ops should have global addresses as their args.
388  if (auto bcOp = dyn_cast_or_null<BitcastOp>(value.getDefiningOp())) {
389  if (auto addrOp =
390  dyn_cast_or_null<AddressOfOp>(bcOp.arg().getDefiningOp()))
391  continue;
392  return op.emitError("constant clauses expected")
393  .attachNote(bcOp.getLoc())
394  << "global addresses expected as operand to "
395  "bitcast used in clauses for landingpad";
396  }
397  // NullOp and AddressOfOp allowed
398  if (dyn_cast_or_null<NullOp>(value.getDefiningOp()))
399  continue;
400  if (dyn_cast_or_null<AddressOfOp>(value.getDefiningOp()))
401  continue;
402  return op.emitError("clause #")
403  << idx << " is not a known constant - null, addressof, bitcast";
404  }
405  }
406  return success();
407 }

426 /// <operation> ::= `llvm.landingpad` `cleanup`?
427 /// ((`catch` | `filter`) operand-type ssa-use)* attribute-dict?
428 static ParseResult parseLandingpadOp(OpAsmParser &parser,
429  OperationState &result) {
430  // Check for cleanup
431  if (succeeded(parser.parseOptionalKeyword("cleanup")))
432  result.addAttribute("cleanup", parser.getBuilder().getUnitAttr());
433 
434  // Parse clauses with types
435  while (succeeded(parser.parseOptionalLParen()) &&
436  (succeeded(parser.parseOptionalKeyword("filter")) ||
437  succeeded(parser.parseOptionalKeyword("catch")))) {
438  OpAsmParser::OperandType operand;
439  Type ty;
440  if (parser.parseOperand(operand) || parser.parseColon() ||
441  parser.parseType(ty) ||
442  parser.resolveOperand(operand, ty, result.operands) ||
443  parser.parseRParen())
444  return failure();
445  }
446 
447  Type type;
448  if (parser.parseColon() || parser.parseType(type))
449  return failure();
450 
451  result.addTypes(type);
452  return success();
453 }

483 // attribute-dict? `:` function-type
484 static ParseResult parseCallOp(OpAsmParser &parser, OperationState &result) {
485  SmallVector<OpAsmParser::OperandType, 8> operands;
486  Type type;
487  SymbolRefAttr funcAttr;
488  llvm::SMLoc trailingTypeLoc;
489 
490  // Parse an operand list that will, in practice, contain 0 or 1 operand. In
491  // case of an indirect call, there will be 1 operand before `(`. In case of a
492  // direct call, there will be no operands and the parser will stop at the
493  // function identifier without complaining.
494  if (parser.parseOperandList(operands))
495  return failure();

497 
498  // Optionally parse a function identifier.
499  if (isDirect)
500  if (parser.parseAttribute(funcAttr, "callee", result.attributes))
501  return failure();
502 
503  if (parser.parseOperandList(operands, OpAsmParser::Delimiter::Paren) ||
504  parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
505  parser.getCurrentLocation(&trailingTypeLoc) || parser.parseType(type))
506  return failure();
507 
508  auto funcType = type.dyn_cast<FunctionType>();
509  if (!funcType)
510  return parser.emitError(trailingTypeLoc, "expected function type");
511  if (isDirect) {
512  // Make sure types match.
513  if (parser.resolveOperands(operands, funcType.getInputs(),
514  parser.getNameLoc(), result.operands))
515  return failure();
516  result.addTypes(funcType.getResults());
517  } else {
518  // Construct the LLVM IR Dialect function type that the first operand
519  // should match.
520  if (funcType.getNumResults() > 1)
521  return parser.emitError(trailingTypeLoc,
522  "expected function with 0 or 1 result");
523 
524  Builder &builder = parser.getBuilder();
525  auto *llvmDialect =
526  builder.getContext()->getRegisteredDialect<LLVM::LLVMDialect>();
527  LLVM::LLVMType llvmResultType;
528  if (funcType.getNumResults() == 0) {
529  llvmResultType = LLVM::LLVMType::getVoidTy(llvmDialect);
530  } else {
531  llvmResultType = funcType.getResult(0).dyn_cast<LLVM::LLVMType>();
532  if (!llvmResultType)
533  return parser.emitError(trailingTypeLoc,
534  "expected result to have LLVM type");
535  }
536 
537  SmallVector<LLVM::LLVMType, 8> argTypes;
538  argTypes.reserve(funcType.getNumInputs());
539  for (int i = 0, e = funcType.getNumInputs(); i < e; ++i) {
540  auto argType = funcType.getInput(i).dyn_cast<LLVM::LLVMType>();
541  if (!argType)
542  return parser.emitError(trailingTypeLoc,
543  "expected LLVM types as inputs");

545  }

547  /*isVarArg=*/false);
548  auto wrappedFuncType = llvmFuncType.getPointerTo();
549

552 
553  // Make sure that the first operand (indirect callee) matches the wrapped
554  // LLVM IR function type, and that the types of the other call operands
555  // match the types of the function arguments.
556  if (parser.resolveOperand(operands[0], wrappedFuncType, result.operands) ||
557  parser.resolveOperands(funcArguments, funcType.getInputs(),
558  parser.getNameLoc(), result.operands))
559  return failure();
560 
561  result.addTypes(llvmResultType);
562  }
563 
564  return success();

623 // `containerType`. Position is an integer array attribute where each value
624 // is a zero-based position of the element in the aggregate type. Return the
625 // resulting type wrapped in MLIR, or nullptr on error.
626 static LLVM::LLVMType getInsertExtractValueElementType(OpAsmParser &parser,
627  Type containerType,
628  ArrayAttr positionAttr,
629  llvm::SMLoc attributeLoc,
630  llvm::SMLoc typeLoc) {

632  if (!wrappedContainerType)
633  return parser.emitError(typeLoc, "expected LLVM IR Dialect type"), nullptr;

636  // type by taking the element type, indexed by the position attribute for
637  // structures. Check the position index before accessing, it is supposed to
638  // be in bounds.
639  for (Attribute subAttr : positionAttr) {
640  auto positionElementAttr = subAttr.dyn_cast<IntegerAttr>();

647  if (llvmContainerType->isArrayTy()) {
648  if (position < 0 || static_cast<unsigned>(position) >=
649  llvmContainerType->getArrayNumElements())
650  return parser.emitError(attributeLoc, "position out of bounds"),
651  nullptr;

653  } else if (llvmContainerType->isStructTy()) {

659  wrappedContainerType.getStructElementType(position);
660  } else {
661  return parser.emitError(typeLoc,

663  nullptr;

1209 // Builds an LLVM function type from the given lists of input and output types.
1210 // Returns a null type if any of the types provided are non-LLVM types, or if
1211 // there is more than one output type.
1212 static Type buildLLVMFunctionType(OpAsmParser &parser, llvm::SMLoc loc,
1213  ArrayRef<Type> inputs, ArrayRef<Type> outputs,
1214  impl::VariadicFlag variadicFlag) {
1215  Builder &b = parser.getBuilder();
1216  if (outputs.size() > 1) {
1217  parser.emitError(loc, "failed to construct function type: expected zero or "
1218  "one function result");
1219  return {};
1220  }
1221 
1222  // Convert inputs to LLVM types, exit early on error.
1223  SmallVector<LLVMType, 4> llvmInputs;
1224  for (auto t : inputs) {
1225  auto llvmTy = t.dyn_cast<LLVMType>();
1226  if (!llvmTy) {
1227  parser.emitError(loc, "failed to construct function type: expected LLVM "
1228  "type for function arguments");
1229  return {};
1230  }
1231  llvmInputs.push_back(llvmTy);
1232  }
1233 
1234  // Get the dialect from the input type, if any exist. Look it up in the
1235  // context otherwise.
1236  LLVMDialect *dialect =
1237  llvmInputs.empty() ? b.getContext()->getRegisteredDialect<LLVMDialect>()
1238  : &llvmInputs.front().getDialect();
1239 
1240  // No output is denoted as "void" in LLVM type system.
1241  LLVMType llvmOutput = outputs.empty() ? LLVMType::getVoidTy(dialect)
1242  : outputs.front().dyn_cast<LLVMType>();
1243  if (!llvmOutput) {
1244  parser.emitError(loc, "failed to construct function type: expected LLVM "
1245  "type for function results");
1246  return {};
1247  }
1248  return LLVMType::getFunctionTy(llvmOutput, llvmInputs,
1249  variadicFlag.isVariadic());

1252 // Parses an LLVM function.
1253 //
1254 // operation ::= `llvm.func` linkage? function-signature function-attributes?
1255 // function-body
1256 //
1257 static ParseResult parseLLVMFuncOp(OpAsmParser &parser,
1258  OperationState &result) {
1259  // Default to external linkage if no keyword is provided.
1260  if (failed(parseOptionalLLVMKeyword<Linkage>(parser, result,
1261  getLinkageAttrName())))
1262  result.addAttribute(getLinkageAttrName(),
1263  parser.getBuilder().getI64IntegerAttr(
1264  static_cast<int64_t>(LLVM::Linkage::External)));
1265 
1266  StringAttr nameAttr;
1267  SmallVector<OpAsmParser::OperandType, 8> entryArgs;
1268  SmallVector<SmallVector<NamedAttribute, 2>, 1> argAttrs;
1269  SmallVector<SmallVector<NamedAttribute, 2>, 1> resultAttrs;
1270  SmallVector<Type, 8> argTypes;
1271  SmallVector<Type, 4> resultTypes;
1272  bool isVariadic;
1273 
1274  auto signatureLocation = parser.getCurrentLocation();
1275  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
1276  result.attributes) ||
1277  impl::parseFunctionSignature(parser, /*allowVariadic=*/true, entryArgs,
1278  argTypes, argAttrs, isVariadic, resultTypes,
1279  resultAttrs))
1280  return failure();
1281 
1282  auto type =
1283  buildLLVMFunctionType(parser, signatureLocation, argTypes, resultTypes,
1284  impl::VariadicFlag(isVariadic));
1285  if (!type)
1286  return failure();
1287  result.addAttribute(impl::getTypeAttrName(), TypeAttr::get(type));
1288 
1289  if (failed(parser.parseOptionalAttrDictWithKeyword(result.attributes)))
1290  return failure();
1291  impl::addArgAndResultAttrs(parser.getBuilder(), result, argAttrs,
1292  resultAttrs);
1293 
1294  auto *body = result.addRegion();
1295  return parser.parseOptionalRegion(
1296  *body, entryArgs, entryArgs.empty() ? ArrayRef<Type>() : argTypes);
1297 }

1363 // Verifies LLVM- and implementation-specific properties of the LLVM func Op:
1364 // - functions don't have 'common' linkage
1365 // - external functions have 'external' or 'extern_weak' linkage;
1366 // - vararg is (currently) only supported for external functions;
1367 // - entry block arguments are of LLVM types and match the function signature.
1368 static LogicalResult verify(LLVMFuncOp op) {
1369  if (op.linkage() == LLVM::Linkage::Common)
1370  return op.emitOpError()
1371  << "functions cannot have '"
1372  << stringifyLinkage(LLVM::Linkage::Common) << "' linkage";
1373 
1374  if (op.isExternal()) {
1375  if (op.linkage() != LLVM::Linkage::External &&
1376  op.linkage() != LLVM::Linkage::ExternWeak)
1377  return op.emitOpError()
1378  << "external functions must have '"
1379  << stringifyLinkage(LLVM::Linkage::External) << "' or '"
1380  << stringifyLinkage(LLVM::Linkage::ExternWeak) << "' linkage";
1381  return success();
1382  }
1383 
1384  if (op.isVarArg())
1385  return op.emitOpError("only external functions can be variadic");
1386 
1387  auto *funcType = cast<llvm::FunctionType>(op.getType().getUnderlyingType());
1388  unsigned numArguments = funcType->getNumParams();
1389  Block &entryBlock = op.front();
1390  for (unsigned i = 0; i < numArguments; ++i) {
1391  Type argType = entryBlock.getArgument(i).getType();
1392  auto argLLVMType = argType.dyn_cast<LLVMType>();
1393  if (!argLLVMType)
1394  return op.emitOpError("entry block argument #")
1395  << i << " is not of LLVM type";
1396  if (funcType->getParamType(i) != argLLVMType.getUnderlyingType())
1397  return op.emitOpError("the type of entry block argument #")
1398  << i << " does not match the function signature";
1399  }
1400 
1401  return success();
1402 }
