31 /// Interface of the function that translates the sources managed by `sourceMgr`
32 /// to MLIR. The source manager has at least one buffer. The implementation
33 /// should create a new MLIR ModuleOp in the given context and return a pointer
34 /// to it, or a nullptr in case of any error.
35 using TranslateSourceMgrToMLIRFunction =
36  std::function<OwningModuleRef(llvm::SourceMgr &sourceMgr, MLIRContext *)>;
37 
38 /// Interface of the function that translates the given string to MLIR. The
39 /// implementation should create a new MLIR ModuleOp in the given context. If
40 /// source-related error reporting is required from within the function, use
41 /// TranslateSourceMgrToMLIRFunction instead.
42 using TranslateStringRefToMLIRFunction =
43  std::function<OwningModuleRef(llvm::StringRef, MLIRContext *)>;
44 
45 /// Interface of the function that translates MLIR to a different format and
46 /// outputs the result to a stream. It is allowed to modify the module.
47 using TranslateFromMLIRFunction =
48  std::function<LogicalResult(ModuleOp, llvm::raw_ostream &output)>;
49 
50 /// Interface of the function that performs file-to-file translation involving
51 /// MLIR. The input file is held in the given MemoryBuffer; the output file
52 /// should be written to the given raw_ostream. The implementation should create
53 /// all MLIR constructs needed during the process inside the given context. This
54 /// can be used for round-tripping external formats through the MLIR system.
55 using TranslateFunction = std::function<LogicalResult(
56  llvm::SourceMgr &sourceMgr, llvm::raw_ostream &output, MLIRContext *)>;
