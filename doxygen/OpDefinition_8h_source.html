259 /// This template defines the foldHook as used by AbstractOperation.
260 ///
261 /// The default implementation uses a general fold method that can be defined on
262 /// custom ops which can return multiple results.
263 template <typename ConcreteType, bool isSingleResult, typename = void>
264 class FoldingHook {
265 public:
266  /// This is an implementation detail of the constant folder hook for
267  /// AbstractOperation.
268  static LogicalResult foldHook(Operation *op, ArrayRef<Attribute> operands,
269  SmallVectorImpl<OpFoldResult> &results) {
270  return cast<ConcreteType>(op).fold(operands, results);
271  }
272 
273  /// This hook implements a generalized folder for this operation. Operations
274  /// can implement this to provide simplifications rules that are applied by
275  /// the Builder::createOrFold API and the canonicalization pass.
276  ///
277  /// This is an intentionally limited interface - implementations of this hook
278  /// can only perform the following changes to the operation:
279  ///
280  /// 1. They can leave the operation alone and without changing the IR, and
281  /// return failure.
282  /// 2. They can mutate the operation in place, without changing anything else
283  /// in the IR. In this case, return success.
284  /// 3. They can return a list of existing values that can be used instead of
285  /// the operation. In this case, fill in the results list and return
286  /// success. The caller will remove the operation and use those results
287  /// instead.
288  ///
289  /// This allows expression of some simple in-place canonicalizations (e.g.
290  /// "x+0 -> x", "min(x,y,x,z) -> min(x,y,z)", "x+y-x -> y", etc), as well as
291  /// generalized constant folding.
292  ///
293  /// If not overridden, this fallback implementation always fails to fold.
294  ///
295  LogicalResult fold(ArrayRef<Attribute> operands,
296  SmallVectorImpl<OpFoldResult> &results) {
297  return failure();
298  }
299 };

301 /// This template specialization defines the foldHook as used by
302 /// AbstractOperation for single-result operations. This gives the hook a nicer
303 /// signature that is easier to implement.
304 template <typename ConcreteType, bool isSingleResult>
305 class FoldingHook<ConcreteType, isSingleResult,
306  typename std::enable_if<isSingleResult>::type> {
307 public:
308  /// If the operation returns a single value, then the Op can be implicitly
309  /// converted to an Value. This yields the value of the only result.
310  operator Value() {
311  return static_cast<ConcreteType *>(this)->getOperation()->getResult(0);
312  }
313 
314  /// This is an implementation detail of the constant folder hook for
315  /// AbstractOperation.
316  static LogicalResult foldHook(Operation *op, ArrayRef<Attribute> operands,
317  SmallVectorImpl<OpFoldResult> &results) {
318  auto result = cast<ConcreteType>(op).fold(operands);
319  if (!result)
320  return failure();
321 
322  // Check if the operation was folded in place. In this case, the operation
323  // returns itself.
324  if (result.template dyn_cast<Value>() != op->getResult(0))
325  results.push_back(result);
326  return success();
327  }
328 
329  /// This hook implements a generalized folder for this operation. Operations
330  /// can implement this to provide simplifications rules that are applied by
331  /// the Builder::createOrFold API and the canonicalization pass.
332  ///
333  /// This is an intentionally limited interface - implementations of this hook
334  /// can only perform the following changes to the operation:
335  ///
336  /// 1. They can leave the operation alone and without changing the IR, and
337  /// return nullptr.
338  /// 2. They can mutate the operation in place, without changing anything else
339  /// in the IR. In this case, return the operation itself.
340  /// 3. They can return an existing SSA value that can be used instead of
341  /// the operation. In this case, return that value. The caller will
342  /// remove the operation and use that result instead.
343  ///
344  /// This allows expression of some simple in-place canonicalizations (e.g.
345  /// "x+0 -> x", "min(x,y,x,z) -> min(x,y,z)", "x+y-x -> y", etc), as well as
346  /// generalized constant folding.
347  ///
348  /// If not overridden, this fallback implementation always fails to fold.
349  ///
350  OpFoldResult fold(ArrayRef<Attribute> operands) { return {}; }
351 };

387 /// Helper class for implementing traits. Clients are not expected to interact
388 /// with this directly, so its members are all protected.
389 template <typename ConcreteType, template <typename> class TraitType>
390 class TraitBase {
391 protected:
392  /// Return the ultimate Operation being worked on.
393  Operation *getOperation() {
394  // We have to cast up to the trait type, then to the concrete type, then to
395  // the BaseState class in explicit hops because the concrete type will
396  // multiply derive from the (content free) TraitBase class, and we need to
397  // be able to disambiguate the path for the C++ compiler.
398  auto *trait = static_cast<TraitType<ConcreteType> *>(this);
399  auto *concrete = static_cast<ConcreteType *>(trait);
400  auto *base = static_cast<OpState *>(concrete);
401  return base->getOperation();
402  }
403 
404  /// Provide default implementations of trait hooks. This allows traits to
405  /// provide exactly the overrides they care about.
406  static LogicalResult verifyTrait(Operation *op) { return success(); }
407  static AbstractOperation::OperationProperties getTraitProperties() {
408  return 0;
409  }
410 };
