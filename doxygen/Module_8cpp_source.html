35 ParseResult ModuleOp::parse(OpAsmParser &parser, OperationState &result) {
36  // If the name is present, parse it.
37  StringAttr nameAttr;
38  (void)parser.parseOptionalSymbolName(
39  nameAttr, mlir::SymbolTable::getSymbolAttrName(), result.attributes);
40 
41  // If module attributes are present, parse them.
42  if (parser.parseOptionalAttrDictWithKeyword(result.attributes))
43  return failure();
44 
45  // Parse the module body.
46  auto *body = result.addRegion();
47  if (parser.parseRegion(*body, llvm::None, llvm::None))
48  return failure();
49 
50  // Ensure that this module has a valid terminator.
51  ensureTerminator(*body, parser.getBuilder(), result.location);
52  return success();
53 }

72 LogicalResult ModuleOp::verify() {
73  auto &bodyRegion = getOperation()->getRegion(0);
74 
75  // The body must contain a single basic block.
76  if (!has_single_element(bodyRegion))
77  return emitOpError("expected body region to have a single block");
78 
79  // Check that the body has no block arguments.
80  auto *body = &bodyRegion.front();
81  if (body->getNumArguments() != 0)
82  return emitOpError("expected body to have no arguments");
83 
84  // Check that none of the attributes are non-dialect attributes, except for
85  // the symbol related attributes.
86  for (auto attr : getOperation()->getAttrList().getAttrs()) {
87  if (!attr.first.strref().contains('.') &&
88  !llvm::is_contained(
89  ArrayRef<StringRef>{mlir::SymbolTable::getSymbolAttrName(),
90  mlir::SymbolTable::getVisibilityAttrName()},
91  attr.first.strref()))
92  return emitOpError(
93  "can only contain dialect-specific attributes, found: '")
94  << attr.first << "'";
95  }
96 
97  return success();
98 }
