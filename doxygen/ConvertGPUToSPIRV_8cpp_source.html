118 PatternMatchResult
119 ForOpConversion::matchAndRewrite(loop::ForOp forOp, ArrayRef<Value> operands,
120  ConversionPatternRewriter &rewriter) const {
121  // loop::ForOp can be lowered to the structured control flow represented by
122  // spirv::LoopOp by making the continue block of the spirv::LoopOp the loop
123  // latch and the merge block the exit block. The resulting spirv::LoopOp has a
124  // single back edge from the continue to header block, and a single exit from
125  // header to merge.
126  loop::ForOpOperandAdaptor forOperands(operands);
127  auto loc = forOp.getLoc();
128  auto loopControl = rewriter.getI32IntegerAttr(
129  static_cast<uint32_t>(spirv::LoopControl::None));
130  auto loopOp = rewriter.create<spirv::LoopOp>(loc, loopControl);
131  loopOp.addEntryAndMergeBlock();
132 
133  OpBuilder::InsertionGuard guard(rewriter);
134  // Create the block for the header.
135  auto header = new Block();
136  // Insert the header.
137  loopOp.body().getBlocks().insert(std::next(loopOp.body().begin(), 1), header);
138 
139  // Create the new induction variable to use.
140  BlockArgument newIndVar =
141  header->addArgument(forOperands.lowerBound().getType());
142  Block *body = forOp.getBody();
143 
144  // Apply signature conversion to the body of the forOp. It has a single block,
145  // with argument which is the induction variable. That has to be replaced with
146  // the new induction variable.
147  TypeConverter::SignatureConversion signatureConverter(
148  body->getNumArguments());
149  signatureConverter.remapInput(0, newIndVar);
150  body = rewriter.applySignatureConversion(&forOp.getLoopBody(),
151  signatureConverter);
152 
153  // Delete the loop terminator.
154  rewriter.eraseOp(body->getTerminator());
155 
156  // Move the blocks from the forOp into the loopOp. This is the body of the
157  // loopOp.
158  rewriter.inlineRegionBefore(forOp.getOperation()->getRegion(0), loopOp.body(),
159  std::next(loopOp.body().begin(), 2));
160 
161  // Branch into it from the entry.
162  rewriter.setInsertionPointToEnd(&(loopOp.body().front()));
163  rewriter.create<spirv::BranchOp>(loc, header, forOperands.lowerBound());
164 
165  // Generate the rest of the loop header.
166  rewriter.setInsertionPointToEnd(header);
167  auto mergeBlock = loopOp.getMergeBlock();
168  auto cmpOp = rewriter.create<spirv::SLessThanOp>(
169  loc, rewriter.getI1Type(), newIndVar, forOperands.upperBound());
170  rewriter.create<spirv::BranchConditionalOp>(
171  loc, cmpOp, body, ArrayRef<Value>(), mergeBlock, ArrayRef<Value>());
172 
173  // Generate instructions to increment the step of the induction variable and
174  // branch to the header.
175  Block *continueBlock = loopOp.getContinueBlock();
176  rewriter.setInsertionPointToEnd(continueBlock);
177 
178  // Add the step to the induction variable and branch to the header.
179  Value updatedIndVar = rewriter.create<spirv::IAddOp>(
180  loc, newIndVar.getType(), newIndVar, forOperands.step());
181  rewriter.create<spirv::BranchOp>(loc, header, updatedIndVar);
182 
183  rewriter.eraseOp(forOp);
184  return matchSuccess();
185 }

191 PatternMatchResult
192 IfOpConversion::matchAndRewrite(loop::IfOp ifOp, ArrayRef<Value> operands,
193  ConversionPatternRewriter &rewriter) const {
194  // When lowering `loop::IfOp` we explicitly create a selection header block
195  // before the control flow diverges and a merge block where control flow
196  // subsequently converges.
197  loop::IfOpOperandAdaptor ifOperands(operands);
198  auto loc = ifOp.getLoc();
199 
200  // Create `spv.selection` operation, selection header block and merge block.
201  auto selectionControl = rewriter.getI32IntegerAttr(
202  static_cast<uint32_t>(spirv::SelectionControl::None));
203  auto selectionOp = rewriter.create<spirv::SelectionOp>(loc, selectionControl);
204  selectionOp.addMergeBlock();
205  auto *mergeBlock = selectionOp.getMergeBlock();
206 
207  OpBuilder::InsertionGuard guard(rewriter);
208  auto *selectionHeaderBlock = new Block();
209  selectionOp.body().getBlocks().push_front(selectionHeaderBlock);
210 
211  // Inline `then` region before the merge block and branch to it.
212  auto &thenRegion = ifOp.thenRegion();
213  auto *thenBlock = &thenRegion.front();
214  rewriter.setInsertionPointToEnd(&thenRegion.back());
215  rewriter.create<spirv::BranchOp>(loc, mergeBlock);
216  rewriter.inlineRegionBefore(thenRegion, mergeBlock);
217 
218  auto *elseBlock = mergeBlock;
219  // If `else` region is not empty, inline that region before the merge block
220  // and branch to it.
221  if (!ifOp.elseRegion().empty()) {
222  auto &elseRegion = ifOp.elseRegion();
223  elseBlock = &elseRegion.front();
224  rewriter.setInsertionPointToEnd(&elseRegion.back());
225  rewriter.create<spirv::BranchOp>(loc, mergeBlock);
226  rewriter.inlineRegionBefore(elseRegion, mergeBlock);
227  }
228 
229  // Create a `spv.BranchConditional` operation for selection header block.
230  rewriter.setInsertionPointToEnd(selectionHeaderBlock);
231  rewriter.create<spirv::BranchConditionalOp>(loc, ifOperands.condition(),
232  thenBlock, ArrayRef<Value>(),
233  elseBlock, ArrayRef<Value>());
234 
235  rewriter.eraseOp(ifOp);
236  return matchSuccess();
237 }

243 template <typename SourceOp, spirv::BuiltIn builtin>
244 PatternMatchResult LaunchConfigConversion<SourceOp, builtin>::matchAndRewrite(
245  SourceOp op, ArrayRef<Value> operands,
246  ConversionPatternRewriter &rewriter) const {
247  auto dimAttr =
248  op.getOperation()->template getAttrOfType<StringAttr>("dimension");
249  if (!dimAttr) {
250  return this->matchFailure();
251  }
252  int32_t index = 0;
253  if (dimAttr.getValue() == "x") {
254  index = 0;
255  } else if (dimAttr.getValue() == "y") {
256  index = 1;
257  } else if (dimAttr.getValue() == "z") {
258  index = 2;
259  } else {
260  return this->matchFailure();
261  }
262 
263  // SPIR-V invocation builtin variables are a vector of type <3xi32>
264  auto spirvBuiltin = spirv::getBuiltinVariableValue(op, builtin, rewriter);
265  rewriter.replaceOpWithNewOp<spirv::CompositeExtractOp>(
266  op, rewriter.getIntegerType(32), spirvBuiltin,
267  rewriter.getI32ArrayAttr({index}));
268  return this->matchSuccess();
269 }

275 // Legalizes a GPU function as an entry SPIR-V function.
276 static FuncOp
277 lowerAsEntryFunction(gpu::GPUFuncOp funcOp, SPIRVTypeConverter &typeConverter,
278  ConversionPatternRewriter &rewriter,
279  spirv::EntryPointABIAttr entryPointInfo,
280  ArrayRef<spirv::InterfaceVarABIAttr> argABIInfo) {
281  auto fnType = funcOp.getType();
282  if (fnType.getNumResults()) {
283  funcOp.emitError("SPIR-V lowering only supports entry functions"
284  "with no return values right now");
285  return nullptr;
286  }
287  if (fnType.getNumInputs() != argABIInfo.size()) {
288  funcOp.emitError(
289  "lowering as entry functions requires ABI info for all arguments");
290  return nullptr;
291  }
292  // Update the signature to valid SPIR-V types and add the ABI
293  // attributes. These will be "materialized" by using the
294  // LowerABIAttributesPass.
295  TypeConverter::SignatureConversion signatureConverter(fnType.getNumInputs());
296  {
297  for (auto argType : enumerate(funcOp.getType().getInputs())) {
298  auto convertedType = typeConverter.convertType(argType.value());
299  signatureConverter.addInputs(argType.index(), convertedType);
300  }
301  }
302  auto newFuncOp = rewriter.create<FuncOp>(
303  funcOp.getLoc(), funcOp.getName(),
304  rewriter.getFunctionType(signatureConverter.getConvertedTypes(),
305  llvm::None),
306  ArrayRef<NamedAttribute>());
307  for (const auto &namedAttr : funcOp.getAttrs()) {
308  if (namedAttr.first.is(impl::getTypeAttrName()) ||
309  namedAttr.first.is(SymbolTable::getSymbolAttrName()))
310  continue;
311  newFuncOp.setAttr(namedAttr.first, namedAttr.second);
312  }
313  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
314  newFuncOp.end());
315  rewriter.applySignatureConversion(&newFuncOp.getBody(), signatureConverter);
316  rewriter.eraseOp(funcOp);
317 
318  spirv::setABIAttrs(newFuncOp, entryPointInfo, argABIInfo);
319  return newFuncOp;
320 }

322 PatternMatchResult
323 KernelFnConversion::matchAndRewrite(gpu::GPUFuncOp funcOp,
324  ArrayRef<Value> operands,
325  ConversionPatternRewriter &rewriter) const {
326  if (!gpu::GPUDialect::isKernel(funcOp)) {
327  return matchFailure();
328  }
329 
330  SmallVector<spirv::InterfaceVarABIAttr, 4> argABI;
331  for (auto argNum : llvm::seq<unsigned>(0, funcOp.getNumArguments())) {
332  argABI.push_back(spirv::getInterfaceVarABIAttr(
333  0, argNum, spirv::StorageClass::StorageBuffer, rewriter.getContext()));
334  }
335 
336  auto context = rewriter.getContext();
337  auto entryPointAttr =
338  spirv::getEntryPointABIAttr(workGroupSizeAsInt32, context);
339  FuncOp newFuncOp = lowerAsEntryFunction(funcOp, typeConverter, rewriter,
340  entryPointAttr, argABI);
341  if (!newFuncOp) {
342  return matchFailure();
343  }
344  newFuncOp.removeAttr(Identifier::get(gpu::GPUDialect::getKernelFuncAttrName(),
345  rewriter.getContext()));
346  return matchSuccess();
347 }

353 PatternMatchResult GPUModuleConversion::matchAndRewrite(
354  gpu::GPUModuleOp moduleOp, ArrayRef<Value> operands,
355  ConversionPatternRewriter &rewriter) const {
356  // TODO : Generalize this to account for different extensions,
357  // capabilities, extended_instruction_sets, other addressing models
358  // and memory models.
359  auto spvModule = rewriter.create<spirv::ModuleOp>(
360  moduleOp.getLoc(), spirv::AddressingModel::Logical,
361  spirv::MemoryModel::GLSL450, spirv::Capability::Shader,
362  spirv::Extension::SPV_KHR_storage_buffer_storage_class);
363  // Move the region from the module op into the SPIR-V module.
364  Region &spvModuleRegion = spvModule.body();
365  rewriter.inlineRegionBefore(moduleOp.body(), spvModuleRegion,
366  spvModuleRegion.begin());
367  // The spv.module build method adds a block with a terminator. Remove that
368  // block. The terminator of the module op in the remaining block will be
369  // legalized later.
370  spvModuleRegion.back().erase();
371  rewriter.eraseOp(moduleOp);
372  return matchSuccess();
373 }
