86 /// A NestedBuilder is a scoping abstraction to create an idiomatic syntax
87 /// embedded in C++ that serves the purpose of building nested MLIR.
88 /// Nesting and compositionality is obtained by using the strict ordering that
89 /// exists between object construction and method invocation on said object (in
90 /// our case, the call to `operator()`).
91 /// This ordering allows implementing an abstraction that decouples definition
92 /// from declaration (in a PL sense) on placeholders of type ValueHandle and
93 /// BlockHandle.
94 class NestedBuilder {
95 protected:
96  NestedBuilder() = default;
97  NestedBuilder(const NestedBuilder &) = delete;
98  NestedBuilder(NestedBuilder &&other) : bodyScope(other.bodyScope) {
99  other.bodyScope = nullptr;
100  }
101 
102  NestedBuilder &operator=(const NestedBuilder &) = delete;
103  NestedBuilder &operator=(NestedBuilder &&other) {
104  std::swap(bodyScope, other.bodyScope);
105  return *this;
106  }
107 
108  /// Enter an mlir::Block and setup a ScopedContext to insert operations at
109  /// the end of it. Since we cannot use c++ language-level scoping to implement
110  /// scoping itself, we use enter/exit pairs of operations.
111  /// As a consequence we must allocate a new OpBuilder + ScopedContext and
112  /// let the escape.
113  /// Step back "prev" times from the end of the block to set up the insertion
114  /// point, which is useful for non-empty blocks.
115  void enter(mlir::Block *block, int prev = 0) {
116  bodyScope = new ScopedContext(
117  ScopedContext::getBuilder(),
118  OpBuilder::InsertPoint(block, std::prev(block->end(), prev)),
119  ScopedContext::getLocation());
120  bodyScope->nestedBuilder = this;
121  }
122 
123  /// Exit the current mlir::Block by explicitly deleting the dynamically
124  /// allocated OpBuilder and ScopedContext.
125  void exit() {
126  // Reclaim now to exit the scope.
127  bodyScope->nestedBuilder = nullptr;
128  delete bodyScope;
129  bodyScope = nullptr;
130  }
131 
132  /// Custom destructor does nothing because we already destroyed bodyScope
133  /// manually in `exit`. Insert an assertion to defensively guard against
134  /// improper usage of scoping.
135  ~NestedBuilder() {
136  assert(!bodyScope &&
137  "Illegal use of NestedBuilder; must have called exit()");
138  }
139 
140 private:
141  ScopedContext *bodyScope = nullptr;
142 };

281 /// ValueHandle implements a (potentially "delayed") typed Value abstraction.
282 /// ValueHandle should be captured by pointer but otherwise passed by Value
283 /// everywhere.
284 /// A ValueHandle can have 3 states:
285 /// 1. null state (empty type and empty value), in which case it does not hold
286 /// a value and must never hold a Value (now or in the future). This is
287 /// used for MLIR operations with zero returns as well as the result of
288 /// calling a NestedBuilder::operator(). In both cases the objective is to
289 /// have an object that can be inserted in an ArrayRef<ValueHandle> to
290 /// implement nesting;
291 /// 2. delayed state (empty value), in which case it represents an eagerly
292 /// typed "delayed" value that can be hold a Value in the future;
293 /// 3. constructed state,in which case it holds a Value.
294 ///
295 /// A ValueHandle is meant to capture a single Value and should be used for
296 /// operations that have a single result. For convenience of use, we also
297 /// include AffineForOp in this category although it does not return a value.
298 /// In the case of AffineForOp, the captured Value is the loop induction
299 /// variable.
300 class ValueHandle : public CapturableHandle {
301 public:
302  /// A ValueHandle in a null state can never be captured;
303  static ValueHandle null() { return ValueHandle(); }
304 
305  /// A ValueHandle that is constructed from a Type represents a typed "delayed"
306  /// Value. A delayed Value can only capture Values of the specified type.
307  /// Such a delayed value represents the declaration (in the PL sense) of a
308  /// placeholder for an mlir::Value that will be constructed and captured at
309  /// some later point in the program.
310  explicit ValueHandle(Type t) : t(t), v(nullptr) {}
311 
312  /// A ValueHandle that is constructed from an mlir::Value is an "eager"
313  /// Value. An eager Value represents both the declaration and the definition
314  /// (in the PL sense) of a placeholder for an mlir::Value that has already
315  /// been constructed in the past and that is captured "now" in the program.
316  explicit ValueHandle(Value v) : t(v.getType()), v(v) {}
317 
318  /// Builds a ConstantIndexOp of value `cst`. The constant is created at the
319  /// current insertion point.
320  /// This implicit constructor is provided to each build an eager Value for a
321  /// constant at the current insertion point in the IR. An implicit constructor
322  /// allows idiomatic expressions mixing ValueHandle and literals.
323  ValueHandle(index_type cst);
324 
325  /// ValueHandle is a value type, use the default copy constructor.
326  ValueHandle(const ValueHandle &other) = default;
327 
328  /// ValueHandle is a value type, the assignment operator typechecks before
329  /// assigning.
330  ValueHandle &operator=(const ValueHandle &other);
331 
332  /// Provide a swap operator.
333  void swap(ValueHandle &other) {
334  if (this == &other)
335  return;
336  std::swap(t, other.t);
337  std::swap(v, other.v);
338  }
339 
340  /// Implicit conversion useful for automatic conversion to Container<Value>.
341  operator Value() const { return getValue(); }
342  operator Type() const { return getType(); }
343  operator bool() const { return hasValue(); }
344 
345  /// Generic mlir::Op create. This is the key to being extensible to the whole
346  /// of MLIR without duplicating the type system or the op definitions.
347  template <typename Op, typename... Args>
348  static ValueHandle create(Args... args);
349 
350  /// Generic mlir::Op create. This is the key to being extensible to the whole
351  /// of MLIR without duplicating the type system or the op definitions.
352  /// When non-null, the optional pointer `folder` is used to call into the
353  /// `createAndFold` builder method. If `folder` is null, the regular `create`
354  /// method is called.
355  template <typename Op, typename... Args>
356  static ValueHandle create(OperationFolder *folder, Args... args);
357 
358  /// Special case to build composed AffineApply operations.
359  // TODO: createOrFold when available and move inside of the `create` method.
360  static ValueHandle createComposedAffineApply(AffineMap map,
361  ArrayRef<Value> operands);
362 
363  /// Generic create for a named operation producing a single value.
364  static ValueHandle create(StringRef name, ArrayRef<ValueHandle> operands,
365  ArrayRef<Type> resultTypes,
366  ArrayRef<NamedAttribute> attributes = {});
367 
368  bool hasValue() const { return v != nullptr; }
369  Value getValue() const {
370  assert(hasValue() && "Unexpected null value;");
371  return v;
372  }
373  bool hasType() const { return t != Type(); }
374  Type getType() const { return t; }
375 
376  Operation *getOperation() const {
377  if (!v)
378  return nullptr;
379  return v.getDefiningOp();
380  }
381 
382 protected:
383  ValueHandle() : t(), v(nullptr) {}
384 
385  Type t;
386  Value v;
387 };
