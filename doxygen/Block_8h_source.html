261  /// Walk the operations in this block in postorder, calling the callback for
262  /// each operation.
263  /// See Operation::walk for more details.
264  template <typename FnT, typename RetT = detail::walkResultType<FnT>>
265  RetT walk(FnT &&callback) {
266  return walk(begin(), end(), std::forward<FnT>(callback));
267  }
268 
269  /// Walk the operations in the specified [begin, end) range of this block in
270  /// postorder, calling the callback for each operation. This method is invoked
271  /// for void return callbacks.
272  /// See Operation::walk for more details.
273  template <typename FnT, typename RetT = detail::walkResultType<FnT>>
274  typename std::enable_if<std::is_same<RetT, void>::value, RetT>::type
275  walk(Block::iterator begin, Block::iterator end, FnT &&callback) {
276  for (auto &op : llvm::make_early_inc_range(llvm::make_range(begin, end)))
277  detail::walkOperations(&op, callback);
278  }
279 
280  /// Walk the operations in the specified [begin, end) range of this block in
281  /// postorder, calling the callback for each operation. This method is invoked
282  /// for interruptible callbacks.
283  /// See Operation::walk for more details.
284  template <typename FnT, typename RetT = detail::walkResultType<FnT>>
285  typename std::enable_if<std::is_same<RetT, WalkResult>::value, RetT>::type
286  walk(Block::iterator begin, Block::iterator end, FnT &&callback) {
287  for (auto &op : llvm::make_early_inc_range(llvm::make_range(begin, end)))
288  if (detail::walkOperations(&op, callback).wasInterrupted())
289  return WalkResult::interrupt();
290  return WalkResult::advance();
291  }
