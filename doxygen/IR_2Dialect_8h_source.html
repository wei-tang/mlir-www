33 /// entire group. For example, hooks into other systems for constant folding,
34 /// default named types for asm printing, etc.

65  /// Registered fallback constant fold hook for the dialect. Like the constant
66  /// fold hook of each operation, it attempts to constant fold the operation
67  /// with the specified constant operand values - the elements in "operands"
68  /// will correspond directly to the operands of the operation, but may be null
69  /// if non-constant. If constant folding is successful, this fills in the
70  /// `results` vector. If not, this returns failure and `results` is
71  /// unspecified.
72  DialectConstantFoldHook constantFoldHook =
73  [](Operation *op, ArrayRef<Attribute> operands,
74  SmallVectorImpl<Attribute> &results) { return failure(); };
75 
76  /// Registered hook to decode opaque constants associated with this
77  /// dialect. The hook function attempts to decode an opaque constant tensor
78  /// into a tensor with non-opaque content. If decoding is successful, this
79  /// method returns false and sets 'output' attribute. If not, it returns true
80  /// and leaves 'output' unspecified. The default hook fails to decode.
81  DialectConstantDecodeHook decodeHook =
82  [](const OpaqueElementsAttr input, ElementsAttr &output) { return true; };
83 
84  /// Registered hook to extract an element from an opaque constant associated
85  /// with this dialect. If element has been successfully extracted, this
86  /// method returns that element. If not, it returns an empty attribute.
87  /// The default hook fails to extract an element.
88  DialectExtractElementHook extractElementHook =
89  [](const OpaqueElementsAttr input, ArrayRef<uint64_t> index) {
90  return Attribute();
91  };

93  /// Registered hook to materialize a single constant operation from a given
94  /// attribute value with the desired resultant type. This method should use
95  /// the provided builder to create the operation without changing the
96  /// insertion position. The generated operation is expected to be constant
97  /// like, i.e. single result, zero operands, non side-effecting, etc. On
98  /// success, this hook should return the value generated to represent the
99  /// constant value. Otherwise, it should return null on failure.
100  virtual Operation *materializeConstant(OpBuilder &builder, Attribute value,
101  Type type, Location loc) {
102  return nullptr;

181  /// This method is used by derived classes to add their operations to the set.

183  template <typename... Args> void addOperations() {

187  // It would be nice to define this as variadic functions instead of a nested
188  // variadic type, but we can't do that: function template partial
189  // specialization is not allowed, and we can't define an overload set because
190  // we don't have any arguments of the types we are pushing around.
191  template <typename First, typename... Rest> class VariadicOperationAdder {
192  public:
193  static void addToSet(Dialect &dialect) {
194  dialect.addOperation(AbstractOperation::get<First>(dialect));
195  VariadicOperationAdder<Rest...>::addToSet(dialect);
196  }
197  };
