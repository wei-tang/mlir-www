25 /// ShapedTypeComponents that represents the components of a ShapedType.
26 /// The components consist of
27 /// - A ranked or unranked shape with the dimension specification match those
28 /// of ShapeType's getShape() (e.g., dynamic dimension represented using
29 /// ShapedType::kDynamicSize)
30 /// - A element type, may be unset (nullptr)
31 /// - A attribute, may be unset (nullptr)
32 /// Used by ShapedType type inferences.
33 class ShapedTypeComponents {
34  /// Internal storage type for shape.
35  using ShapeStorageT = SmallVector<int64_t, 3>;
36 
37 public:
38  /// Default construction is an unranked shape.
39  ShapedTypeComponents() : ranked(false), elementType(nullptr), attr(nullptr){};
40 
41  template <typename Arg, typename = typename std::enable_if_t<
42  std::is_constructible<ShapeStorageT, Arg>::value>>
43  ShapedTypeComponents(Arg &&arg, Type elementType = nullptr,
44  Attribute attr = nullptr)
45  : dims(std::forward<Arg>(arg)), ranked(true), elementType(elementType),
46  attr(attr) {}
47  ShapedTypeComponents(ArrayRef<int64_t> vec, Type elementType = nullptr,
48  Attribute attr = nullptr)
49  : dims(vec.begin(), vec.end()), ranked(true), elementType(elementType),
50  attr(attr) {}
51 
52  /// Return the dimensions of the shape.
53  /// Requires: shape is ranked.
54  ArrayRef<int64_t> getDims() const {
55  assert(ranked && "requires ranked shape");
56  return dims;
57  }
58 
59  /// Return whether the shape has a rank.
60  bool hasRank() const { return ranked; };
61 
62  /// Return the element type component.
63  Type getElementType() const { return elementType; };
64 
65  /// Return the raw attribute component.
66  Attribute getAttribute() const { return attr; };
67 
68 private:
69  ShapeStorageT dims;
70  bool ranked;
71  Type elementType;
72  Attribute attr;
73 };
