207 /// A trait used to provide symbol table functionalities to a region operation.
208 /// This operation must hold exactly 1 region. Once attached, all operations
209 /// that are directly within the region, i.e not including those within child
210 /// regions, that contain a 'SymbolTable::getSymbolAttrName()' StringAttr will
211 /// be verified to ensure that the names are uniqued. These operations must also
212 /// adhere to the constraints defined by the `Symbol` trait, even if they do not
213 /// inherit from it.
214 template <typename ConcreteType>
215 class SymbolTable : public TraitBase<ConcreteType, SymbolTable> {
216 public:
217  static LogicalResult verifyTrait(Operation *op) {
218  return impl::verifySymbolTable(op);
219  }
220 
221  /// Look up a symbol with the specified name, returning null if no such
222  /// name exists. Symbol names never include the @ on them. Note: This
223  /// performs a linear scan of held symbols.
224  Operation *lookupSymbol(StringRef name) {
225  return mlir::SymbolTable::lookupSymbolIn(this->getOperation(), name);
226  }
227  template <typename T> T lookupSymbol(StringRef name) {
228  return dyn_cast_or_null<T>(lookupSymbol(name));
229  }
230 };

232 /// A trait used to define a symbol that can be used on operations within a
233 /// symbol table. Operations using this trait must adhere to the following:
234 /// * Have a StringAttr attribute named 'SymbolTable::getSymbolAttrName()'.
235 template <typename ConcreteType>
236 class Symbol : public TraitBase<ConcreteType, Symbol> {
237 public:
238  using Visibility = mlir::SymbolTable::Visibility;
239 
240  static LogicalResult verifyTrait(Operation *op) {
241  return impl::verifySymbol(op);
242  }

244  /// Returns the name of this symbol.
245  StringRef getName() {
246  return this->getOperation()
247  ->template getAttrOfType<StringAttr>(
248  mlir::SymbolTable::getSymbolAttrName())
249  .getValue();
250  }
251 
252  /// Set the name of this symbol.
253  void setName(StringRef name) {
254  this->getOperation()->setAttr(
255  mlir::SymbolTable::getSymbolAttrName(),
256  StringAttr::get(name, this->getOperation()->getContext()));
257  }
258 
259  /// Returns the visibility of the current symbol.
260  Visibility getVisibility() {
261  return mlir::SymbolTable::getSymbolVisibility(this->getOperation());
262  }

264  /// Sets the visibility of the current symbol.
265  void setVisibility(Visibility vis) {
266  mlir::SymbolTable::setSymbolVisibility(this->getOperation(), vis);
267  }

272  Optional<::mlir::SymbolTable::UseRange> getSymbolUses(Operation *from) {
273  return ::mlir::SymbolTable::getSymbolUses(this->getOperation(), from);

281  }

286  LLVM_NODISCARD LogicalResult replaceAllSymbolUses(StringRef newSymbol,
287  Operation *from) {
