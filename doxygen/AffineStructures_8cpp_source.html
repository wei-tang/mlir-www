33 // An AffineExprFlattener extends a SimpleAffineExprFlattener by recording

39  // div's) to existing (dimensional and symbolic) ones. These are always

44  : SimpleAffineExprFlattener(nDims, nSymbols) {

65 // Flattens the expressions in map. Returns failure if 'expr' was unable to be
66 // flattened (i.e., semi-affine expressions not handled yet).
67 static LogicalResult
68 getFlattenedAffineExprs(ArrayRef<AffineExpr> exprs, unsigned numDims,
69  unsigned numSymbols,
70  std::vector<SmallVector<int64_t, 8>> *flattenedExprs,
71  FlatAffineConstraints *localVarCst) {
72  if (exprs.empty()) {
73  localVarCst->reset(numDims, numSymbols);
74  return success();
75  }
76 
77  AffineExprFlattener flattener(numDims, numSymbols, exprs[0].getContext());
78  // Use the same flattener to simplify each expression successively. This way
79  // local identifiers / expressions are shared.
80  for (auto expr : exprs) {
81  if (!expr.isPureAffine())
82  return failure();
83 
84  flattener.walkPostOrder(expr);
85  }
86 
87  assert(flattener.operandExprStack.size() == exprs.size());
88  flattenedExprs->clear();
89  flattenedExprs->assign(flattener.operandExprStack.begin(),
90  flattener.operandExprStack.end());
91 
92  if (localVarCst) {
93  localVarCst->clearAndCopyFrom(flattener.localVarCst);
94  }
95 
96  return success();
97 }

220 void AffineValueMap::difference(const AffineValueMap &a,
221  const AffineValueMap &b, AffineValueMap *res) {
222  assert(a.getNumResults() == b.getNumResults() && "invalid inputs");
223 
224  // Fully compose A's map + operands.
225  auto aMap = a.getAffineMap();
226  SmallVector<Value, 4> aOperands(a.getOperands().begin(),
227  a.getOperands().end());
228  fullyComposeAffineMapAndOperands(&aMap, &aOperands);
229 
230  // Use the affine apply normalizer to get B's map into A's coordinate space.
231  AffineApplyNormalizer normalizer(aMap, aOperands);
232  SmallVector<Value, 4> bOperands(b.getOperands().begin(),
233  b.getOperands().end());
234  auto bMap = b.getAffineMap();
235  normalizer.normalize(&bMap, &bOperands);
236 
237  assert(std::equal(bOperands.begin(), bOperands.end(),
238  normalizer.getOperands().begin()) &&
239  "operands are expected to be the same after normalization");
240 
241  // Construct the difference expressions.
242  SmallVector<AffineExpr, 4> diffExprs;
243  diffExprs.reserve(a.getNumResults());
244  for (unsigned i = 0, e = bMap.getNumResults(); i < e; ++i)
245  diffExprs.push_back(normalizer.getAffineMap().getResult(i) -
246  bMap.getResult(i));
247 
248  auto diffMap = AffineMap::get(normalizer.getNumDims(),
249  normalizer.getNumSymbols(), diffExprs);
250  canonicalizeMapAndOperands(&diffMap, &bOperands);
251  diffMap = simplifyAffineMap(diffMap);
252  res->reset(diffMap, bOperands);
253 }

333 // Construct from an IntegerSet.
334 FlatAffineConstraints::FlatAffineConstraints(IntegerSet set)
335  : numReservedCols(set.getNumInputs() + 1),
336  numIds(set.getNumDims() + set.getNumSymbols()), numDims(set.getNumDims()),
337  numSymbols(set.getNumSymbols()) {
338  equalities.reserve(set.getNumEqualities() * numReservedCols);
339  inequalities.reserve(set.getNumInequalities() * numReservedCols);
340  ids.resize(numIds, None);
341 
342  // Flatten expressions and add them to the constraint system.
343  std::vector<SmallVector<int64_t, 8>> flatExprs;
344  FlatAffineConstraints localVarCst;
345  if (failed(getFlattenedAffineExprs(set, &flatExprs, &localVarCst))) {
346  assert(false && "flattening unimplemented for semi-affine integer sets");
347  return;
348  }
349  assert(flatExprs.size() == set.getNumConstraints());
350  for (unsigned l = 0, e = localVarCst.getNumLocalIds(); l < e; l++) {
351  addLocalId(getNumLocalIds());
352  }
353 
354  for (unsigned i = 0, e = flatExprs.size(); i < e; ++i) {
355  const auto &flatExpr = flatExprs[i];
356  assert(flatExpr.size() == getNumCols());
357  if (set.getEqFlags()[i]) {
358  addEquality(flatExpr);
359  } else {
360  addInequality(flatExpr);
361  }
362  }
363  // Add the other constraints involving local id's from flattening.
364  append(localVarCst);
365 }

430 /// Adds a dimensional identifier. The added column is initialized to
431 /// zero.
432 void FlatAffineConstraints::addId(IdKind kind, unsigned pos, Value id) {
433  if (kind == IdKind::Dimension) {
434  assert(pos <= getNumDimIds());
435  } else if (kind == IdKind::Symbol) {
436  assert(pos <= getNumSymbolIds());
437  } else {
438  assert(pos <= getNumLocalIds());
439  }
440 
441  unsigned oldNumReservedCols = numReservedCols;
442 
443  // Check if a resize is necessary.
444  if (getNumCols() + 1 > numReservedCols) {
445  equalities.resize(getNumEqualities() * (getNumCols() + 1));
446  inequalities.resize(getNumInequalities() * (getNumCols() + 1));
447  numReservedCols++;
448  }
449 
450  int absolutePos;
451 
452  if (kind == IdKind::Dimension) {
453  absolutePos = pos;
454  numDims++;
455  } else if (kind == IdKind::Symbol) {
456  absolutePos = pos + getNumDimIds();
457  numSymbols++;
458  } else {
459  absolutePos = pos + getNumDimIds() + getNumSymbolIds();
460  }
461  numIds++;
462 
463  // Note that getNumCols() now will already return the new size, which will be
464  // at least one.
465  int numInequalities = static_cast<int>(getNumInequalities());
466  int numEqualities = static_cast<int>(getNumEqualities());
467  int numCols = static_cast<int>(getNumCols());
468  for (int r = numInequalities - 1; r >= 0; r--) {
469  for (int c = numCols - 2; c >= 0; c--) {
470  if (c < absolutePos)
471  atIneq(r, c) = inequalities[r * oldNumReservedCols + c];
472  else
473  atIneq(r, c + 1) = inequalities[r * oldNumReservedCols + c];
474  }
475  atIneq(r, absolutePos) = 0;
476  }
477 
478  for (int r = numEqualities - 1; r >= 0; r--) {
479  for (int c = numCols - 2; c >= 0; c--) {
480  // All values in column absolutePositions < absolutePos have the same
481  // coordinates in the 2-d view of the coefficient buffer.
482  if (c < absolutePos)
483  atEq(r, c) = equalities[r * oldNumReservedCols + c];
484  else
485  // Those at absolutePosition >= absolutePos, get a shifted
486  // absolutePosition.
487  atEq(r, c + 1) = equalities[r * oldNumReservedCols + c];
488  }
489  // Initialize added dimension to zero.
490  atEq(r, absolutePos) = 0;
491  }
492 
493  // If an 'id' is provided, insert it; otherwise use None.
494  if (id) {
495  ids.insert(ids.begin() + absolutePos, id);
496  } else {
497  ids.insert(ids.begin() + absolutePos, None);
498  }
499  assert(ids.size() == getNumIds());
500 }

530 static void swapId(FlatAffineConstraints *A, unsigned posA, unsigned posB) {

546 /// Merge and align the identifiers of A and B starting at 'offset', so that
547 /// both constraint systems get the union of the contained identifiers that is
548 /// dimension-wise and symbol-wise unique; both constraint systems are updated
549 /// so that they have the union of all identifiers, with A's original
550 /// identifiers appearing first followed by any of B's identifiers that didn't
551 /// appear in A. Local identifiers of each system are by design separate/local
552 /// and are placed one after other (A's followed by B's).
553 // Eg: Input: A has ((%i %j) [%M %N]) and B has (%k, %j) [%P, %N, %M])
554 // Output: both A, B have (%i, %j, %k) [%M, %N, %P]
555 //
556 static void mergeAndAlignIds(unsigned offset, FlatAffineConstraints *A,
557  FlatAffineConstraints *B) {
558  assert(offset <= A->getNumDimIds() && offset <= B->getNumDimIds());
559  // A merge/align isn't meaningful if a cst's ids aren't distinct.
560  assert(areIdsUnique(*A) && "A's id values aren't unique");
561  assert(areIdsUnique(*B) && "B's id values aren't unique");
562 
563  assert(std::all_of(A->getIds().begin() + offset,
564  A->getIds().begin() + A->getNumDimAndSymbolIds(),
565  [](Optional<Value> id) { return id.hasValue(); }));
566 
567  assert(std::all_of(B->getIds().begin() + offset,
568  B->getIds().begin() + B->getNumDimAndSymbolIds(),
569  [](Optional<Value> id) { return id.hasValue(); }));
570 
571  // Place local id's of A after local id's of B.
572  for (unsigned l = 0, e = A->getNumLocalIds(); l < e; l++) {
573  B->addLocalId(0);
574  }
575  for (unsigned t = 0, e = B->getNumLocalIds() - A->getNumLocalIds(); t < e;
576  t++) {
577  A->addLocalId(A->getNumLocalIds());
578  }
579 
580  SmallVector<Value, 4> aDimValues, aSymValues;
581  A->getIdValues(offset, A->getNumDimIds(), &aDimValues);
582  A->getIdValues(A->getNumDimIds(), A->getNumDimAndSymbolIds(), &aSymValues);
583  {
584  // Merge dims from A into B.
585  unsigned d = offset;
586  for (auto aDimValue : aDimValues) {

641 LogicalResult FlatAffineConstraints::composeMap(const AffineValueMap *vMap) {
642  std::vector<SmallVector<int64_t, 8>> flatExprs;
643  FlatAffineConstraints localCst;
644  if (failed(getFlattenedAffineExprs(vMap->getAffineMap(), &flatExprs,
645  &localCst))) {

648  return failure();

651 
652  // Add localCst information.
653  if (localCst.getNumLocalIds() > 0) {
654  localCst.setIdValues(0, /*end=*/localCst.getNumDimAndSymbolIds(),
655  /*values=*/vMap->getOperands());
656  // Align localCst and this.
657  mergeAndAlignIds(/*offset=*/0, &localCst, this);
658  // Finally, append localCst to this constraint set.
659  append(localCst);
660  }
661 
662  // Add dimensions corresponding to the map's results.
663  for (unsigned t = 0, e = vMap->getNumResults(); t < e; t++) {
664  // TODO: Consider using a batched version to add a range of IDs.
665  addDimId(0);
666  }

669  // the other identifiers.
670  // For eg: if the expression is 16*i0 + i1, and this is the r^th
671  // iteration/result of the value map, we are adding the equality:
672  // d_r - 16*i0 - i1 = 0. Hence, when flattening say (i0 + 1, i0 + 8*i2), we
673  // add two equalities overall: d_0 - i0 - 1 == 0, d1 - i0 - 8*i2 == 0.
674  for (unsigned r = 0, e = flatExprs.size(); r < e; r++) {
675  const auto &flatExpr = flatExprs[r];
676  assert(flatExpr.size() >= vMap->getNumOperands() + 1);
677 
678  // eqToAdd is the equality corresponding to the flattened affine expression.
679  SmallVector<int64_t, 8> eqToAdd(getNumCols(), 0);
680  // Set the coefficient for this result to one.
681  eqToAdd[r] = 1;
682 
683  // Dims and symbols.
684  for (unsigned i = 0, e = vMap->getNumOperands(); i < e; i++) {
685  unsigned loc;
686  bool ret = findId(vMap->getOperand(i), &loc);
687  assert(ret && "value map's id can't be found");
688  (void)ret;
689  // Negate 'eq[r]' since the newly added dimension will be set to this one.
690  eqToAdd[loc] = -flatExpr[i];
691  }
692  // Local vars common to eq and localCst are at the beginning.
693  unsigned j = getNumDimIds() + getNumSymbolIds();
694  unsigned end = flatExpr.size() - 1;
695  for (unsigned i = vMap->getNumOperands(); i < end; i++, j++) {
696  eqToAdd[j] = -flatExpr[i];
697  }
698 
699  // Constant term.
700  eqToAdd[getNumCols() - 1] = -flatExpr[flatExpr.size() - 1];
701 
702  // Add the equality connecting the result of the map to this constraint set.
703  addEquality(eqToAdd);
704  }
705 
706  return success();
707 }

833 LogicalResult FlatAffineConstraints::addAffineForOpDomain(AffineForOp forOp) {
834  unsigned pos;
835  // Pre-condition for this method.
836  if (!findId(forOp.getInductionVar(), &pos)) {
837  assert(false && "Value not found");
838  return failure();
839  }
840 
841  int64_t step = forOp.getStep();
842  if (step != 1) {
843  if (!forOp.hasConstantLowerBound())
844  forOp.emitWarning("domain conservatively approximated");
845  else {
846  // Add constraints for the stride.
847  // (iv - lb) % step = 0 can be written as:
848  // (iv - lb) - step * q = 0 where q = (iv - lb) / step.
849  // Add local variable 'q' and add the above equality.
850  // The first constraint is q = (iv - lb) floordiv step
851  SmallVector<int64_t, 8> dividend(getNumCols(), 0);
852  int64_t lb = forOp.getConstantLowerBound();
853  dividend[pos] = 1;
854  dividend.back() -= lb;
855  addLocalFloorDiv(dividend, step);
856  // Second constraint: (iv - lb) - step * q = 0.
857  SmallVector<int64_t, 8> eq(getNumCols(), 0);
858  eq[pos] = 1;
859  eq.back() -= lb;
860  // For the local var just added above.
861  eq[getNumCols() - 2] = -step;
862  addEquality(eq);
863  }
864  }
865 
866  if (forOp.hasConstantLowerBound()) {
867  addConstantLowerBound(pos, forOp.getConstantLowerBound());
868  } else {
869  // Non-constant lower bound case.
870  SmallVector<Value, 4> lbOperands(forOp.getLowerBoundOperands().begin(),
871  forOp.getLowerBoundOperands().end());
872  if (failed(addLowerOrUpperBound(pos, forOp.getLowerBoundMap(), lbOperands,
873  /*eq=*/false, /*lower=*/true)))
874  return failure();
875  }
876 
877  if (forOp.hasConstantUpperBound()) {
878  addConstantUpperBound(pos, forOp.getConstantUpperBound() - 1);
879  return success();
880  }
881  // Non-constant upper bound case.
882  SmallVector<Value, 4> ubOperands(forOp.getUpperBoundOperands().begin(),
883  forOp.getUpperBoundOperands().end());
884  return addLowerOrUpperBound(pos, forOp.getUpperBoundMap(), ubOperands,
885  /*eq=*/false, /*lower=*/false);
886 }

910 // GCD in equality or inequality constraints as specified by 'isEq'.

913  unsigned rowIdx) {

955 /// Checks all rows of equality/inequality constraints for trivial
956 /// contradictions (for example: 1 == 0, 0 >= 1), which may have surfaced
957 /// after elimination. Returns 'true' if an invalid constraint is found;
958 /// 'false' otherwise.
959 bool FlatAffineConstraints::hasInvalidConstraint() const {
960  assert(hasConsistentState());
961  auto check = [&](bool isEq) -> bool {
962  unsigned numCols = getNumCols();
963  unsigned numRows = isEq ? getNumEqualities() : getNumInequalities();
964  for (unsigned i = 0, e = numRows; i < e; ++i) {
965  unsigned j;
966  for (j = 0; j < numCols - 1; ++j) {
967  int64_t v = isEq ? atEq(i, j) : atIneq(i, j);
968  // Skip rows with non-zero variable coefficients.
969  if (v != 0)
970  break;
971  }
972  if (j < numCols - 1) {
973  continue;
974  }
975  // Check validity of constant term at 'numCols - 1' w.r.t 'isEq'.
976  // Example invalid constraints include: '1 == 0' or '-1 >= 0'
977  int64_t v = isEq ? atEq(i, numCols - 1) : atIneq(i, numCols - 1);
978  if ((isEq && v != 0) || (!isEq && v < 0)) {
979  return true;
980  }
981  }
982  return false;
983  };
984  if (check(/*isEq=*/true))
985  return true;
986  return check(/*isEq=*/false);
987 }

989 // Eliminate identifier from constraint at 'rowIdx' based on coefficient at
990 // pivotRow, pivotCol. Columns in range [elimColStart, pivotCol) will not be
991 // updated as they have already been eliminated.
992 static void eliminateFromConstraint(FlatAffineConstraints *constraints,
993  unsigned rowIdx, unsigned pivotRow,
994  unsigned pivotCol, unsigned elimColStart,
995  bool isEq) {
996  // Skip if equality 'rowIdx' if same as 'pivotRow'.
997  if (isEq && rowIdx == pivotRow)
998  return;
999  auto at = [&](unsigned i, unsigned j) -> int64_t {
1000  return isEq ? constraints->atEq(i, j) : constraints->atIneq(i, j);
1001  };
1002  int64_t leadCoeff = at(rowIdx, pivotCol);
1003  // Skip if leading coefficient at 'rowIdx' is already zero.
1004  if (leadCoeff == 0)
1005  return;
1006  int64_t pivotCoeff = constraints->atEq(pivotRow, pivotCol);
1007  int64_t sign = (leadCoeff * pivotCoeff > 0) ? -1 : 1;
1008  int64_t lcm = mlir::lcm(pivotCoeff, leadCoeff);
1009  int64_t pivotMultiplier = sign * (lcm / std::abs(pivotCoeff));
1010  int64_t rowMultiplier = lcm / std::abs(leadCoeff);
1011 
1012  unsigned numCols = constraints->getNumCols();
1013  for (unsigned j = 0; j < numCols; ++j) {
1014  // Skip updating column 'j' if it was just eliminated.
1015  if (j >= elimColStart && j < pivotCol)
1016  continue;
1017  int64_t v = pivotMultiplier * constraints->atEq(pivotRow, j) +
1018  rowMultiplier * at(rowIdx, j);
1019  isEq ? constraints->atEq(rowIdx, j) = v
1020  : constraints->atIneq(rowIdx, j) = v;
1021  }
1022 }

1092 /// Returns the position of the identifier that has the minimum <number of lower
1093 /// bounds> times <number of upper bounds> from the specified range of
1094 /// identifiers [start, end). It is often best to eliminate in the increasing
1095 /// order of these counts when doing Fourier-Motzkin elimination since FM adds

1143  // hasInvalidConstraint() are linear time and single sweep on the constraint
1144  // buffer, this appears reasonable - but can optimize in the future.
1145  if (tmpCst.hasInvalidConstraint() || tmpCst.isEmptyByGCDTest())
1146  return true;
1147  }

1150  for (unsigned i = 0, e = tmpCst.getNumIds(); i < e; i++) {
1151  tmpCst.FourierMotzkinEliminate(

1154  // this check exists as a safeguard against improperly constructed
1155  // constraint systems or artificially created arbitrarily complex systems

1158  if (tmpCst.getNumConstraints() >= kExplosionFactor * getNumIds()) {

1165  if (tmpCst.hasInvalidConstraint())

1171 // Runs the GCD test on all equality constraints. Returns 'true' if this test
1172 // fails on any equality. Returns 'false' otherwise.
1173 // This test can be used to disprove the existence of a solution. If it returns
1174 // true, no integer solution to the equality constraints can exist.
1175 //
1176 // GCD test definition:
1177 //
1178 // The equality constraint:
1179 //
1180 // c_1*x_1 + c_2*x_2 + ... + c_n*x_n = c_0
1181 //
1182 // has an integer solution iff:
1183 //
1184 // GCD of c_1, c_2, ..., c_n divides c_0.
1185 //
1186 bool FlatAffineConstraints::isEmptyByGCDTest() const {
1187  assert(hasConsistentState());
1188  unsigned numCols = getNumCols();
1189  for (unsigned i = 0, e = getNumEqualities(); i < e; ++i) {
1190  uint64_t gcd = std::abs(atEq(i, 0));
1191  for (unsigned j = 1; j < numCols - 1; ++j) {
1192  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(atEq(i, j)));
1193  }
1194  int64_t v = std::abs(atEq(i, numCols - 1));
1195  if (gcd > 0 && (v % gcd != 0)) {
1196  return true;
1197  }
1198  }
1199  return false;
1200 }

1202 /// Tightens inequalities given that we are dealing with integer spaces. This is
1203 /// analogous to the GCD test but applied to inequalities. The constant term can
1204 /// be reduced to the preceding multiple of the GCD of the coefficients, i.e.,
1205 /// 64*i - 100 >= 0 => 64*i - 128 >= 0 (since 'i' is an integer). This is a
1206 /// fast method - linear in the number of coefficients.
1207 // Example on how this affects practical cases: consider the scenario:
1208 // 64*i >= 100, j = 64*i; without a tightening, elimination of i would yield
1209 // j >= 100 instead of the tighter (exact) j >= 128.
1210 void FlatAffineConstraints::GCDTightenInequalities() {
1211  unsigned numCols = getNumCols();
1212  for (unsigned i = 0, e = getNumInequalities(); i < e; ++i) {
1213  uint64_t gcd = std::abs(atIneq(i, 0));
1214  for (unsigned j = 1; j < numCols - 1; ++j) {
1215  gcd = llvm::GreatestCommonDivisor64(gcd, std::abs(atIneq(i, j)));
1216  }
1217  if (gcd > 0 && gcd != 1) {
1218  int64_t gcdI = static_cast<int64_t>(gcd);
1219  // Tighten the constant term and normalize the constraint by the GCD.
1220  atIneq(i, numCols - 1) = mlir::floorDiv(atIneq(i, numCols - 1), gcdI);
1221  for (unsigned j = 0, e = numCols - 1; j < e; ++j)
1222  atIneq(i, j) /= gcdI;
1223  }
1224  }
1225 }

1279 // Detect the identifier at 'pos' (say id_r) as modulo of another identifier
1280 // (say id_n) w.r.t a constant. When this happens, another identifier (say id_q)
1281 // could be detected as the floordiv of n. For eg:
1282 // id_n - 4*id_q - id_r = 0, 0 <= id_r <= 3 <=>
1283 // id_r = id_n mod 4, id_q = id_n floordiv 4.
1284 // lbConst and ubConst are the constant lower and upper bounds for 'pos' -
1285 // pre-detected at the caller.
1286 static bool detectAsMod(const FlatAffineConstraints &cst, unsigned pos,
1287  int64_t lbConst, int64_t ubConst,
1288  SmallVectorImpl<AffineExpr> *memo) {
1289  assert(pos < cst.getNumIds() && "invalid position");
1290 
1291  // Check if 0 <= id_r <= divisor - 1 and if id_r is equal to
1292  // id_n - divisor * id_q. If these are true, then id_n becomes the dividend
1293  // and id_q the quotient when dividing id_n by the divisor.
1294 
1295  if (lbConst != 0 || ubConst < 1)
1296  return false;
1297 
1298  int64_t divisor = ubConst + 1;
1299 
1300  // Now check for: id_r = id_n - divisor * id_q. As an example, we
1301  // are looking r = d - 4q, i.e., either r - d + 4q = 0 or -r + d - 4q = 0.
1302  unsigned seenQuotient = 0, seenDividend = 0;
1303  int quotientPos = -1, dividendPos = -1;
1304  for (unsigned r = 0, e = cst.getNumEqualities(); r < e; r++) {
1305  // id_n should have coeff 1 or -1.
1306  if (std::abs(cst.atEq(r, pos)) != 1)
1307  continue;
1308  // constant term should be 0.
1309  if (cst.atEq(r, cst.getNumCols() - 1) != 0)
1310  continue;
1311  unsigned c, f;
1312  int quotientSign = 1, dividendSign = 1;
1313  for (c = 0, f = cst.getNumDimAndSymbolIds(); c < f; c++) {
1314  if (c == pos)
1315  continue;
1316  // The coefficient of the quotient should be +/-divisor.
1317  // TODO(bondhugula): could be extended to detect an affine function for
1318  // the quotient (i.e., the coeff could be a non-zero multiple of divisor).
1319  int64_t v = cst.atEq(r, c) * cst.atEq(r, pos);
1320  if (v == divisor || v == -divisor) {
1321  seenQuotient++;
1322  quotientPos = c;
1323  quotientSign = v > 0 ? 1 : -1;
1324  }
1325  // The coefficient of the dividend should be +/-1.
1326  // TODO(bondhugula): could be extended to detect an affine function of
1327  // the other identifiers as the dividend.
1328  else if (v == -1 || v == 1) {
1329  seenDividend++;
1330  dividendPos = c;
1331  dividendSign = v < 0 ? 1 : -1;
1332  } else if (cst.atEq(r, c) != 0) {
1333  // Cannot be inferred as a mod since the constraint has a coefficient
1334  // for an identifier that's neither a unit nor the divisor (see TODOs
1335  // above).
1336  break;
1337  }
1338  }
1339  if (c < f)
1340  // Cannot be inferred as a mod since the constraint has a coefficient for
1341  // an identifier that's neither a unit nor the divisor (see TODOs above).
1342  continue;
1343 
1344  // We are looking for exactly one identifier as the dividend.
1345  if (seenDividend == 1 && seenQuotient >= 1) {
1346  if (!(*memo)[dividendPos])
1347  return false;
1348  // Successfully detected a mod.
1349  (*memo)[pos] = (*memo)[dividendPos] % divisor * dividendSign;
1350  auto ub = cst.getConstantUpperBound(dividendPos);
1351  if (ub.hasValue() && ub.getValue() < divisor)
1352  // The mod can be optimized away.
1353  (*memo)[pos] = (*memo)[dividendPos] * dividendSign;
1354  else
1355  (*memo)[pos] = (*memo)[dividendPos] % divisor * dividendSign;
1356 
1357  if (seenQuotient == 1 && !(*memo)[quotientPos])
1358  // Successfully detected a floordiv as well.
1359  (*memo)[quotientPos] =
1360  (*memo)[dividendPos].floorDiv(divisor) * quotientSign;
1361  return true;
1362  }
1363  }
1364  return false;
1365 }

1367 // Gather lower and upper bounds for the pos^th identifier.
1368 static void getLowerAndUpperBoundIndices(const FlatAffineConstraints &cst,
1369  unsigned pos,
1370  SmallVectorImpl<unsigned> *lbIndices,
1371  SmallVectorImpl<unsigned> *ubIndices) {
1372  assert(pos < cst.getNumIds() && "invalid position");
1373 
1374  // Gather all lower bounds and upper bounds of the variable. Since the
1375  // canonical form c_1*x_1 + c_2*x_2 + ... + c_0 >= 0, a constraint is a lower
1376  // bound for x_i if c_i >= 1, and an upper bound if c_i <= -1.
1377  for (unsigned r = 0, e = cst.getNumInequalities(); r < e; r++) {
1378  if (cst.atIneq(r, pos) >= 1) {
1379  // Lower bound.
1380  lbIndices->push_back(r);
1381  } else if (cst.atIneq(r, pos) <= -1) {
1382  // Upper bound.
1383  ubIndices->push_back(r);
1384  }
1385  }
1386 }
1387 
1388 // Check if the pos^th identifier can be expressed as a floordiv of an affine
1389 // function of other identifiers (where the divisor is a positive constant).
1390 // For eg: 4q <= i + j <= 4q + 3 <=> q = (i + j) floordiv 4.
1391 static bool detectAsFloorDiv(const FlatAffineConstraints &cst, unsigned pos,
1392  SmallVectorImpl<AffineExpr> *memo,
1393  MLIRContext *context) {
1394  assert(pos < cst.getNumIds() && "invalid position");
1395 
1396  SmallVector<unsigned, 4> lbIndices, ubIndices;
1397  getLowerAndUpperBoundIndices(cst, pos, &lbIndices, &ubIndices);
1398 
1399  // Check if any lower bound, upper bound pair is of the form:
1400  // divisor * id >= expr - (divisor - 1) <-- Lower bound for 'id'
1401  // divisor * id <= expr <-- Upper bound for 'id'
1402  // Then, 'id' is equivalent to 'expr floordiv divisor'. (where divisor > 1).
1403  //
1404  // For example, if -32*k + 16*i + j >= 0
1405  // 32*k - 16*i - j + 31 >= 0 <=>
1406  // k = ( 16*i + j ) floordiv 32
1407  unsigned seenDividends = 0;
1408  for (auto ubPos : ubIndices) {
1409  for (auto lbPos : lbIndices) {
1410  // Check if lower bound's constant term is 'divisor - 1'. The 'divisor'
1411  // here is cst.atIneq(lbPos, pos) and we already know that it's positive
1412  // (since cst.Ineq(lbPos, ...) is a lower bound expression for 'pos'.
1413  if (cst.atIneq(lbPos, cst.getNumCols() - 1) != cst.atIneq(lbPos, pos) - 1)
1414  continue;
1415  // Check if upper bound's constant term is 0.
1416  if (cst.atIneq(ubPos, cst.getNumCols() - 1) != 0)
1417  continue;
1418  // For the remaining part, check if the lower bound expr's coeff's are
1419  // negations of corresponding upper bound ones'.
1420  unsigned c, f;
1421  for (c = 0, f = cst.getNumCols() - 1; c < f; c++) {
1422  if (cst.atIneq(lbPos, c) != -cst.atIneq(ubPos, c))
1423  break;
1424  if (c != pos && cst.atIneq(lbPos, c) != 0)
1425  seenDividends++;
1426  }
1427  // Lb coeff's aren't negative of ub coeff's (for the non constant term
1428  // part).
1429  if (c < f)
1430  continue;
1431  if (seenDividends >= 1) {
1432  // The divisor is the constant term of the lower bound expression.
1433  // We already know that cst.atIneq(lbPos, pos) > 0.
1434  int64_t divisor = cst.atIneq(lbPos, pos);
1435  // Construct the dividend expression.
1436  auto dividendExpr = getAffineConstantExpr(0, context);
1437  unsigned c, f;
1438  for (c = 0, f = cst.getNumCols() - 1; c < f; c++) {
1439  if (c == pos)
1440  continue;
1441  int64_t ubVal = cst.atIneq(ubPos, c);
1442  if (ubVal == 0)
1443  continue;
1444  if (!(*memo)[c])
1445  break;
1446  dividendExpr = dividendExpr + ubVal * (*memo)[c];
1447  }
1448  // Expression can't be constructed as it depends on a yet unknown
1449  // identifier.
1450  // TODO(mlir-team): Visit/compute the identifiers in an order so that
1451  // this doesn't happen. More complex but much more efficient.
1452  if (c < f)
1453  continue;
1454  // Successfully detected the floordiv.
1455  (*memo)[pos] = dividendExpr.floorDiv(divisor);
1456  return true;
1457  }
1458  }
1459  }
1460  return false;
1461 }

1479 // to check if a constraint is redundant.
1480 void FlatAffineConstraints::removeRedundantInequalities() {
1481  SmallVector<bool, 32> redun(getNumInequalities(), false);
1482  // To check if an inequality is redundant, we replace the inequality by its
1483  // complement (for eg., i - 1 >= 0 by i <= 0), and check if the resulting
1484  // system is empty. If it is, the inequality is redundant.
1485  FlatAffineConstraints tmpCst(*this);
1486  for (unsigned r = 0, e = getNumInequalities(); r < e; r++) {
1487  // Change the inequality to its complement.
1488  negateInequality(&tmpCst, r);
1489  tmpCst.atIneq(r, tmpCst.getNumCols() - 1)--;
1490  if (tmpCst.isEmpty()) {
1491  redun[r] = true;
1492  // Zero fill the redundant inequality.
1493  fillInequality(this, r, /*val=*/0);
1494  fillInequality(&tmpCst, r, /*val=*/0);
1495  } else {
1496  // Reverse the change (to avoid recreating tmpCst each time).
1497  tmpCst.atIneq(r, tmpCst.getNumCols() - 1)++;
1498  negateInequality(&tmpCst, r);
1499  }
1500  }

1509  };

1511  for (unsigned r = 0, e = getNumInequalities(); r < e; r++) {

1582  SmallVectorImpl<AffineMap> *ubMaps) {

1586  normalizeConstraintsByGCD();

1665  expr = expr + atEq(idx, getNumIds());
1666  int64_t vPos = atEq(idx, pos);
1667  assert(vPos != 0 && "expected non-zero here");
1668  if (vPos > 0)
1669  expr = (-expr).floorDiv(vPos);
1670  else
1671  // vPos < 0.
1672  expr = expr.floorDiv(-vPos);
1673  // Successfully constructed expression.
1674  memo[pos] = expr;
1675  changed = true;
1676  }
1677  // This loop is guaranteed to reach a fixed point - since once an
1678  // identifier's explicit form is computed (in memo[pos]), it's not updated
1679  // again.
1680  } while (changed);
1681 
1682  // Set the lower and upper bound maps for all the identifiers that were
1683  // computed as affine expressions of the rest as the "detected expr" and
1684  // "detected expr + 1" respectively; set the undetected ones to null.
1685  Optional<FlatAffineConstraints> tmpClone;
1686  for (unsigned pos = 0; pos < num; pos++) {
1687  unsigned numMapDims = getNumDimIds() - num;
1688  unsigned numMapSymbols = getNumSymbolIds();
1689  AffineExpr expr = memo[pos + offset];
1690  if (expr)
1691  expr = simplifyAffineExpr(expr, numMapDims, numMapSymbols);
1692 
1693  AffineMap &lbMap = (*lbMaps)[pos];
1694  AffineMap &ubMap = (*ubMaps)[pos];
1695 
1696  if (expr) {
1697  lbMap = AffineMap::get(numMapDims, numMapSymbols, expr);
1698  ubMap = AffineMap::get(numMapDims, numMapSymbols, expr + 1);
1699  } else {
1700  // TODO(bondhugula): Whenever there are local identifiers in the
1701  // dependence constraints, we'll conservatively over-approximate, since we
1702  // don't always explicitly compute them above (in the while loop).
1703  if (getNumLocalIds() == 0) {
1704  // Work on a copy so that we don't update this constraint system.
1705  if (!tmpClone) {
1706  tmpClone.emplace(FlatAffineConstraints(*this));
1707  // Removing redundant inequalities is necessary so that we don't get
1708  // redundant loop bounds.
1709  tmpClone->removeRedundantInequalities();
1710  }
1711  std::tie(lbMap, ubMap) = tmpClone->getLowerAndUpperBound(
1712  pos, offset, num, getNumDimIds(), {}, context);
1713  }
1714 
1715  // If the above fails, we'll just use the constant lower bound and the
1716  // constant upper bound (if they exist) as the slice bounds.
1717  // TODO(b/126426796): being conservative for the moment in cases that
1718  // lead to multiple bounds - until getConstDifference in LoopFusion.cpp is
1719  // fixed (b/126426796).
1720  if (!lbMap || lbMap.getNumResults() > 1) {
1721  LLVM_DEBUG(llvm::dbgs()
1722  << "WARNING: Potentially over-approximating slice lb\n");
1723  auto lbConst = getConstantLowerBound(pos + offset);
1724  if (lbConst.hasValue()) {
1725  lbMap = AffineMap::get(
1726  numMapDims, numMapSymbols,
1727  getAffineConstantExpr(lbConst.getValue(), context));
1728  }
1729  }
1730  if (!ubMap || ubMap.getNumResults() > 1) {
1731  LLVM_DEBUG(llvm::dbgs()
1732  << "WARNING: Potentially over-approximating slice ub\n");
1733  auto ubConst = getConstantUpperBound(pos + offset);
1734  if (ubConst.hasValue()) {
1735  (ubMap) = AffineMap::get(
1736  numMapDims, numMapSymbols,
1737  getAffineConstantExpr(ubConst.getValue() + 1, context));
1738  }
1739  }
1740  }
1741  LLVM_DEBUG(llvm::dbgs()
1742  << "lb map for pos = " << Twine(pos + offset) << ", expr: ");
1743  LLVM_DEBUG(lbMap.dump(););
1744  LLVM_DEBUG(llvm::dbgs()
1745  << "ub map for pos = " << Twine(pos + offset) << ", expr: ");
1746  LLVM_DEBUG(ubMap.dump(););
1747  }

1750 LogicalResult
1751 FlatAffineConstraints::addLowerOrUpperBound(unsigned pos, AffineMap boundMap,
1752  ArrayRef<Value> boundOperands,
1753  bool eq, bool lower) {
1754  assert(pos < getNumDimAndSymbolIds() && "invalid position");
1755  // Equality follows the logic of lower bound except that we add an equality
1756  // instead of an inequality.
1757  assert((!eq || boundMap.getNumResults() == 1) && "single result expected");
1758  if (eq)
1759  lower = true;
1760 
1761  // Fully compose map and operands; canonicalize and simplify so that we
1762  // transitively get to terminal symbols or loop IVs.
1763  auto map = boundMap;
1764  SmallVector<Value, 4> operands(boundOperands.begin(), boundOperands.end());
1765  fullyComposeAffineMapAndOperands(&map, &operands);
1766  map = simplifyAffineMap(map);
1767  canonicalizeMapAndOperands(&map, &operands);
1768  for (auto operand : operands)
1769  addInductionVarOrTerminalSymbol(operand);
1770 
1771  FlatAffineConstraints localVarCst;
1772  std::vector<SmallVector<int64_t, 8>> flatExprs;
1773  if (failed(getFlattenedAffineExprs(map, &flatExprs, &localVarCst))) {
1774  LLVM_DEBUG(llvm::dbgs() << "semi-affine expressions not yet supported\n");
1775  return failure();
1776  }
1777 
1778  // Merge and align with localVarCst.
1779  if (localVarCst.getNumLocalIds() > 0) {
1780  // Set values for localVarCst.
1781  localVarCst.setIdValues(0, localVarCst.getNumDimAndSymbolIds(), operands);
1782  for (auto operand : operands) {
1783  unsigned pos;
1784  if (findId(operand, &pos)) {
1785  if (pos >= getNumDimIds() && pos < getNumDimAndSymbolIds()) {
1786  // If the local var cst has this as a dim, turn it into its symbol.
1787  turnDimIntoSymbol(&localVarCst, operand);
1788  } else if (pos < getNumDimIds()) {
1789  // Or vice versa.
1790  turnSymbolIntoDim(&localVarCst, operand);
1791  }
1792  }
1793  }
1794  mergeAndAlignIds(/*offset=*/0, this, &localVarCst);
1795  append(localVarCst);
1796  }
1797 
1798  // Record positions of the operands in the constraint system. Need to do
1799  // this here since the constraint system changes after a bound is added.
1800  SmallVector<unsigned, 8> positions;
1801  unsigned numOperands = operands.size();
1802  for (auto operand : operands) {
1803  unsigned pos;
1804  if (!findId(operand, &pos))
1805  assert(0 && "expected to be found");
1806  positions.push_back(pos);
1807  }
1808 
1809  for (const auto &flatExpr : flatExprs) {
1810  SmallVector<int64_t, 4> ineq(getNumCols(), 0);
1811  ineq[pos] = lower ? 1 : -1;
1812  // Dims and symbols.
1813  for (unsigned j = 0, e = map.getNumInputs(); j < e; j++) {
1814  ineq[positions[j]] = lower ? -flatExpr[j] : flatExpr[j];
1815  }
1816  // Copy over the local id coefficients.
1817  unsigned numLocalIds = flatExpr.size() - 1 - numOperands;
1818  for (unsigned jj = 0, j = getNumIds() - numLocalIds; jj < numLocalIds;
1819  jj++, j++) {
1820  ineq[j] =
1821  lower ? -flatExpr[numOperands + jj] : flatExpr[numOperands + jj];
1822  }
1823  // Constant term.
1824  ineq[getNumCols() - 1] =
1825  lower ? -flatExpr[flatExpr.size() - 1]
1826  // Upper bound in flattenedExpr is an exclusive one.
1827  : flatExpr[flatExpr.size() - 1] - 1;
1828  eq ? addEquality(ineq) : addInequality(ineq);
1829  }
1830  return success();
1831 }

1833 // Adds slice lower bounds represented by lower bounds in 'lbMaps' and upper
1834 // bounds in 'ubMaps' to each value in `values' that appears in the constraint
1835 // system. Note that both lower/upper bounds share the same operand list
1836 // 'operands'.
1837 // This function assumes 'values.size' == 'lbMaps.size' == 'ubMaps.size', and
1838 // skips any null AffineMaps in 'lbMaps' or 'ubMaps'.
1839 // Note that both lower/upper bounds use operands from 'operands'.
1840 // Returns failure for unimplemented cases such as semi-affine expressions or
1841 // expressions with mod/floordiv.
1842 LogicalResult FlatAffineConstraints::addSliceBounds(ArrayRef<Value> values,
1843  ArrayRef<AffineMap> lbMaps,
1844  ArrayRef<AffineMap> ubMaps,
1845  ArrayRef<Value> operands) {
1846  assert(values.size() == lbMaps.size());
1847  assert(lbMaps.size() == ubMaps.size());
1848 
1849  for (unsigned i = 0, e = lbMaps.size(); i < e; ++i) {
1850  unsigned pos;
1851  if (!findId(values[i], &pos))
1852  continue;
1853 
1854  AffineMap lbMap = lbMaps[i];
1855  AffineMap ubMap = ubMaps[i];
1856  assert(!lbMap || lbMap.getNumInputs() == operands.size());
1857  assert(!ubMap || ubMap.getNumInputs() == operands.size());

1859  // Check if this slice is just an equality along this dimension.
1860  if (lbMap && ubMap && lbMap.getNumResults() == 1 &&
1861  ubMap.getNumResults() == 1 &&
1862  lbMap.getResult(0) + 1 == ubMap.getResult(0)) {

2084 /// Returns the extent (upper bound - lower bound) of the specified
2085 /// identifier if it is found to be a constant; returns None if it's not a
2086 /// constant. This methods treats symbolic identifiers specially, i.e.,
2087 /// it looks for constant differences between affine expressions involving
2088 /// only the symbolic identifiers. See comments at function definition for
2089 /// example. 'lb', if provided, is set to the lower bound associated with the
2090 /// constant difference. Note that 'lb' is purely symbolic and thus will contain
2091 /// the coefficients of the symbolic identifiers and the constant coefficient.
2092 // Egs: 0 <= i <= 15, return 16.
2093 // s0 + 2 <= i <= s0 + 17, returns 16. (s0 has to be a symbol)
2094 // s0 + s1 + 16 <= d0 <= s0 + s1 + 31, returns 16.
2095 // s0 - 7 <= 8*j <= s0 returns 1 with lb = s0, lbDivisor = 8 (since lb =
2096 // ceil(s0 - 7 / 8) = floor(s0 / 8)).
2097 Optional<int64_t> FlatAffineConstraints::getConstantBoundOnDimSize(
2098  unsigned pos, SmallVectorImpl<int64_t> *lb, int64_t *lbFloorDivisor,
2099  SmallVectorImpl<int64_t> *ub) const {
2100  assert(pos < getNumDimIds() && "Invalid identifier position");
2101  assert(getNumLocalIds() == 0);
2102 
2103  // TODO(bondhugula): eliminate all remaining dimensional identifiers (other
2104  // than the one at 'pos' to make this more powerful. Not needed for
2105  // hyper-rectangular spaces.
2106 
2107  // Find an equality for 'pos'^th identifier that equates it to some function
2108  // of the symbolic identifiers (+ constant).
2109  int eqRow = findEqualityToConstant(*this, pos, /*symbolic=*/true);
2110  if (eqRow != -1) {
2111  // This identifier can only take a single value.
2112  if (lb) {
2113  // Set lb to the symbolic value.
2114  lb->resize(getNumSymbolIds() + 1);
2115  if (ub)
2116  ub->resize(getNumSymbolIds() + 1);
2117  for (unsigned c = 0, f = getNumSymbolIds() + 1; c < f; c++) {
2118  int64_t v = atEq(eqRow, pos);
2119  // atEq(eqRow, pos) is either -1 or 1.
2120  assert(v * v == 1);
2121  (*lb)[c] = v < 0 ? atEq(eqRow, getNumDimIds() + c) / -v
2122  : -atEq(eqRow, getNumDimIds() + c) / v;
2123  // Since this is an equality, ub = lb.
2124  if (ub)
2125  (*ub)[c] = (*lb)[c];
2126  }
2127  assert(lbFloorDivisor &&
2128  "both lb and divisor or none should be provided");
2129  *lbFloorDivisor = 1;
2130  }
2131  return 1;
2132  }
2133 
2134  // Check if the identifier appears at all in any of the inequalities.
2135  unsigned r, e;
2136  for (r = 0, e = getNumInequalities(); r < e; r++) {
2137  if (atIneq(r, pos) != 0)
2138  break;
2139  }
2140  if (r == e)
2141  // If it doesn't, there isn't a bound on it.
2142  return None;
2143 
2144  // Positions of constraints that are lower/upper bounds on the variable.
2145  SmallVector<unsigned, 4> lbIndices, ubIndices;
2146 
2147  // Gather all symbolic lower bounds and upper bounds of the variable. Since
2148  // the canonical form c_1*x_1 + c_2*x_2 + ... + c_0 >= 0, a constraint is a
2149  // lower bound for x_i if c_i >= 1, and an upper bound if c_i <= -1.
2150  for (unsigned r = 0, e = getNumInequalities(); r < e; r++) {
2151  unsigned c, f;
2152  for (c = 0, f = getNumDimIds(); c < f; c++) {
2153  if (c != pos && atIneq(r, c) != 0)
2154  break;
2155  }
2156  if (c < getNumDimIds())
2157  // Not a pure symbolic bound.
2158  continue;
2159  if (atIneq(r, pos) >= 1)
2160  // Lower bound.
2161  lbIndices.push_back(r);
2162  else if (atIneq(r, pos) <= -1)
2163  // Upper bound.
2164  ubIndices.push_back(r);
2165  }
2166 
2167  // TODO(bondhugula): eliminate other dimensional identifiers to make this more
2168  // powerful. Not needed for hyper-rectangular iteration spaces.
2169 
2170  Optional<int64_t> minDiff = None;
2171  unsigned minLbPosition, minUbPosition;
2172  for (auto ubPos : ubIndices) {
2173  for (auto lbPos : lbIndices) {
2174  // Look for a lower bound and an upper bound that only differ by a
2175  // constant, i.e., pairs of the form 0 <= c_pos - f(c_i's) <= diffConst.
2176  // For example, if ii is the pos^th variable, we are looking for
2177  // constraints like ii >= i, ii <= ii + 50, 50 being the difference. The
2178  // minimum among all such constant differences is kept since that's the
2179  // constant bounding the extent of the pos^th variable.
2180  unsigned j, e;
2181  for (j = 0, e = getNumCols() - 1; j < e; j++)
2182  if (atIneq(ubPos, j) != -atIneq(lbPos, j)) {
2183  break;
2184  }
2185  if (j < getNumCols() - 1)
2186  continue;
2187  int64_t diff = ceilDiv(atIneq(ubPos, getNumCols() - 1) +
2188  atIneq(lbPos, getNumCols() - 1) + 1,
2189  atIneq(lbPos, pos));
2190  if (minDiff == None || diff < minDiff) {
2191  minDiff = diff;
2192  minLbPosition = lbPos;
2193  minUbPosition = ubPos;
2194  }
2195  }
2196  }
2197  if (lb && minDiff.hasValue()) {
2198  // Set lb to the symbolic lower bound.

2459 /// Eliminates identifier at the specified position using Fourier-Motzkin
2460 /// variable elimination. This technique is exact for rational spaces but
2461 /// conservative (in "rare" cases) for integer spaces. The operation corresponds
2462 /// to a projection operation yielding the (convex) set of integer points
2463 /// contained in the rational shadow of the set. An emptiness test that relies
2464 /// on this method will guarantee emptiness, i.e., it disproves the existence of
2465 /// a solution if it says it's empty.
2466 /// If a non-null isResultIntegerExact is passed, it is set to true if the
2467 /// result is also integer exact. If it's set to false, the obtained solution
2468 /// *may* not be exact, i.e., it may contain integer points that do not have an
2469 /// integer pre-image in the original set.
2470 ///
2471 /// Eg:
2472 /// j >= 0, j <= i + 1
2473 /// i >= 0, i <= N + 1
2474 /// Eliminating i yields,
2475 /// j >= 0, 0 <= N + 1, j - 1 <= N + 1
2476 ///
2477 /// If darkShadow = true, this method computes the dark shadow on elimination;
2478 /// the dark shadow is a convex integer subset of the exact integer shadow. A
2479 /// non-empty dark shadow proves the existence of an integer solution. The
2480 /// elimination in such a case could however be an under-approximation, and thus
2481 /// should not be used for scanning sets or used by itself for dependence
2482 /// checking.
2483 ///
2484 /// Eg: 2-d set, * represents grid points, 'o' represents a point in the set.
2485 /// ^
2486 /// |
2487 /// | * * * * o o
2488 /// i | * * o o o o
2489 /// | o * * * * *
2490 /// --------------->
2491 /// j ->
2492 ///
2493 /// Eliminating i from this system (projecting on the j dimension):
2494 /// rational shadow / integer light shadow: 1 <= j <= 6
2495 /// dark shadow: 3 <= j <= 6
2496 /// exact integer shadow: j = 1 \union 3 <= j <= 6
2497 /// holes/splinters: j = 2
2498 ///
2499 /// darkShadow = false, isResultIntegerExact = nullptr are default values.
