68 std::unique_ptr<llvm::Module> mlir::translateModuleToNVVMIR(Operation *m) {
69  ModuleTranslation translation(m);
70  auto llvmModule =
71  LLVM::ModuleTranslation::translateModule<ModuleTranslation>(m);
72  if (!llvmModule)
73  return llvmModule;
74 
75  // Insert the nvvm.annotations kernel so that the NVVM backend recognizes the
76  // function as a kernel.
77  for (auto func :
78  ModuleTranslation::getModuleBody(m).getOps<LLVM::LLVMFuncOp>()) {
79  if (!gpu::GPUDialect::isKernel(func))
80  continue;
81 
82  auto *llvmFunc = llvmModule->getFunction(func.getName());
83 
84  llvm::Metadata *llvmMetadata[] = {
85  llvm::ValueAsMetadata::get(llvmFunc),
86  llvm::MDString::get(llvmModule->getContext(), "kernel"),
87  llvm::ValueAsMetadata::get(llvm::ConstantInt::get(
88  llvm::Type::getInt32Ty(llvmModule->getContext()), 1))};
89  llvm::MDNode *llvmMetadataNode =
90  llvm::MDNode::get(llvmModule->getContext(), llvmMetadata);
91  llvmModule->getOrInsertNamedMetadata("nvvm.annotations")
92  ->addOperand(llvmMetadataNode);
93  }
94 
95  return llvmModule;
96 }
