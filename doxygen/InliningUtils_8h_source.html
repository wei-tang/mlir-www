36 /// This is the interface that must be implemented by the dialects of operations
37 /// to be inlined. This interface should only handle the operations of the
38 /// given dialect.
39 class DialectInlinerInterface
40  : public DialectInterface::Base<DialectInlinerInterface> {
41 public:
42  DialectInlinerInterface(Dialect *dialect) : Base(dialect) {}

48  /// Returns true if the given region 'src' can be inlined into the region
49  /// 'dest' that is attached to an operation registered to the current dialect.
50  /// 'valueMapping' contains any remapped values from within the 'src' region.
51  /// This can be used to examine what values will replace entry arguments into
52  /// the 'src' region for example.
53  virtual bool isLegalToInline(Region *dest, Region *src,
54  BlockAndValueMapping &valueMapping) const {
55  return false;
56  }
57 
58  /// Returns true if the given operation 'op', that is registered to this
59  /// dialect, can be inlined into the given region, false otherwise.
60  /// 'valueMapping' contains any remapped values from within the 'src' region.
61  /// This can be used to examine what values may potentially replace the
62  /// operands to 'op'.
63  virtual bool isLegalToInline(Operation *op, Region *dest,
64  BlockAndValueMapping &valueMapping) const {
65  return false;
66  }
67 
68  /// This hook is invoked on an operation that contains regions. It should
69  /// return true if the analyzer should recurse within the regions of this
70  /// operation when computing legality and cost, false otherwise. The default
71  /// implementation returns true.
72  virtual bool shouldAnalyzeRecursively(Operation *op) const { return true; }
73 
74  //===--------------------------------------------------------------------===//
75  // Transformation Hooks
76  //===--------------------------------------------------------------------===//
77 
78  /// Handle the given inlined terminator by replacing it with a new operation
79  /// as necessary. This overload is called when the inlined region has more
80  /// than one block. The 'newDest' block represents the new final branching
81  /// destination of blocks within this region, i.e. operations that release
82  /// control to the parent operation will likely now branch to this block.
83  /// Its block arguments correspond to any values that need to be replaced by
84  /// terminators within the inlined region.
85  virtual void handleTerminator(Operation *op, Block *newDest) const {
86  llvm_unreachable("must implement handleTerminator in the case of multiple "
87  "inlined blocks");
88  }
89 
90  /// Handle the given inlined terminator by replacing it with a new operation
91  /// as necessary. This overload is called when the inlined region only
92  /// contains one block. 'valuesToReplace' contains the previously returned
93  /// values of the call site before inlining. These values must be replaced by
94  /// this callback if they had any users (for example for traditional function
95  /// calls, these are directly replaced with the operands of the `return`
96  /// operation). The given 'op' will be removed by the caller, after this
97  /// function has been called.
98  virtual void handleTerminator(Operation *op,
99  ArrayRef<Value> valuesToReplace) const {
100  llvm_unreachable(
101  "must implement handleTerminator in the case of one inlined block");
102  }
103 
104  /// Attempt to materialize a conversion for a type mismatch between a call
105  /// from this dialect, and a callable region. This method should generate an
106  /// operation that takes 'input' as the only operand, and produces a single
107  /// result of 'resultType'. If a conversion can not be generated, nullptr
108  /// should be returned. For example, this hook may be invoked in the following
109  /// scenarios:
110  /// func @foo(i32) -> i32 { ... }
111  ///
112  /// // Mismatched input operand
113  /// ... = foo.call @foo(%input : i16) -> i32
114  ///
115  /// // Mismatched result type.
116  /// ... = foo.call @foo(%input : i32) -> i16
117  ///
118  /// NOTE: This hook may be invoked before the 'isLegal' checks above.
119  virtual Operation *materializeCallConversion(OpBuilder &builder, Value input,
120  Type resultType,
121  Location conversionLoc) const {
122  return nullptr;
123  }
124 };

166 /// This function inlines a region, 'src', into another. This function returns
167 /// failure if it is not possible to inline this function. If the function
168 /// returned failure, then no changes to the module have been made.
169 ///
170 /// The provided 'inlinePoint' must be within a region, and corresponds to the
171 /// location where the 'src' region should be inlined. 'mapping' contains any
172 /// remapped operands that are used within the region, and *must* include
173 /// remappings for the entry arguments to the region. 'resultsToReplace'
174 /// corresponds to any results that should be replaced by terminators within the
175 /// inlined region. 'inlineLoc' is an optional Location that, if provided, will
176 /// be used to update the inlined operations' location information.
177 /// 'shouldCloneInlinedRegion' corresponds to whether the source region should
178 /// be cloned into the 'inlinePoint' or spliced directly.
179 LogicalResult inlineRegion(InlinerInterface &interface, Region *src,
180  Operation *inlinePoint, BlockAndValueMapping &mapper,
181  ArrayRef<Value> resultsToReplace,
182  Optional<Location> inlineLoc = llvm::None,
183  bool shouldCloneInlinedRegion = true);
184 
185 /// This function is an overload of the above 'inlineRegion' that allows for
186 /// providing the set of operands ('inlinedOperands') that should be used
187 /// in-favor of the region arguments when inlining.
188 LogicalResult inlineRegion(InlinerInterface &interface, Region *src,
189  Operation *inlinePoint,
190  ArrayRef<Value> inlinedOperands,
191  ArrayRef<Value> resultsToReplace,
192  Optional<Location> inlineLoc = llvm::None,
193  bool shouldCloneInlinedRegion = true);
194 
195 /// This function inlines a given region, 'src', of a callable operation,
196 /// 'callable', into the location defined by the given call operation. This
197 /// function returns failure if inlining is not possible, success otherwise. On
198 /// failure, no changes are made to the module. 'shouldCloneInlinedRegion'
199 /// corresponds to whether the source region should be cloned into the 'call' or
200 /// spliced directly.
201 LogicalResult inlineCall(InlinerInterface &interface, CallOpInterface call,
202  CallableOpInterface callable, Region *src,
203  bool shouldCloneInlinedRegion = true);
