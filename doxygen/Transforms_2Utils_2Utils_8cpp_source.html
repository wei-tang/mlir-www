49 LogicalResult mlir::replaceAllMemRefUsesWith(Value oldMemRef, Value newMemRef,
50  Operation *op,
51  ArrayRef<Value> extraIndices,
52  AffineMap indexRemap,
53  ArrayRef<Value> extraOperands,
54  ArrayRef<Value> symbolOperands) {
55  unsigned newMemRefRank = newMemRef.getType().cast<MemRefType>().getRank();
56  (void)newMemRefRank; // unused in opt mode
57  unsigned oldMemRefRank = oldMemRef.getType().cast<MemRefType>().getRank();
58  (void)oldMemRefRank; // unused in opt mode
59  if (indexRemap) {
60  assert(indexRemap.getNumSymbols() == symbolOperands.size() &&
61  "symbolic operand count mismatch");
62  assert(indexRemap.getNumInputs() ==
63  extraOperands.size() + oldMemRefRank + symbolOperands.size());
64  assert(indexRemap.getNumResults() + extraIndices.size() == newMemRefRank);
65  } else {
66  assert(oldMemRefRank + extraIndices.size() == newMemRefRank);
67  }
68 
69  // Assert same elemental type.
70  assert(oldMemRef.getType().cast<MemRefType>().getElementType() ==
71  newMemRef.getType().cast<MemRefType>().getElementType());
72 
73  if (!isMemRefDereferencingOp(*op))
74  // Failure: memref used in a non-dereferencing context (potentially
75  // escapes); no replacement in these cases.
76  return failure();
77 
78  SmallVector<unsigned, 2> usePositions;
79  for (const auto &opEntry : llvm::enumerate(op->getOperands())) {
80  if (opEntry.value() == oldMemRef)
81  usePositions.push_back(opEntry.index());
82  }
83 
84  // If memref doesn't appear, nothing to do.
85  if (usePositions.empty())
86  return success();
87 
88  if (usePositions.size() > 1) {
89  // TODO(mlir-team): extend it for this case when needed (rare).
90  assert(false && "multiple dereferencing uses in a single op not supported");
91  return failure();
92  }
93 
94  unsigned memRefOperandPos = usePositions.front();
95 
96  OpBuilder builder(op);
97  NamedAttribute oldMapAttrPair = getAffineMapAttrForMemRef(op, oldMemRef);
98  AffineMap oldMap = oldMapAttrPair.second.cast<AffineMapAttr>().getValue();
99  unsigned oldMapNumInputs = oldMap.getNumInputs();
100  SmallVector<Value, 4> oldMapOperands(
101  op->operand_begin() + memRefOperandPos + 1,
102  op->operand_begin() + memRefOperandPos + 1 + oldMapNumInputs);
103 
104  // Apply 'oldMemRefOperands = oldMap(oldMapOperands)'.
105  SmallVector<Value, 4> oldMemRefOperands;
106  SmallVector<Value, 4> affineApplyOps;
107  oldMemRefOperands.reserve(oldMemRefRank);
108  if (oldMap != builder.getMultiDimIdentityMap(oldMap.getNumDims())) {
109  for (auto resultExpr : oldMap.getResults()) {
110  auto singleResMap = AffineMap::get(oldMap.getNumDims(),
111  oldMap.getNumSymbols(), resultExpr);
112  auto afOp = builder.create<AffineApplyOp>(op->getLoc(), singleResMap,
113  oldMapOperands);
114  oldMemRefOperands.push_back(afOp);
115  affineApplyOps.push_back(afOp);
116  }
117  } else {
118  oldMemRefOperands.append(oldMapOperands.begin(), oldMapOperands.end());
119  }
120 
121  // Construct new indices as a remap of the old ones if a remapping has been
122  // provided. The indices of a memref come right after it, i.e.,
123  // at position memRefOperandPos + 1.
124  SmallVector<Value, 4> remapOperands;
125  remapOperands.reserve(extraOperands.size() + oldMemRefRank +
126  symbolOperands.size());
127  remapOperands.append(extraOperands.begin(), extraOperands.end());
128  remapOperands.append(oldMemRefOperands.begin(), oldMemRefOperands.end());
129  remapOperands.append(symbolOperands.begin(), symbolOperands.end());
130 
131  SmallVector<Value, 4> remapOutputs;
132  remapOutputs.reserve(oldMemRefRank);
133 
134  if (indexRemap &&
135  indexRemap != builder.getMultiDimIdentityMap(indexRemap.getNumDims())) {
136  // Remapped indices.
137  for (auto resultExpr : indexRemap.getResults()) {
138  auto singleResMap = AffineMap::get(
139  indexRemap.getNumDims(), indexRemap.getNumSymbols(), resultExpr);
140  auto afOp = builder.create<AffineApplyOp>(op->getLoc(), singleResMap,
141  remapOperands);
142  remapOutputs.push_back(afOp);
143  affineApplyOps.push_back(afOp);
144  }
145  } else {
146  // No remapping specified.
147  remapOutputs.append(remapOperands.begin(), remapOperands.end());
148  }
149 
150  SmallVector<Value, 4> newMapOperands;
151  newMapOperands.reserve(newMemRefRank);
152 
153  // Prepend 'extraIndices' in 'newMapOperands'.
154  for (auto extraIndex : extraIndices) {
155  assert(extraIndex.getDefiningOp()->getNumResults() == 1 &&
156  "single result op's expected to generate these indices");
157  assert((isValidDim(extraIndex) || isValidSymbol(extraIndex)) &&
158  "invalid memory op index");
159  newMapOperands.push_back(extraIndex);
160  }
161 
162  // Append 'remapOutputs' to 'newMapOperands'.
163  newMapOperands.append(remapOutputs.begin(), remapOutputs.end());
164 
165  // Create new fully composed AffineMap for new op to be created.
166  assert(newMapOperands.size() == newMemRefRank);
167  auto newMap = builder.getMultiDimIdentityMap(newMemRefRank);
168  // TODO(b/136262594) Avoid creating/deleting temporary AffineApplyOps here.
169  fullyComposeAffineMapAndOperands(&newMap, &newMapOperands);
170  newMap = simplifyAffineMap(newMap);
171  canonicalizeMapAndOperands(&newMap, &newMapOperands);
172  // Remove any affine.apply's that became dead as a result of composition.
173  for (auto value : affineApplyOps)
174  if (value.use_empty())
175  value.getDefiningOp()->erase();
176 
177  // Construct the new operation using this memref.
178  OperationState state(op->getLoc(), op->getName());
179  state.setOperandListToResizable(op->hasResizableOperandsList());
180  state.operands.reserve(op->getNumOperands() + extraIndices.size());
181  // Insert the non-memref operands.
182  state.operands.append(op->operand_begin(),
183  op->operand_begin() + memRefOperandPos);
184  // Insert the new memref value.
185  state.operands.push_back(newMemRef);
186 
187  // Insert the new memref map operands.
188  state.operands.append(newMapOperands.begin(), newMapOperands.end());
189 
190  // Insert the remaining operands unmodified.
191  state.operands.append(op->operand_begin() + memRefOperandPos + 1 +
192  oldMapNumInputs,
193  op->operand_end());
194 
195  // Result types don't change. Both memref's are of the same elemental type.
196  state.types.reserve(op->getNumResults());
197  for (auto result : op->getResults())
198  state.types.push_back(result.getType());
199 
200  // Add attribute for 'newMap', other Attributes do not change.
201  auto newMapAttr = AffineMapAttr::get(newMap);
202  for (auto namedAttr : op->getAttrs()) {
203  if (namedAttr.first == oldMapAttrPair.first) {
204  state.attributes.push_back({namedAttr.first, newMapAttr});
205  } else {
206  state.attributes.push_back(namedAttr);
207  }
208  }
209 
210  // Create the new operation.
211  auto *repOp = builder.createOperation(state);
212  op->replaceAllUsesWith(repOp);
213  op->erase();
214 
215  return success();
216 }

218 LogicalResult mlir::replaceAllMemRefUsesWith(Value oldMemRef, Value newMemRef,
219  ArrayRef<Value> extraIndices,
220  AffineMap indexRemap,
221  ArrayRef<Value> extraOperands,
222  ArrayRef<Value> symbolOperands,
223  Operation *domInstFilter,
224  Operation *postDomInstFilter) {
225  unsigned newMemRefRank = newMemRef.getType().cast<MemRefType>().getRank();
226  (void)newMemRefRank; // unused in opt mode
227  unsigned oldMemRefRank = oldMemRef.getType().cast<MemRefType>().getRank();
228  (void)oldMemRefRank;
229  if (indexRemap) {
230  assert(indexRemap.getNumSymbols() == symbolOperands.size() &&
231  "symbol operand count mismatch");
232  assert(indexRemap.getNumInputs() ==
233  extraOperands.size() + oldMemRefRank + symbolOperands.size());
234  assert(indexRemap.getNumResults() + extraIndices.size() == newMemRefRank);
235  } else {
236  assert(oldMemRefRank + extraIndices.size() == newMemRefRank);
237  }
238 
239  // Assert same elemental type.
240  assert(oldMemRef.getType().cast<MemRefType>().getElementType() ==
241  newMemRef.getType().cast<MemRefType>().getElementType());
242 
243  std::unique_ptr<DominanceInfo> domInfo;
244  std::unique_ptr<PostDominanceInfo> postDomInfo;
245  if (domInstFilter)
246  domInfo = std::make_unique<DominanceInfo>(
247  domInstFilter->getParentOfType<FuncOp>());
248 
249  if (postDomInstFilter)
250  postDomInfo = std::make_unique<PostDominanceInfo>(
251  postDomInstFilter->getParentOfType<FuncOp>());
252 
253  // Walk all uses of old memref; collect ops to perform replacement. We use a
254  // DenseSet since an operation could potentially have multiple uses of a
255  // memref (although rare), and the replacement later is going to erase ops.
256  DenseSet<Operation *> opsToReplace;
257  for (auto *op : oldMemRef.getUsers()) {
258  // Skip this use if it's not dominated by domInstFilter.
259  if (domInstFilter && !domInfo->dominates(domInstFilter, op))
260  continue;
261 
262  // Skip this use if it's not post-dominated by postDomInstFilter.
263  if (postDomInstFilter && !postDomInfo->postDominates(postDomInstFilter, op))
264  continue;
265 
266  // Skip dealloc's - no replacement is necessary, and a memref replacement
267  // at other uses doesn't hurt these dealloc's.
268  if (isa<DeallocOp>(op))
269  continue;
270 
271  // Check if the memref was used in a non-dereferencing context. It is fine
272  // for the memref to be used in a non-dereferencing way outside of the
273  // region where this replacement is happening.
274  if (!isMemRefDereferencingOp(*op))
275  // Failure: memref used in a non-dereferencing op (potentially escapes);
276  // no replacement in these cases.
277  return failure();
278 
279  // We'll first collect and then replace --- since replacement erases the op
280  // that has the use, and that op could be postDomFilter or domFilter itself!
281  opsToReplace.insert(op);
282  }
283 
284  for (auto *op : opsToReplace) {
285  if (failed(replaceAllMemRefUsesWith(oldMemRef, newMemRef, op, extraIndices,
286  indexRemap, extraOperands,
287  symbolOperands)))
288  llvm_unreachable("memref replacement guaranteed to succeed here");
289  }
290 
291  return success();
292 }

294 /// Given an operation, inserts one or more single result affine
295 /// apply operations, results of which are exclusively used by this operation
296 /// operation. The operands of these newly created affine apply ops are
297 /// guaranteed to be loop iterators or terminal symbols of a function.
298 ///
299 /// Before
300 ///
301 /// affine.for %i = 0 to #map(%N)
302 /// %idx = affine.apply (d0) -> (d0 mod 2) (%i)
303 /// "send"(%idx, %A, ...)
304 /// "compute"(%idx)
305 ///
306 /// After
307 ///
308 /// affine.for %i = 0 to #map(%N)
309 /// %idx = affine.apply (d0) -> (d0 mod 2) (%i)
310 /// "send"(%idx, %A, ...)
311 /// %idx_ = affine.apply (d0) -> (d0 mod 2) (%i)
312 /// "compute"(%idx_)
313 ///
314 /// This allows applying different transformations on send and compute (for eg.
315 /// different shifts/delays).
316 ///
317 /// Returns nullptr either if none of opInst's operands were the result of an
318 /// affine.apply and thus there was no affine computation slice to create, or if
319 /// all the affine.apply op's supplying operands to this opInst did not have any
320 /// uses besides this opInst; otherwise returns the list of affine.apply
321 /// operations created in output argument `sliceOps`.
322 void mlir::createAffineComputationSlice(
323  Operation *opInst, SmallVectorImpl<AffineApplyOp> *sliceOps) {
324  // Collect all operands that are results of affine apply ops.
325  SmallVector<Value, 4> subOperands;
326  subOperands.reserve(opInst->getNumOperands());
327  for (auto operand : opInst->getOperands())
328  if (isa_and_nonnull<AffineApplyOp>(operand.getDefiningOp()))
329  subOperands.push_back(operand);
330 
331  // Gather sequence of AffineApplyOps reachable from 'subOperands'.
332  SmallVector<Operation *, 4> affineApplyOps;
333  getReachableAffineApplyOps(subOperands, affineApplyOps);
334  // Skip transforming if there are no affine maps to compose.
335  if (affineApplyOps.empty())
336  return;
337 
338  // Check if all uses of the affine apply op's lie only in this op op, in
339  // which case there would be nothing to do.
340  bool localized = true;
341  for (auto *op : affineApplyOps) {
342  for (auto result : op->getResults()) {
343  for (auto *user : result.getUsers()) {
344  if (user != opInst) {
345  localized = false;
346  break;
347  }
348  }
349  }
350  }
351  if (localized)
352  return;
353 
354  OpBuilder builder(opInst);
355  SmallVector<Value, 4> composedOpOperands(subOperands);
356  auto composedMap = builder.getMultiDimIdentityMap(composedOpOperands.size());
357  fullyComposeAffineMapAndOperands(&composedMap, &composedOpOperands);
358 
359  // Create an affine.apply for each of the map results.
360  sliceOps->reserve(composedMap.getNumResults());
361  for (auto resultExpr : composedMap.getResults()) {
362  auto singleResMap = AffineMap::get(composedMap.getNumDims(),
363  composedMap.getNumSymbols(), resultExpr);
364  sliceOps->push_back(builder.create<AffineApplyOp>(
365  opInst->getLoc(), singleResMap, composedOpOperands));
366  }
367 
368  // Construct the new operands that include the results from the composed
369  // affine apply op above instead of existing ones (subOperands). So, they
370  // differ from opInst's operands only for those operands in 'subOperands', for
371  // which they will be replaced by the corresponding one from 'sliceOps'.
372  SmallVector<Value, 4> newOperands(opInst->getOperands());
373  for (unsigned i = 0, e = newOperands.size(); i < e; i++) {
374  // Replace the subOperands from among the new operands.
375  unsigned j, f;
376  for (j = 0, f = subOperands.size(); j < f; j++) {
377  if (newOperands[i] == subOperands[j])
378  break;
379  }
380  if (j < subOperands.size()) {
381  newOperands[i] = (*sliceOps)[j];
382  }
383  }
384  for (unsigned idx = 0, e = newOperands.size(); idx < e; idx++) {
385  opInst->setOperand(idx, newOperands[idx]);
386  }
387 }

389 // TODO: Currently works for static memrefs with a single layout map.
390 LogicalResult mlir::normalizeMemRef(AllocOp allocOp) {
391  MemRefType memrefType = allocOp.getType();
392  unsigned rank = memrefType.getRank();
393  if (rank == 0)
394  return success();
395 
396  auto layoutMaps = memrefType.getAffineMaps();
397  OpBuilder b(allocOp);
398  if (layoutMaps.size() != 1)
399  return failure();
400 
401  AffineMap layoutMap = layoutMaps.front();
402 
403  // Nothing to do for identity layout maps.
404  if (layoutMap == b.getMultiDimIdentityMap(rank))
405  return success();
406 
407  // We don't do any checks for one-to-one'ness; we assume that it is
408  // one-to-one.
409 
410  // TODO: Only for static memref's for now.
411  if (memrefType.getNumDynamicDims() > 0)
412  return failure();
413 
414  // We have a single map that is not an identity map. Create a new memref with
415  // the right shape and an identity layout map.
416  auto shape = memrefType.getShape();
417  FlatAffineConstraints fac(rank, allocOp.getNumSymbolicOperands());
418  for (unsigned d = 0; d < rank; ++d) {
419  fac.addConstantLowerBound(d, 0);
420  fac.addConstantUpperBound(d, shape[d] - 1);
421  }
422 
423  // We compose this map with the original index (logical) space to derive the
424  // upper bounds for the new index space.
425  unsigned newRank = layoutMap.getNumResults();
426  if (failed(fac.composeMatchingMap(layoutMap)))
427  // TODO: semi-affine maps.
428  return failure();
429 
430  // Project out the old data dimensions.
431  fac.projectOut(newRank, fac.getNumIds() - newRank - fac.getNumLocalIds());
432  SmallVector<int64_t, 4> newShape(newRank);
433  for (unsigned d = 0; d < newRank; ++d) {
434  // The lower bound for the shape is always zero.
435  auto ubConst = fac.getConstantUpperBound(d);
436  // For a static memref and an affine map with no symbols, this is always
437  // bounded.
438  assert(ubConst.hasValue() && "should always have an upper bound");
439  if (ubConst.getValue() < 0)
440  // This is due to an invalid map that maps to a negative space.
441  return failure();
442  newShape[d] = ubConst.getValue() + 1;
443  }
444 
445  auto oldMemRef = allocOp.getResult();
446  SmallVector<Value, 4> symbolOperands(allocOp.getSymbolicOperands());
447 
448  auto newMemRefType = MemRefType::get(newShape, memrefType.getElementType(),
449  b.getMultiDimIdentityMap(newRank));
450  auto newAlloc = b.create<AllocOp>(allocOp.getLoc(), newMemRefType);
451 
452  // Replace all uses of the old memref.
453  if (failed(replaceAllMemRefUsesWith(oldMemRef, /*newMemRef=*/newAlloc,
454  /*extraIndices=*/{},
455  /*indexRemap=*/layoutMap,
456  /*extraOperands=*/{},
457  /*symbolOperands=*/symbolOperands))) {
458  // If it failed (due to escapes for example), bail out.
459  newAlloc.erase();
460  return failure();
461  }
462  // Replace any uses of the original alloc op and erase it. All remaining uses
463  // have to be dealloc's; RAMUW above would've failed otherwise.
464  assert(llvm::all_of(oldMemRef.getUsers(),
465  [](Operation *op) { return isa<DeallocOp>(op); }));
466  oldMemRef.replaceAllUsesWith(newAlloc);
467  allocOp.erase();
468  return success();
469 }
