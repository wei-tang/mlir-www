50  SmallVector<Type, 4> inferedReturnTypes;
51  auto retTypeFn = cast<InferTypeOpInterface>(op);

53  op->getOperands(), op->getAttrs(),
54  op->getRegions(), inferedReturnTypes)))
55  return failure();
