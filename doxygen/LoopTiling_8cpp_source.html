95 /// Constructs and sets new loop bounds after tiling for the case of
96 /// hyper-rectangular index sets, where the bounds of one dimension do not
97 /// depend on other dimensions. Bounds of each dimension can thus be treated
98 /// independently, and deriving the new bounds is much simpler and faster
99 /// than for the case of tiling arbitrary polyhedral shapes.
100 static void
101 constructTiledIndexSetHyperRect(MutableArrayRef<AffineForOp> origLoops,
102  MutableArrayRef<AffineForOp> newLoops,
103  ArrayRef<unsigned> tileSizes) {
104  assert(!origLoops.empty());
105  assert(origLoops.size() == tileSizes.size());
106 
107  OpBuilder b(origLoops[0].getOperation());
108  unsigned width = origLoops.size();
109 
110  // Bounds for tile space loops.
111  for (unsigned i = 0; i < width; i++) {
112  auto lbOperands = origLoops[i].getLowerBoundOperands();
113  auto ubOperands = origLoops[i].getUpperBoundOperands();
114  SmallVector<Value, 4> newLbOperands(lbOperands);
115  SmallVector<Value, 4> newUbOperands(ubOperands);
116  newLoops[i].setLowerBound(newLbOperands, origLoops[i].getLowerBoundMap());
117  newLoops[i].setUpperBound(newUbOperands, origLoops[i].getUpperBoundMap());
118  newLoops[i].setStep(tileSizes[i]);
119  }
120  // Bounds for intra-tile loops.
121  for (unsigned i = 0; i < width; i++) {
122  int64_t largestDiv = getLargestDivisorOfTripCount(origLoops[i]);
123  auto mayBeConstantCount = getConstantTripCount(origLoops[i]);
124  // The lower bound is just the tile-space loop.
125  AffineMap lbMap = b.getDimIdentityMap();
126  newLoops[width + i].setLowerBound(
127  /*operands=*/newLoops[i].getInductionVar(), lbMap);
128 
129  // Set the upper bound.
130  if (mayBeConstantCount.hasValue() &&
131  mayBeConstantCount.getValue() < tileSizes[i]) {
132  // Trip count is less than tile size; upper bound is the trip count.
133  auto ubMap = b.getConstantAffineMap(mayBeConstantCount.getValue());
134  newLoops[width + i].setUpperBoundMap(ubMap);
135  } else if (largestDiv % tileSizes[i] != 0) {
136  // Intra-tile loop ii goes from i to min(i + tileSize, ub_i).
137  // Construct the upper bound map; the operands are the original operands
138  // with 'i' (tile-space loop) appended to it. The new upper bound map is
139  // the original one with an additional expression i + tileSize appended.
140  auto ub = origLoops[i].getUpperBound();
141  SmallVector<Value, 4> ubOperands;
142  ubOperands.reserve(ub.getNumOperands() + 1);
143  auto origUbMap = ub.getMap();
144  // Add dim operands from original upper bound.
145  for (unsigned j = 0, e = origUbMap.getNumDims(); j < e; ++j) {
146  ubOperands.push_back(ub.getOperand(j));
147  }
148  // Add dim operand for new loop upper bound.
149  ubOperands.push_back(newLoops[i].getInductionVar());
150  // Add symbol operands from original upper bound.
151  for (unsigned j = 0, e = origUbMap.getNumSymbols(); j < e; ++j) {
152  ubOperands.push_back(ub.getOperand(origUbMap.getNumDims() + j));
153  }
154  SmallVector<AffineExpr, 4> boundExprs;
155  boundExprs.reserve(1 + origUbMap.getNumResults());
156  auto dim = b.getAffineDimExpr(origUbMap.getNumDims());
157  // The new upper bound map is the original one with an additional
158  // expression i + tileSize appended.
159  boundExprs.push_back(dim + tileSizes[i]);
160  boundExprs.append(origUbMap.getResults().begin(),
161  origUbMap.getResults().end());
162  auto ubMap = AffineMap::get(origUbMap.getNumDims() + 1,
163  origUbMap.getNumSymbols(), boundExprs);
164  newLoops[width + i].setUpperBound(/*operands=*/ubOperands, ubMap);
165  } else {
166  // No need of the min expression.
167  auto dim = b.getAffineDimExpr(0);
168  auto ubMap = AffineMap::get(1, 0, dim + tileSizes[i]);
169  newLoops[width + i].setUpperBound(newLoops[i].getInductionVar(), ubMap);
170  }
171  }
172 }

174 /// Tiles the specified band of perfectly nested loops creating tile-space loops
175 /// and intra-tile loops. A band is a contiguous set of loops.
176 // TODO(bondhugula): handle non hyper-rectangular spaces.
177 LogicalResult mlir::tileCodeGen(MutableArrayRef<AffineForOp> band,
178  ArrayRef<unsigned> tileSizes) {
179  assert(!band.empty());
180  assert(band.size() == tileSizes.size() && "Incorrect number of tile sizes");
181 
182  // Check if the supplied for op's are all successively nested.
183  for (unsigned i = 1, e = band.size(); i < e; i++) {
184  assert(band[i].getParentOp() == band[i - 1].getOperation());
185  }
186 
187  auto origLoops = band;
188 
189  AffineForOp rootAffineForOp = origLoops[0];
190  auto loc = rootAffineForOp.getLoc();
191  // Note that width is at least one since band isn't empty.
192  unsigned width = band.size();
193 
194  SmallVector<AffineForOp, 12> newLoops(2 * width);
195  AffineForOp innermostPointLoop;
196 
197  // The outermost among the loops as we add more..
198  auto *topLoop = rootAffineForOp.getOperation();
199 
200  // Add intra-tile (or point) loops.
201  for (unsigned i = 0; i < width; i++) {
202  OpBuilder b(topLoop);
203  // Loop bounds will be set later.
204  auto pointLoop = b.create<AffineForOp>(loc, 0, 0);
205  pointLoop.getBody()->getOperations().splice(
206  pointLoop.getBody()->begin(), topLoop->getBlock()->getOperations(),
207  topLoop);
208  newLoops[2 * width - 1 - i] = pointLoop;
209  topLoop = pointLoop.getOperation();
210  if (i == 0)
211  innermostPointLoop = pointLoop;
212  }
213 
214  // Add tile space loops;
215  for (unsigned i = width; i < 2 * width; i++) {
216  OpBuilder b(topLoop);
217  // Loop bounds will be set later.
218  auto tileSpaceLoop = b.create<AffineForOp>(loc, 0, 0);
219  tileSpaceLoop.getBody()->getOperations().splice(
220  tileSpaceLoop.getBody()->begin(), topLoop->getBlock()->getOperations(),
221  topLoop);
222  newLoops[2 * width - i - 1] = tileSpaceLoop;
223  topLoop = tileSpaceLoop.getOperation();
224  }
225 
226  // Move the loop body of the original nest to the new one.
227  moveLoopBody(origLoops[origLoops.size() - 1], innermostPointLoop);
228 
229  SmallVector<Value, 8> origLoopIVs;
230  extractForInductionVars(band, &origLoopIVs);
231  SmallVector<Optional<Value>, 6> ids(origLoopIVs.begin(), origLoopIVs.end());
232  FlatAffineConstraints cst;
233  getIndexSet(band, &cst);
234 
235  if (!cst.isHyperRectangular(0, width)) {
236  rootAffineForOp.emitError("tiled code generation unimplemented for the "
237  "non-hyperrectangular case");
238  return failure();
239  }
240 
241  constructTiledIndexSetHyperRect(origLoops, newLoops, tileSizes);
242  // In this case, the point loop IVs just replace the original ones.
243  for (unsigned i = 0; i < width; i++) {
244  origLoopIVs[i].replaceAllUsesWith(newLoops[i + width].getInductionVar());
245  }
246 
247  // Erase the old loop nest.
248  rootAffineForOp.erase();
249 
250  return success();
251 }

253 // Identify valid and profitable bands of loops to tile. This is currently just
254 // a temporary placeholder to test the mechanics of tiled code generation.
255 // Returns all maximal outermost perfect loop nests to tile.
256 static void getTileableBands(FuncOp f,
257  std::vector<SmallVector<AffineForOp, 6>> *bands) {
258  // Get maximal perfect nest of 'affine.for' insts starting from root
259  // (inclusive).
260  auto getMaximalPerfectLoopNest = [&](AffineForOp root) {
261  SmallVector<AffineForOp, 6> band;
262  getPerfectlyNestedLoops(band, root);
263  bands->push_back(band);
264  };
265 
266  for (auto &block : f)
267  for (auto &op : block)
268  if (auto forOp = dyn_cast<AffineForOp>(op))
269  getMaximalPerfectLoopNest(forOp);
270 }

272 // Reduce each tile size to the largest divisor of the corresponding trip count
273 // (if the trip count is known).
274 static void adjustToDivisorsOfTripCounts(ArrayRef<AffineForOp> band,
275  SmallVectorImpl<unsigned> *tileSizes) {
276  assert(band.size() == tileSizes->size() && "invalid tile size count");
277  for (unsigned i = 0, e = band.size(); i < e; i++) {
278  unsigned &tSizeAdjusted = (*tileSizes)[i];
279  auto mayConst = getConstantTripCount(band[i]);
280  if (!mayConst.hasValue())
281  continue;
282  // Adjust the tile size to largest factor of the trip count less than
283  // tSize.
284  uint64_t constTripCount = mayConst.getValue();
285  if (constTripCount > 1 && tSizeAdjusted > constTripCount / 2)
286  tSizeAdjusted = constTripCount / 2;
287  while (constTripCount % tSizeAdjusted != 0)
288  tSizeAdjusted--;
289  }
290 }

292 // Returns tile sizes to use. Checks CL options; if none are specified, sets it
293 // based on a simple model that looks at the memory footprint and determines
294 // tile sizes assuming identity accesses / 1:1 tile size proportional footprint
295 // along each of the dimensions being tiled.
296 // TODO(mlir-team): evolve this model. Tile size determination is a large area
297 // to play with in general.
298 void LoopTiling::getTileSizes(ArrayRef<AffineForOp> band,
299  SmallVectorImpl<unsigned> *tileSizes) {
300  if (band.empty())
301  return;
302 
303  tileSizes->resize(band.size());
304 
305  // Use clTileSize for all loops if specified.
306  if (clTileSize.getNumOccurrences() > 0) {
307  std::fill(tileSizes->begin(), tileSizes->end(), clTileSize);
308  return;
309  }
310 
311  // Use clTileSizes and fill them with default tile size if it's short.
312  if (!clTileSizes.empty()) {
313  std::fill(tileSizes->begin(), tileSizes->end(),
314  LoopTiling::kDefaultTileSize);
315  std::copy(clTileSizes.begin(),
316  clTileSizes.begin() + std::min(clTileSizes.size(), band.size()),
317  tileSizes->begin());
318  return;
319  }
320 
321  // The first loop in the band.
322  auto rootForOp = band[0];
323  (void)rootForOp;
324 
325  // Obtain memory footprint and set tile sizes so that a tile fits in
326  // the cache size. This is an approximation with the assumption that the
327  // footprint increases with the tile size linearly in that dimension (i.e.,
328  // assumes one-to-one access function).
329  auto fp = getMemoryFootprintBytes(band[0], 0);
330  if (!fp.hasValue()) {
331  // Fill with default tile sizes if footprint is unknown.
332  std::fill(tileSizes->begin(), tileSizes->end(),
333  LoopTiling::kDefaultTileSize);
334  if (avoidMaxMinBounds)
335  adjustToDivisorsOfTripCounts(band, tileSizes);
336  LLVM_DEBUG(
337  rootForOp.emitWarning("memory footprint unknown: using default tile "
338  "sizes adjusted to trip count divisors"));
339  return;
340  }
341 
342  // Check how many times larger the cache size is when compared to footprint.
343  uint64_t excessFactor = llvm::divideCeil(fp.getValue(), cacheSizeBytes);
344  if (excessFactor <= 1) {
345  // No need of any tiling - set tile size to 1.
346  std::fill(tileSizes->begin(), tileSizes->end(), 1);
347  return;
348  }
349 
350  // Divide all loops equally in an attempt to reduce footprint.
351  // TODO(bondhugula): this is approximate. Ideally, obtain reuse factor /
352  // profitability along each dimension and weight tile sizes based on that as
353  // one possible approach. Or compute a polynomial in tile sizes and solve for
354  // it.
355 
356  // For an n-d tileable band, compute n^th root of the excess.
357  unsigned tSize =
358  static_cast<unsigned>(floorl(std::pow(excessFactor, 1.0 / band.size())));
359  // We'll keep a running product to determine the last tile size better.
360  unsigned cumulProductOfTileSizes = 1;
361  for (unsigned i = 0, e = band.size(); i < e; i++) {
362  if (i < e - 1)
363  (*tileSizes)[i] = tSize;
364  else
365  // Set last tile size to cover the balance.
366  (*tileSizes)[i] = std::max(
367  1U, static_cast<unsigned>(excessFactor / cumulProductOfTileSizes));
368  cumulProductOfTileSizes *= (*tileSizes)[i];
369  }
370  if (avoidMaxMinBounds)
371  adjustToDivisorsOfTripCounts(band, tileSizes);
372 }

374 void LoopTiling::runOnFunction() {
375  // Override cache size if provided on command line.
376  if (clCacheSizeKiB.getNumOccurrences() > 0)
377  cacheSizeBytes = clCacheSizeKiB * 1024;
378 
379  // Bands of loops to tile.
380  std::vector<SmallVector<AffineForOp, 6>> bands;
381  getTileableBands(getFunction(), &bands);
382 
383  for (auto &band : bands) {
384  // Set up tile sizes; fill missing tile sizes at the end with default tile
385  // size or clTileSize if one was provided.
386  SmallVector<unsigned, 6> tileSizes;
387  getTileSizes(band, &tileSizes);
388  if (llvm::DebugFlag) {
389  auto diag = band[0].emitRemark("using tile sizes [");
390  for (auto tSize : tileSizes)
391  diag << tSize << " ";
392  diag << "]\n";
393  }
394  if (failed(tileCodeGen(band, tileSizes)))
395  return signalPassFailure();
396  }
397 }
