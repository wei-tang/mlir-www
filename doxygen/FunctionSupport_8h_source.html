71 /// This trait provides APIs for Ops that behave like functions. In particular:
72 /// - Ops must be symbols, i.e. also have the `Symbol` trait;
73 /// - Ops have a single region with multiple blocks that corresponds to the body
74 /// of the function;
75 /// - the absence of a region corresponds to an external function;
76 /// - leading arguments of the first block of the region are treated as function
77 /// arguments;
78 /// - they can have argument attributes that are stored in a dictionary
79 /// attribute on the Op itself.
80 /// This trait does *NOT* provide type support for the functions, meaning that
81 /// concrete Ops must handle the type of the declared or defined function.
82 /// `getTypeAttrName()` is a convenience function that returns the name of the
83 /// attribute that can be used to store the function type, but the trait makes
84 /// no assumption based on it.
85 ///
86 /// - Concrete ops *must* define a member function `getNumFuncArguments()` that
87 /// returns the number of function arguments based exclusively on type (so
88 /// that it can be called on function declarations).
89 /// - Concrete ops *must* define a member function `getNumFuncResults()` that
90 /// returns the number of function results based exclusively on type (so that
91 /// it can be called on function declarations).
92 /// - To verify that the type respects op-specific invariants, concrete ops may
93 /// redefine the `verifyType()` hook that will be called after verifying the
94 /// presence of the `type` attribute and before any call to
95 /// `getNumFuncArguments`/`getNumFuncResults` from the verifier.
96 /// - To verify that the body respects op-specific invariants, concrete ops may
97 /// redefine the `verifyBody()` hook that will be called after verifying the
98 /// function type and the presence of the (potentially empty) body region.
99 template <typename ConcreteType>

351 /// Default verifier checks that if the entry block exists, it has the same
352 /// number of arguments as the function-like operation.
353 template <typename ConcreteType>
354 LogicalResult FunctionLike<ConcreteType>::verifyBody() {
355  auto funcOp = cast<ConcreteType>(this->getOperation());
356 
357  if (funcOp.isExternal())
358  return success();
359 
360  unsigned numArguments = funcOp.getNumArguments();
361  if (funcOp.front().getNumArguments() != numArguments)
362  return funcOp.emitOpError("entry block must have ")
363  << numArguments << " arguments to match function signature";
364 
365  return success();
366 }

368 template <typename ConcreteType>
369 LogicalResult FunctionLike<ConcreteType>::verifyTrait(Operation *op) {
370  MLIRContext *ctx = op->getContext();
371  auto funcOp = cast<ConcreteType>(op);
372 
373  if (!funcOp.isTypeAttrValid())
374  return funcOp.emitOpError("requires a type attribute '")
375  << getTypeAttrName() << '\'';
376 
377  if (failed(funcOp.verifyType()))
378  return failure();
379 
380  for (unsigned i = 0, e = funcOp.getNumArguments(); i != e; ++i) {
381  // Verify that all of the argument attributes are dialect attributes, i.e.
382  // that they contain a dialect prefix in their name. Call the dialect, if
383  // registered, to verify the attributes themselves.
384  for (auto attr : funcOp.getArgAttrs(i)) {
385  if (!attr.first.strref().contains('.'))
386  return funcOp.emitOpError("arguments may only have dialect attributes");
387  auto dialectNamePair = attr.first.strref().split('.');
388  if (auto *dialect = ctx->getRegisteredDialect(dialectNamePair.first)) {
389  if (failed(dialect->verifyRegionArgAttribute(op, /*regionIndex=*/0,
390  /*argIndex=*/i, attr)))
391  return failure();
392  }
393  }
394  }
395 
396  for (unsigned i = 0, e = funcOp.getNumResults(); i != e; ++i) {
397  // Verify that all of the result attributes are dialect attributes, i.e.
398  // that they contain a dialect prefix in their name. Call the dialect, if
399  // registered, to verify the attributes themselves.
400  for (auto attr : funcOp.getResultAttrs(i)) {
401  if (!attr.first.strref().contains('.'))
402  return funcOp.emitOpError("results may only have dialect attributes");
403  auto dialectNamePair = attr.first.strref().split('.');
404  if (auto *dialect = ctx->getRegisteredDialect(dialectNamePair.first)) {
405  if (failed(dialect->verifyRegionResultAttribute(op, /*regionIndex=*/0,
406  /*resultIndex=*/i,
407  attr)))
408  return failure();
409  }
410  }
411  }
412 
413  // Check that the op has exactly one region for the body.
414  if (op->getNumRegions() != 1)
415  return funcOp.emitOpError("expects one region");
416 
417  return funcOp.verifyBody();
418 }
