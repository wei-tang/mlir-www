34 // Deserializes the SPIR-V binary module stored in the file named as
35 // `inputFilename` and returns a module containing the SPIR-V module.
36 static OwningModuleRef deserializeModule(const llvm::MemoryBuffer *input,
37  MLIRContext *context) {
38  Builder builder(context);
39 
40  // Make sure the input stream can be treated as a stream of SPIR-V words
41  auto start = input->getBufferStart();
42  auto size = input->getBufferSize();
43  if (size % sizeof(uint32_t) != 0) {
44  emitError(UnknownLoc::get(context))
45  << "SPIR-V binary module must contain integral number of 32-bit words";
46  return {};
47  }
48 
49  auto binary = llvm::makeArrayRef(reinterpret_cast<const uint32_t *>(start),
50  size / sizeof(uint32_t));
51 
52  auto spirvModule = spirv::deserialize(binary, context);
53  if (!spirvModule)
54  return {};
55 
56  OwningModuleRef module(ModuleOp::create(FileLineColLoc::get(
57  input->getBufferIdentifier(), /*line=*/0, /*column=*/0, context)));
58  module->getBody()->push_front(spirvModule->getOperation());
59 
60  return module;
61 }

74 static LogicalResult serializeModule(ModuleOp module, raw_ostream &output) {
75  if (!module)
76  return failure();
77 
78  SmallVector<uint32_t, 0> binary;
79 
80  SmallVector<spirv::ModuleOp, 1> spirvModules;
81  module.walk([&](spirv::ModuleOp op) { spirvModules.push_back(op); });
82 
83  if (spirvModules.empty())
84  return module.emitError("found no 'spv.module' op");
85 
86  if (spirvModules.size() != 1)
87  return module.emitError("found more than one 'spv.module' op");
88 
89  if (failed(spirv::serialize(spirvModules[0], binary)))
90  return failure();
91 
92  output.write(reinterpret_cast<char *>(binary.data()),
93  binary.size() * sizeof(uint32_t));
94 
95  return mlir::success();
96 }

107 static LogicalResult roundTripModule(llvm::SourceMgr &sourceMgr,
108  raw_ostream &output,
109  MLIRContext *context) {
110  // Parse an MLIR module from the source manager.
111  auto srcModule = OwningModuleRef(parseSourceFile(sourceMgr, context));
112  if (!srcModule)
113  return failure();
114 
115  SmallVector<uint32_t, 0> binary;
116 
117  auto spirvModules = srcModule->getOps<spirv::ModuleOp>();
118 
119  if (spirvModules.begin() == spirvModules.end())
120  return srcModule->emitError("found no 'spv.module' op");
121 
122  if (std::next(spirvModules.begin()) != spirvModules.end())
123  return srcModule->emitError("found more than one 'spv.module' op");
124 
125  if (failed(spirv::serialize(*spirvModules.begin(), binary)))
126  return failure();
127 
128  // Then deserialize to get back a SPIR-V module.
129  auto spirvModule = spirv::deserialize(binary, context);
130  if (!spirvModule)
131  return failure();
132 
133  // Wrap around in a new MLIR module.
134  OwningModuleRef dstModule(ModuleOp::create(FileLineColLoc::get(
135  /*filename=*/"", /*line=*/0, /*column=*/0, context)));
136  dstModule->getBody()->push_front(spirvModule->getOperation());
137  dstModule->print(output);
138 
139  return mlir::success();
140 }
