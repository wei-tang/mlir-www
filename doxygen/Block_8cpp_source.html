69 /// Returns 'op' if 'op' lies in this block, or otherwise finds the
70 /// ancestor operation of 'op' that lies in this block. Returns nullptr if
71 /// the latter fails.
72 Operation *Block::findAncestorOpInBlock(Operation &op) {
73  // Traverse up the operation hierarchy starting from the owner of operand to
74  // find the ancestor operation that resides in the block of 'forOp'.
75  auto *currOp = &op;
76  while (currOp->getBlock() != this) {
77  currOp = currOp->getParentOp();
78  if (!currOp)
79  return nullptr;
80  }
81  return currOp;
82 }

111 /// Verifies the current ordering of child operations. Returns false if the
112 /// order is valid, true otherwise.
113 bool Block::verifyOpOrder() {
114  // The order is already known to be invalid.
115  if (!isOpOrderValid())
116  return false;
117  // The order is valid if there are less than 2 operations.
118  if (operations.empty() || std::next(operations.begin()) == operations.end())
119  return false;
120 
121  Operation *prev = nullptr;
122  for (auto &i : *this) {
123  // The previous operation must have a smaller order index than the next as
124  // it appears earlier in the list.
125  if (prev && prev->orderIndex != Operation::kInvalidOrderIdx &&
126  prev->orderIndex >= i.orderIndex)
127  return true;
128  prev = &i;
129  }
130  return false;
131 }

134 void Block::recomputeOpOrder() {

170 void Block::eraseArgument(unsigned index, bool updatePredTerms) {
171  assert(index < arguments.size());
172 
173  // Delete the argument.
174  arguments[index].destroy();
175  arguments.erase(arguments.begin() + index);
176 
177  // If we aren't updating predecessors, there is nothing left to do.
178  if (!updatePredTerms)
179  return;
180 
181  // Erase this argument from each of the predecessor's terminator.
182  for (auto predIt = pred_begin(), predE = pred_end(); predIt != predE;
183  ++predIt) {
184  auto *predTerminator = (*predIt)->getTerminator();
185  predTerminator->eraseSuccessorOperand(predIt.getSuccessorIndex(), index);
186  }
187 }

189 /// Insert one value to the given position of the argument list. The existing
190 /// arguments are shifted. The block is expected not to have predecessors.
191 BlockArgument Block::insertArgument(args_iterator it, Type type) {
192  assert(llvm::empty(getPredecessors()) &&
193  "cannot insert arguments to blocks with predecessors");
194 
195  // Use the args_iterator (on the BlockArgListType) to compute the insertion
196  // iterator in the underlying argument storage.
197  size_t distance = std::distance(args_begin(), it);
198  auto arg = BlockArgument::create(type, this);
199  arguments.insert(std::next(arguments.begin(), distance), arg);
200  return arg;
201 }

227 /// If this block has exactly one predecessor, return it. Otherwise, return
228 /// null.
229 ///
230 /// Note that multiple edges from a single block (e.g. if you have a cond
231 /// branch with the same block as the true/false destinations) is not

246 /// Split the block into two blocks before the specified operation or
247 /// iterator.
248 ///
249 /// Note that all operations BEFORE the specified iterator stay as part of
250 /// the original basic block, and the rest of the operations in the original
251 /// block are moved to the new block, including the old terminator. The
252 /// original block is left without a terminator.
253 ///
254 /// The newly formed Block is returned, and the specified iterator is
255 /// invalidated.
256 Block *Block::splitBlock(iterator splitBefore) {
257  // Start by creating a new basic block, and insert it immediate after this
258  // one in the containing region.
259  auto newBB = new Block();
260  getParent()->getBlocks().insert(std::next(Region::iterator(this)), newBB);
261 
262  // Move all of the operations from the split point to the end of the region
263  // into the new block.
264  newBB->getOperations().splice(newBB->end(), getOperations(), splitBefore,
265  end());
266  return newBB;
267 }
